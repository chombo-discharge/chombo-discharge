150 
151  // Linearize m_scalars onto the buffer
152  for (size_t i = 0; i < M; i++) {
153  *buffer++ = m_scalars[i];
154  }
155 
156  // Linearize vectors onto the buffer
157  for (size_t i = 0; i < N; i++) {
158  const RealVect& v = m_vectors[i];
159 
160  *buffer++ = v[0];
161  *buffer++ = v[1];
162 #if CH_SPACEDIM == 3
163  *buffer++ = v[2];
164 #endif
165  }
166 
167  // Linearize mass onto buffer
168  *buffer = m_mass;
