60inline std::vector<std::pair<Box, int>>
61ScanShop::getSortedBoxesAndTypes(const Vector<Box>& a_boxes, const Vector<int>& a_types) const
62{
63
64 std::vector<std::pair<Box, int>> sortedBoxesAndTypes;
65
66 for (int i = 0; i < a_boxes.size(); i++) {
67 sortedBoxesAndTypes.emplace_back(std::make_pair(a_boxes[i], a_types[i]));
68 }
69
70 auto comparator = [](const std::pair<Box, int>& a, const std::pair<Box, int>& b) -> bool {
71 return a.first < b.first;
72 };
73
74 std::sort(sortedBoxesAndTypes.begin(), sortedBoxesAndTypes.end(), comparator);

76 return sortedBoxesAndTypes;
