53 template <class T>
54 void
55 LoadBalancing::balanceHierarchy(Vector<Vector<int>>& a_procs,
56  const Vector<Vector<T>>& a_loads,
57  const Vector<Vector<Box>>& a_boxes)
58 {
59  CH_TIME("LoadBalancing::balanceHierarchy");
60 
61  const int nlevels = a_procs.size();
62 
63  std::map<std::pair<int, int>, int> boxmap;
64 
65  Vector<Box> allBoxes;
66  Vector<T> allLoads;
67  Vector<int> allProcs;
68 
69  int gbox = 0;
70 
71  // Make the hierarchy to level mapping
72  for (int lvl = 0; lvl < nlevels; lvl++) {
73  for (int ibox = 0; ibox < a_boxes[lvl].size(); ibox++) {
74 
75  allBoxes.push_back(a_boxes[lvl][ibox]);
76  allLoads.push_back(a_loads[lvl][ibox]);
77 
78  boxmap.emplace(std::pair<int, int>(lvl, ibox), gbox);
79 
80  gbox++;
81  }
82  }
83 
84  // Do global load balancing
85  LoadBalancing::makeBalance(allProcs, allLoads, allBoxes);
86 
87  // Resize the a_procs vector to correct size.
88  a_procs.resize(nlevels);
89  for (int lvl = 0; lvl < nlevels; lvl++) {
90  a_procs[lvl].resize(a_boxes[lvl].size(), -1);
91  }
92 
93  // Do the global to local mapping
94  for (const auto& bmap : boxmap) {
95  const int gid = bmap.second; // Global index
96  const int lvl = bmap.first.first; // Current level
97  const int lid = bmap.first.second; // Local box index on level lvl
98  const int pid = allProcs[gid]; // Processor ownership
99 
100  a_procs[lvl][lid] = pid;
101  }
102 }

183 
184  std::vector<std::pair<Box, T>> vec = packPairs(a_boxes, a_loads);
185 
186  // Call std::sort, using box1 < box2 lambda as sorting criterion.
187  std::sort(std::begin(vec), std::end(vec), [](const std::pair<Box, T>& v1, const std::pair<Box, T>& v2) {
188  return v1.first < v2.first;
189  });
190

217 template <class T>
218 void
219 LoadBalancing::mortonSort(Vector<Box>& a_boxes, Vector<T>& a_loads)
220 {
221  CH_TIME("LoadBalancing::mortonSort");
222 
223  auto vec = packPairs(a_boxes, a_loads);
224 
225  // Get max bits
226  std::vector<Box>& b = a_boxes.stdVector();
227  int bits = maxBits(b.begin(), b.end());
228 
229  // Morton sort.
230  std::sort(std::begin(vec), std::end(vec), [bits](const std::pair<Box, T>& v1, const std::pair<Box, T>& v2) -> bool {
231  return mortonComparator(bits, v1, v2);
232  });
233 
234  // Put back in normal form
235  unpackPairs(a_boxes, a_loads, vec);
236 }

238 template <class T>
239 bool
240 LoadBalancing::mortonComparator(const int a_maxBits, const std::pair<Box, T>& a_lhs, const std::pair<Box, T>& a_rhs)
241 {
242  const Box& lbox = a_lhs.first;
243  const Box& rbox = a_rhs.first;
244 
245  const IntVect l = lbox.smallEnd();
246  const IntVect r = rbox.smallEnd();
247 
248  for (int i = a_maxBits; i > 0; i--) {
249 
250  // March from most significant bit to least.
251  const int N = (1 << i);
252 
253  for (int dir = CH_SPACEDIM - 1; dir >= 0; dir--) {
254  if ((l[dir] / N) < (r[dir] / N)) {
255  return true;
256  }
257  else if ((l[dir] / N) > (r[dir] / N)) {
258  return false;
259  }
260  }
261  }
262 
263  return false;
264 }
