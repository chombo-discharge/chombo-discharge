28inline void
29Random::seed()
30{
31 if (!s_seeded) {
32 ParmParse pp("Random");
33
34 if (pp.contains("seed")) {
35 int seed;
36 pp.get("seed", seed);
37 if (seed < 0) {
38 Random::setRandomSeed();
39 }
40 else {
41 Random::setSeed(seed);
42 }
43 }
44 else {
45 Random::setSeed(0);
46 }
47
48 s_seeded = true;
49 }
50}

99template <typename T, typename>
100inline T
101Random::getPoisson(const Real a_mean)
102{
103 CH_TIME("Random::getPoisson");
104
105 CH_assert(s_seeded);
106
107 T ret = (T)0;
108
109 if (a_mean < 250.0) {
110 std::poisson_distribution<T> poisson(a_mean);
111
112 ret = poisson(s_rng);
113 }
114 else {
115 std::normal_distribution<Real> normal(a_mean, sqrt(a_mean));
116
117 ret = (T)std::max(normal(s_rng), (Real)0.0);
118 }
119
120 return ret;
121}

123template <typename T, typename>
124inline T
125Random::getBinomial(const T a_N, const Real a_p) noexcept
126{
127 CH_TIME("Random::getBinomial");
128
129 CH_assert(s_seeded);
130
131 T ret = 0;
132
133 const bool useNormalApprox = (a_N > (T)9.0 * ((1.0 - a_p) / a_p)) && (a_N > (T)9.0 * a_p / (1.0 - a_p));
134
135 if (useNormalApprox) {
136 const Real mean = a_N * a_p;
137
138 std::normal_distribution<Real> normalDist(mean, mean * (1.0 - a_p));
139
140 ret = (T)std::max(normalDist(s_rng), (Real)0.0);
141 }
142 else {
143 std::binomial_distribution<T> binomDist(a_N, a_p);
144
145 ret = binomDist(s_rng);
146 }
147
148 return ret;
149}

181inline RealVect
182Random::getDirection()
183{
184 CH_TIME("Random::getDirection");
185
186 CH_assert(s_seeded);
187
188 constexpr Real safety = 1.E-12;
189
190#if CH_SPACEDIM == 2
191 Real x1 = 2.0;
192 Real x2 = 2.0;
193 Real r = x1 * x1 + x2 * x2;
194 while (r >= 1.0 || r < safety) {
195 x1 = Random::getUniformReal11();
196 x2 = Random::getUniformReal11();
197 r = x1 * x1 + x2 * x2;
198 }
199
200 return RealVect(x1, x2) / sqrt(r);
201#elif CH_SPACEDIM == 3
202 Real x1 = 2.0;
203 Real x2 = 2.0;
204 Real r = x1 * x1 + x2 * x2;
205 while (r >= 1.0 || r < safety) {
206 x1 = Random::getUniformReal11();
207 x2 = Random::getUniformReal11();
208 r = x1 * x1 + x2 * x2;
209 }
210
211 const Real x = 2 * x1 * sqrt(1 - r);
212 const Real y = 2 * x2 * sqrt(1 - r);
213 const Real z = 1 - 2 * r;
214
215 return RealVect(x, y, z);
216#endif
217}

241inline RealVect
242Random::randomPosition(const RealVect a_cellPos,
243 const RealVect a_lo,
244 const RealVect a_hi,
245 const RealVect a_bndryCentroid,
246 const RealVect a_bndryNormal,
247 const Real a_dx,
248 const Real a_kappa) noexcept
249{
250 CH_TIME("Random::randomPosition(full)");
251
252 RealVect pos;
253
254 if (a_kappa < 1.0) { // Rejection sampling.
255 pos = Random::randomPosition(a_lo, a_hi, a_bndryCentroid, a_bndryNormal);
256 }
257 else { // Regular cell. Get a position.
258 pos = Random::randomPosition(a_lo, a_hi);
259 }
260
261 // Convert from unit cell coordinates to physical coordinates.
262 pos = a_cellPos + pos * a_dx;
263
264 return pos;
265}

267inline RealVect
268Random::randomPosition(const RealVect a_lo,
269 const RealVect a_hi,
270 const RealVect a_bndryCentroid,
271 const RealVect a_bndryNormal) noexcept
272{
273 CH_TIME("Random::randomPosition(partial)");
274
275 // Draw a position.
276 RealVect pos = Random::randomPosition(a_lo, a_hi);
277 bool valid = (pos - a_bndryCentroid).dotProduct(a_bndryNormal) >= 0.0;
278
279 // Rejectino sampling.
280 while (!valid) {
281 pos = Random::randomPosition(a_lo, a_hi);
282 valid = (pos - a_bndryCentroid).dotProduct(a_bndryNormal) >= 0.0;
283 }
284
285 return pos;
286}

288inline RealVect
289Random::randomPosition(const RealVect a_lo, const RealVect a_hi) noexcept
290{
291 CH_TIME("Random::randomPosition(box)");
292
293 RealVect pos = RealVect::Zero;
294
295 for (int dir = 0; dir < SpaceDim; dir++) {
296 pos[dir] = a_lo[dir] + Random::getUniformReal01() * (a_hi[dir] - a_lo[dir]);
297 }
298
299 return pos;
300}
