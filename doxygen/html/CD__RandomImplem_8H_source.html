86 Random::getPoisson(const Real a_mean)
87 {

106 template <typename T, typename>
107 inline T
108 Random::getBinomial(const T a_N, const Real a_p) noexcept
109 {
110  CH_assert(s_seeded);
111 
112  T ret = 0;
113 
114  const bool useNormalApprox = (a_N > (T)9.0 * ((1.0 - a_p) / a_p)) && (a_N > (T)9.0 * a_p / (1.0 - a_p));
115 
116  if (useNormalApprox) {
117  const Real mean = a_N * a_p;
118 
119  std::normal_distribution<Real> normalDist(mean, mean * (1.0 - a_p));
120 
121  ret = (T)std::max(normalDist(s_rng), (Real)0.0);
122  }
123  else {
124  std::binomial_distribution<T> binomDist(a_N, a_p);
125 
126  ret = binomDist(s_rng);
127  }
128 
129  return ret;
130 }

156 inline RealVect
157 Random::getDirection()
158 {
159  CH_assert(s_seeded);
160 
161  constexpr Real safety = 1.E-12;
162 
163 #if CH_SPACEDIM == 2
164  Real x1 = 2.0;
165  Real x2 = 2.0;
166  Real r = x1 * x1 + x2 * x2;
167  while (r >= 1.0 || r < safety) {
168  x1 = Random::getUniformReal11();
169  x2 = Random::getUniformReal11();
170  r = x1 * x1 + x2 * x2;
171  }
172 
173  return RealVect(x1, x2) / sqrt(r);
174 #elif CH_SPACEDIM == 3
175  Real x1 = 2.0;
176  Real x2 = 2.0;
177  Real r = x1 * x1 + x2 * x2;
178  while (r >= 1.0 || r < safety) {
179  x1 = Random::getUniformReal11();
180  x2 = Random::getUniformReal11();
181  r = x1 * x1 + x2 * x2;
182  }
183 
184  const Real x = 2 * x1 * sqrt(1 - r);
185  const Real y = 2 * x2 * sqrt(1 - r);
186  const Real z = 1 - 2 * r;
187 
188  return RealVect(x, y, z);
189 #endif
190 }

226 inline RealVect
227 Random::randomPosition(const RealVect a_lo,
228  const RealVect a_hi,
229  const RealVect a_bndryCentroid,
230  const RealVect a_bndryNormal) noexcept
231 {
232 
233  // Draw a position.
234  RealVect pos = Random::randomPosition(a_lo, a_hi);
235  bool valid = (pos - a_bndryCentroid).dotProduct(a_bndryNormal) >= 0.0;
236 
237  // Rejectino sampling.
238  while (!valid) {
239  pos = Random::randomPosition(a_lo, a_hi);
240  valid = (pos - a_bndryCentroid).dotProduct(a_bndryNormal) >= 0.0;
241  }
242 
243  return pos;
244 }
