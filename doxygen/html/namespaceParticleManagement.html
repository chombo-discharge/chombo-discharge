[](const typename KDNode<P>::ParticleList& a_particles) {
constexpr Real splitThresh = 2.0 - std::numeric_limits<Real>::min();
Real W = 0.0;
for (const auto& p : a_particles) {
W += (p.*weight)();
}
return (W < splitThresh) ? true : false;
}
