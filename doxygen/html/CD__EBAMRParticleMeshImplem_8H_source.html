107 template <class P, const Real&(P::*particleScalarField)() const>
108 void EBAMRParticleMesh::depositPVR(EBAMRCellData& a_meshData,
109  const ParticleContainer<P>& a_particles,
110  const DepositionType a_depositionType,
111  const bool a_forceIrregNGP){
112  CH_TIME("EBAMRParticleMesh::depositPVR");
113 
114  CH_assert(a_meshData[0]->nComp() == 1);
115 
116  // TLDR: On each level the particles are deposited as usual. This routine assumes that none of the fine-level particles
117  // have deposition clouds that hang over the refinement boundary, so there is no fine-to-coarse mass transfer from
118  // fine-level particles. On the coarse level the particles might have deposition clouds that extend beneath the
119  // fine grid. That mass needs to find it's way into the fine level. The way we handle this is that we do piecewise
120  // constant interpolation of the coarse-grid data to the fine grid.
121 
122  DataOps::setValue(a_meshData, 0.0);
123 
124  const Interval interv(0,0);
125 
126  for (int lvl = 0; lvl <= m_finestLevel; lvl++){
127  const DisjointBoxLayout& dbl = m_eblgs[lvl]->getDBL ();
128  const bool hasCoar = lvl > 0;
129 
130  // 1. Deposit particles on this level.
131  for (DataIterator dit(dbl); dit.ok(); ++dit){
132  EBCellFAB& meshData = (*a_meshData[lvl])[dit()];
133  const ListBox<P>& boxParticles = a_particles[lvl] [dit()];
134 
135  const EBParticleMesh& interp = (*m_ebParticleMesh[lvl])[dit()];
136 
137  interp.deposit<P, particleScalarField> (boxParticles.listItems(), meshData, a_depositionType, a_forceIrregNGP);
138  }
139 
140  // 2. Exchange ghost data on this level. After this, all the mass should be on the current level.
141  a_meshData[lvl]->exchange(interv, m_levelCopiers[lvl], EBAddOp());
142 
143  // 3. If the particles deposited over the coarse-fine boundary, add the mass to the coarse level.
144  if(hasCoar){
145  m_coarseFinePM[lvl]->addFineGhostsToCoarse(*a_meshData[lvl-1], *a_meshData[lvl]);
146  }
147 
148  // 4. The mass on the part of the coarse level underneath this grid should be added to this level. Fortunately,
149  // EBCoarseFineParticleMesh has buffers and functions that handle that type of interpolation.
150  if(hasCoar){
151  m_coarseFinePM[lvl]->addInvalidCoarseToFine(*a_meshData[lvl], *a_meshData[lvl-1]);
152  }
153  }
154 }

207 template <class P, const Real&(P::*particleScalarField)() const>
208 void EBAMRParticleMesh::depositHalo(EBAMRCellData& a_meshData,
209  const ParticleContainer<P>& a_particles,
210  const DepositionType a_depositionType,
211  const bool a_forceIrregNGP){
212  CH_TIME("EBAMRParticleMesh::depositHalo");
213 
214  CH_assert(a_meshData[0]->nComp() == 1);
215 
216  // TLDR: On each level the particles are deposited as usual. This routine permits particles to live right next to the refinement boundary
217  // so both the coarse-level and fine-level particles have clouds that hang into the fine/coarse level. The way we handle this is as follows:
218  // The fine-level particles will have deposited into ghost cells that lie around the refinement boundary. This mass is put on the coarse grid.
219  // On the coarse level the particles have clouds that extend beneath the fine grid. Instead of interpolating this mass to the fine grid we
220  // run a special deposition procedure which allows us to deposit directly on the fine grid using 2x or 4x the particle width (depending on the
221  // refinement ratio).
222 
223  DataOps::setValue(a_meshData, 0.0);
224 
225  const Interval interv(0,0);
226 
227  for (int lvl = 0; lvl <= m_finestLevel; lvl++){
228  const DisjointBoxLayout& dbl = m_eblgs[lvl]->getDBL ();
229  const EBISLayout& ebisl = m_eblgs[lvl]->getEBISL ();
230 
231  const bool hasCoar = (lvl > 0);
232 
233  // 1. Deposit particles on this level.
234  for (DataIterator dit(dbl); dit.ok(); ++dit){
235  EBCellFAB& meshData = (*a_meshData [lvl])[dit()];
236  const ListBox<P>& boxParticles = a_particles[lvl] [dit()];
237 
238  // Note: EBParticleMesh construction is cheap.
239  const EBParticleMesh& interp = (*m_ebParticleMesh[lvl])[dit()];
240 
241  interp.deposit<P, particleScalarField> (boxParticles.listItems(), meshData, a_depositionType, a_forceIrregNGP);
242  }
243 
244  // 2. Exchange ghost data on this level.
245  a_meshData[lvl]->exchange(interv, m_levelCopiers[lvl], EBAddOp());
246 
247  // 3. Deposition into ghost cells across the refinement boundary should end up to the coarse level. Add that mass right now.
248  if(hasCoar){
249  m_coarseFinePM[lvl]->addFineGhostsToCoarse(*a_meshData[lvl-1], *a_meshData[lvl]);
250  }
251 
252  // 4. The particles on the other side of the refinement bounadry should deposit into this level.
253  if(hasCoar){
254  const ParticleData<P>& coarHaloParticles = *a_particles.getMaskParticles()[lvl-1]; // These are the particles that live "around" the refinement boundary.
255  const int refRat = m_refRat[lvl-1]; // Refinement between coarse level and this level.
256  const EBLevelGrid& eblgFiCo = m_coarseFinePM[lvl]->getEblgFiCo(); // This is the refinement of the coarse grid.
257  const DisjointBoxLayout& dblFiCo = eblgFiCo.getDBL(); // Refinement of coarse grid
258 
259  LevelData<EBCellFAB>& bufferFiCo = m_coarseFinePM[lvl]->getFiCoBuffer(); // Data that is a refinement of the coarse level.
260 
261  for (DataIterator dit(dblFiCo); dit.ok(); ++dit){
262  EBCellFAB& dataFiCo = bufferFiCo [dit()];
263  const ListBox<P>& haloParticles = coarHaloParticles[dit()];
264 
265  dataFiCo.setVal(0.0);
266 
267  const EBParticleMesh& interp = (*m_ebParticleMeshFiCo[lvl])[dit()];
268 
269  switch(refRat){
270  case 2:
271  interp.deposit2<P, particleScalarField> (haloParticles.listItems(), dataFiCo, a_depositionType, a_forceIrregNGP);
272  break;
273  case 4:
274  interp.deposit4<P, particleScalarField> (haloParticles.listItems(), dataFiCo, a_depositionType, a_forceIrregNGP);
275  break;
276  default:
277  MayDay::Error("CD_EBAMRParticleMeshImplem.H - logic bust in EBAMRParticleMesh::depositHalo");
278  break;
279  }
280  }
281 
282  // Add the result of the buffer deposition to this level.
283  m_coarseFinePM[lvl]->addFiCoDataToFine(*a_meshData[lvl], bufferFiCo);
284  }
285  }
286 }

470 template <class P, Real&(P::*particleScalarField)()>

473  const DepositionType a_interpType,

476

478

480

483  const DisjointBoxLayout& dbl = eblg.getDBL ();
484  const ProblemDomain& domain = eblg.getDomain();
485  const EBISLayout& ebisl = eblg.getEBISL ();
486  const Real& dx = m_dx[lvl];
487 
488  CH_assert(a_meshScalarField[lvl]->nComp() == 1);
489 
490  for (DataIterator dit(dbl); dit.ok(); ++dit){
491  const EBCellFAB& data = (*a_meshScalarField[lvl])[dit()];
492 
493  List<P>& particles = a_particles[lvl][dit()].listItems();
494
