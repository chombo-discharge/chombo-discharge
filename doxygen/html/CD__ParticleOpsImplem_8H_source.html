24 inline bool
25 ParticleOps::domainIntersection(const RealVect& a_oldPos,
26  const RealVect& a_newPos,
27  const RealVect& a_probLo,
28  const RealVect& a_probHi,
29  Real& a_s)
30 {
31 
32  // TLDR: This code does a boundary intersection test and returns where on the interval [oldPos, newPos] the intersection
33  // happened. We do this by checking if the particle moves towards a particular domain side and ends up outside of it.
34 
35  a_s = std::numeric_limits<Real>::max();
36 
37  bool crossedDomainBoundary = false;
38 
39  const RealVect path = a_newPos - a_oldPos;
40 
41  for (int dir = 0; dir < SpaceDim; dir++) {
42  for (SideIterator sit; sit.ok(); ++sit) {
43  const Side::LoHiSide side = sit();
44  const RealVect wallPoint = (side == Side::Lo) ? a_probLo : a_probHi; // A point on the domain side
45  const RealVect n0 =
46  sign(side) * RealVect(BASISV(dir)); // Normal vector pointing OUT of the domain on side sit and direction dir.
47  const Real normPath = PolyGeom::dot(n0, path); // Component of the path that is normal to the domain edge/face.
48 
49  // If normPath > 0 then the particle trajectory points towards the domain edge/face and we can have an intersection.
50  if (normPath > 0.0) {
51 
52  // s determines the intersection point between the particle path and the plane corresponding to the domain edge/face. Note that
53  // we consider the edge/face to be an infinite plane and we just compute the intersection point between each edge/face and select the
54  // closest intersection point.
55  const Real s = PolyGeom::dot(wallPoint - a_oldPos, n0) / normPath;
56  if (s >= 0.0 && s <= 1.0) {
57  crossedDomainBoundary = true;
58  if (s < a_s) {
59  a_s = s;
60  }
61  }
62  }
63  }
64  }
65 
66  return crossedDomainBoundary;
67 }

69 inline bool
70 ParticleOps::ebIntersectionBisect(const RefCountedPtr<BaseIF>& a_impFunc,
71  const RealVect& a_oldPos,
72  const RealVect& a_newPos,
73  const Real& a_bisectStep,
74  Real& a_s)
75 {
76 
77  // TLDR: We compute the intersection point using a bisection algorithm. We divide the full path into intervals and check if an interval
78  // has a root. If it does, we compute it using Brent's algorithm.
79 
80  a_s = std::numeric_limits<Real>::max();
81 
82  bool crossedEB = false;
83 
84  const Real pathLen = (a_newPos - a_oldPos).vectorLength(); // Total path len
85  const int nsteps = ceil(pathLen / a_bisectStep); // Number of bisection intervals
86  const RealVect dxStep = (a_newPos - a_oldPos) / nsteps; // Physical length of each bisection interval
87 
88  // Check each interval
89  RealVect curPos = a_oldPos;
90  for (int istep = 0; istep < nsteps; istep++) {
91  const Real fa = a_impFunc->value(curPos); // Value of the implicit function at the start of the bisection interval
92  const Real fb =
93  a_impFunc->value(curPos + dxStep); // Value of the implicit function at the end of the bisection interval
94 
95  if (fa * fb <= 0.0) {
96 
97  // If this triggered we happen to know that f(pos+dxStep) > 0.0 and f(pos) < 0.0 and so we must have a root on the interval. We now compute the precise location
98  // where the particle crossed the EB. For that we use a Brent root finder on the interval [pos, pos+dxStep]. This is a 1D problem.
99  const RealVect intersectionPos = PolyUtils::brentRootFinder(a_impFunc, curPos, curPos + dxStep);
100  a_s = (intersectionPos - a_oldPos).vectorLength() / pathLen;
101  crossedEB = true;
102 
103  break;
104  }
105  else { // Move to next interval
106  curPos += dxStep;
107  }
108  }
109 
110  return crossedEB;
111 }

113 inline bool
114 ParticleOps::ebIntersectionRaycast(const RefCountedPtr<BaseIF>& a_impFunc,
115  const RealVect& a_oldPos,
116  const RealVect& a_newPos,
117  const Real& a_tolerance,
118  Real& a_s)
119 {
120 
121  a_s = std::numeric_limits<Real>::max();
122 
123  bool ret = false;
124 
125  // Absolute distance to EB.
126  auto dist = [&](const RealVect& x) -> Real { return std::abs(a_impFunc->value(x)); };
127 
128  const Real D = (a_newPos - a_oldPos).vectorLength(); // Total particle path length
129  const Real D0 = dist(a_oldPos); // Distance to EB from starting position
130 
131  // If the distance to the EB from the starting position is smaller than the total path length, we need to check for intersections.
132  if (D > D0) {
133 
134  const RealVect t = (a_newPos - a_oldPos) / D; // Particle trajectory.
135 
136  // Move a_oldPos along +t. If we end up too close to the boundary the particle has intersected the BC. Note that this does NOT check for whether or not
137  // the particle moves tangential to the EB surface. The length of each step is the distance to the EB, so if the particle is close to the EB but moves
138  // tangentially to it, this routine will be EXTREMELY slow.
139  RealVect xa = a_oldPos;
140  Real r = D;
141  Real d = dist(xa);
142 
143  while (d < r) {
144 
145  if (d < a_tolerance) { // We collided.
146  a_s = (xa - a_oldPos).vectorLength() / D;
147  ret = true;
148  break;
149  }
150  else { // We did not collide.
151  xa += t * d;
152  r -= d;
153  d = dist(xa);
154  }
155  }
156  }
157 
158  return ret;
159 }
