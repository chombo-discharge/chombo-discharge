51template <typename P, typename Traits>
52inline void
53ParticleOps::getPhysicalParticlesPerCell(EBAMRCellData& a_ppc, const ParticleContainer<P, Traits>& a_src) noexcept
54{
55 CH_TIME("ParticleOps::getPhysicalParticlesPerCell(SoA)");
56
57 const RealVect probLo = a_src.getProbLo();
58
59 for (int lvl = 0; lvl <= a_src.getFinestLevel(); lvl++) {
60 const DisjointBoxLayout& dbl = a_src.getGrids()[lvl];
61 const DataIterator& dit = dbl.dataIterator();
62 const RealVect dx = a_src.getDx()[lvl];
63
64 const int nbox = dit.size();
65
66#pragma omp parallel for schedule(runtime)
67 for (int mybox = 0; mybox < nbox; mybox++) {
68 const DataIndex& din = dit[mybox];
69 const ParticleSoA<P, Traits>& leaf = a_src[lvl][din];
70
71 FArrayBox& ppc = (*a_ppc[lvl])[din].getFArrayBox();
72 ppc.setVal(0.0);
73
74 for (std::size_t i = 0; i < leaf.size(); i++) {
75 const IntVect iv = ParticleOps::getParticleGridCell(leaf.position(i), probLo, dx);
76
77 ppc(iv, 0) += leaf.weight(i);
78 }
79 }
80 }
81}

125inline bool
126ParticleOps::domainIntersection(const RealVect& a_oldPos,
127 const RealVect& a_newPos,
128 const RealVect& a_probLo,
129 const RealVect& a_probHi,
130 Real& a_s)
131{
132
133 // TLDR: This code does a boundary intersection test and returns where on the interval [oldPos, newPos] the intersection
134 // happened. We do this by checking if the particle moves towards a particular domain side and ends up outside of it.
135
136 a_s = std::numeric_limits<Real>::max();
137
138 bool crossedDomainBoundary = false;
139
140 const RealVect path = a_newPos - a_oldPos;
141
142 for (int dir = 0; dir < SpaceDim; dir++) {
143 for (SideIterator sit; sit.ok(); ++sit) {
144 // Normal vector pointing OUT of the domain on side sit and direction dir.
145 const Side::LoHiSide side = sit();
146 const RealVect wallPoint = (side == Side::Lo) ? a_probLo : a_probHi;
147 const RealVect n0 = sign(side) * RealVect(BASISV(dir));
148 const Real normPath = PolyGeom::dot(n0, path);
149
150 // If normPath > 0 then the particle trajectory points towards the domain edge/face and we can have an intersection.
151 if (normPath > 0.0) {
152
153 // s determines the intersection point between the particle path and the plane corresponding to the domain edge/face. Note that
154 // we consider the edge/face to be an infinite plane and we just compute the intersection point between each edge/face and select the
155 // closest intersection point.
156 const Real s = PolyGeom::dot(wallPoint - a_oldPos, n0) / normPath;
157 if (s >= 0.0 && s <= 1.0) {
158 crossedDomainBoundary = true;
159
160 if (s < a_s) {
161 a_s = s;
162 }
163 }
164 }
165 }
166 }
167
168 return crossedDomainBoundary;
169}

171inline bool
172ParticleOps::ebIntersectionBisect(const RefCountedPtr<BaseIF>& a_impFunc,
173 const RealVect& a_oldPos,
174 const RealVect& a_newPos,
175 const Real& a_bisectStep,
176 Real& a_s)
177{
178
179 // TLDR: We compute the intersection point using a bisection algorithm. We divide the full path into intervals and check if an interval
180 // has a root. If it does, we compute it using Brent's algorithm.
181
182 a_s = std::numeric_limits<Real>::max();
183
184 bool crossedEB = false;
185
186 const Real pathLen = (a_newPos - a_oldPos).vectorLength(); // Total path len
187 const int nsteps = ceil(pathLen / a_bisectStep); // Number of bisection intervals
188 const RealVect dxStep = (a_newPos - a_oldPos) / nsteps; // Physical length of each bisection interval
189
190 // Check each interval
191 RealVect curPos = a_oldPos;
192 for (int istep = 0; istep < nsteps; istep++) {
193 // fa: Value of the implicit function at the start of the bisection interval
194 // fb: Value of the implicit function at the end of the bisection interval
195 const Real fa = a_impFunc->value(curPos);
196 const Real fb = a_impFunc->value(curPos + dxStep);
197
198 if (fa * fb <= 0.0) {
199
200 // If this triggered we happen to know that f(pos+dxStep) > 0.0 and f(pos) < 0.0 and so we must have a root on the interval. We now compute the precise location
201 // where the particle crossed the EB. For that we use a Brent root finder on the interval [pos, pos+dxStep]. This is a 1D problem.
202 const RealVect intersectionPos = PolyUtils::brentRootFinder(a_impFunc, curPos, curPos + dxStep);
203
204 a_s = (intersectionPos - a_oldPos).vectorLength() / pathLen;
205 crossedEB = true;
206
207 break;
208 }
209 else { // Move to next interval
210 curPos += dxStep;
211 }
212 }
213
214 return crossedEB;
215}

217inline bool
218ParticleOps::ebIntersectionRaycast(const RefCountedPtr<BaseIF>& a_impFunc,
219 const RealVect& a_oldPos,
220 const RealVect& a_newPos,
221 const Real& a_tolerance,
222 Real& a_s)
223{
224
225 a_s = std::numeric_limits<Real>::max();
226
227 bool ret = false;
228
229 // Absolute distance to EB.
230 auto dist = [&](const RealVect& x) -> Real {
231 return std::abs(a_impFunc->value(x));
232 };
233
234 const Real D = (a_newPos - a_oldPos).vectorLength(); // Total particle path length
235 const Real D0 = dist(a_oldPos); // Distance to EB from starting position
236
237 // If the distance to the EB from the starting position is smaller than the total path length, we need to check for intersections.
238 if (D > D0) {
239
240 const RealVect t = (a_newPos - a_oldPos) / D; // Particle trajectory.
241
242 // Move a_oldPos along +t. If we end up too close to the boundary the particle has intersected the BC. Note that this does NOT check for whether or not
243 // the particle moves tangential to the EB surface. The length of each step is the distance to the EB, so if the particle is close to the EB but moves
244 // tangentially to it, this routine will be EXTREMELY slow.
245 RealVect xa = a_oldPos;
246 Real r = D;
247 Real d = dist(xa);
248
249 while (d < r) {
250
251 if (d < a_tolerance) { // We collided.
252 a_s = (xa - a_oldPos).vectorLength() / D;
253 ret = true;
254
255 break;
256 }
257 else { // We did not collide.
258 xa += t * d;
259 r -= d;
260 d = dist(xa);
261 }
262 }
263 }
264
265 return ret;
266}

268template <typename P, typename Traits>
269inline void
270ParticleOps::copyDestructive(ParticleContainer<P, Traits>& a_dst, ParticleContainer<P, Traits>& a_src) noexcept
271{
272 CH_TIME("ParticleOps::copyDestructive(ParticleContainer<P, Traits> x2)");
273
274 CH_assert(a_dst.getRealm() == a_src.getRealm());
275
276 for (int lvl = 0; lvl <= a_dst.getFinestLevel(); lvl++) {
277 const DisjointBoxLayout& dbl = a_dst.getGrids()[lvl];
278 const DataIterator& dit = dbl.dataIterator();
279
280 const int nbox = dit.size();
281
282#pragma omp parallel for schedule(runtime)
283 for (int mybox = 0; mybox < nbox; mybox++) {
284 const DataIndex& din = dit[mybox];
285
286 a_dst[lvl][din].clear(); // drop the destination leaf's particles
287 a_dst[lvl][din].catenate(a_src[lvl][din]); // move the source leaf in (O(1) swap; src emptied)
288 }
289 }
290}

292template <typename P, typename Traits>
293inline Real
294ParticleOps::sum(const ParticleContainer<P, Traits>& a_particles) noexcept
295{
296 CH_TIME("ParticleOps::sum(ParticleContainer<P>)");
297
298 Real particleSum = 0.0;
299
300 for (int lvl = 0; lvl <= a_particles.getFinestLevel(); lvl++) {
301 const DisjointBoxLayout& dbl = a_particles.getGrids()[lvl];
302 const DataIterator& dit = dbl.dataIterator();
303
304 const int nbox = dit.size();
305
306#pragma omp parallel for schedule(runtime) reduction(+ : particleSum)
307 for (int mybox = 0; mybox < nbox; mybox++) {
308 const DataIndex& din = dit[mybox];
309 const ParticleSoA<P, Traits>& leaf = a_particles[lvl][din];
310
311 const double* w = leaf.weightColumn(); // weight column captured once, OUTSIDE the loop
312
313 // Seed the fold with the running (thread-private) accumulator rather than 0.0 so the
314 // summation order is identical to a plain 'particleSum += w[i]' loop across this thread's boxes.
315 particleSum = ParticleLoops::reduce(leaf, particleSum, [&](Real acc, std::size_t i) {
316 return acc + w[i];
317 });
318 }
319 }
320
321 return ParallelOps::sum(particleSum);
322}
