94 inline bool
95 ParticleOps::domainIntersection(const RealVect& a_oldPos,
96  const RealVect& a_newPos,
97  const RealVect& a_probLo,
98  const RealVect& a_probHi,
99  Real& a_s)
100 {
101 
102  // TLDR: This code does a boundary intersection test and returns where on the interval [oldPos, newPos] the intersection
103  // happened. We do this by checking if the particle moves towards a particular domain side and ends up outside of it.
104 
105  a_s = std::numeric_limits<Real>::max();
106 
107  bool crossedDomainBoundary = false;
108 
109  const RealVect path = a_newPos - a_oldPos;
110 
111  for (int dir = 0; dir < SpaceDim; dir++) {
112  for (SideIterator sit; sit.ok(); ++sit) {
113  const Side::LoHiSide side = sit();
114  const RealVect wallPoint = (side == Side::Lo) ? a_probLo : a_probHi; // A point on the domain side
115  const RealVect n0 = sign(side) *
116  RealVect(
117  BASISV(dir)); // Normal vector pointing OUT of the domain on side sit and direction dir.
118  const Real normPath = PolyGeom::dot(n0, path); // Component of the path that is normal to the domain edge/face.
119 
120  // If normPath > 0 then the particle trajectory points towards the domain edge/face and we can have an intersection.
121  if (normPath > 0.0) {
122 
123  // s determines the intersection point between the particle path and the plane corresponding to the domain edge/face. Note that
124  // we consider the edge/face to be an infinite plane and we just compute the intersection point between each edge/face and select the
125  // closest intersection point.
126  const Real s = PolyGeom::dot(wallPoint - a_oldPos, n0) / normPath;
127  if (s >= 0.0 && s <= 1.0) {
128  crossedDomainBoundary = true;
129  if (s < a_s) {
130  a_s = s;
131  }
132  }
133  }
134  }
135  }
136 
137  return crossedDomainBoundary;
138 }

140 inline bool
141 ParticleOps::ebIntersectionBisect(const RefCountedPtr<BaseIF>& a_impFunc,
142  const RealVect& a_oldPos,
143  const RealVect& a_newPos,
144  const Real& a_bisectStep,
145  Real& a_s)
146 {
147 
148  // TLDR: We compute the intersection point using a bisection algorithm. We divide the full path into intervals and check if an interval
149  // has a root. If it does, we compute it using Brent's algorithm.
150 
151  a_s = std::numeric_limits<Real>::max();
152 
153  bool crossedEB = false;
154 
155  const Real pathLen = (a_newPos - a_oldPos).vectorLength(); // Total path len
156  const int nsteps = ceil(pathLen / a_bisectStep); // Number of bisection intervals
157  const RealVect dxStep = (a_newPos - a_oldPos) / nsteps; // Physical length of each bisection interval
158 
159  // Check each interval
160  RealVect curPos = a_oldPos;
161  for (int istep = 0; istep < nsteps; istep++) {
162  const Real fa = a_impFunc->value(curPos); // Value of the implicit function at the start of the bisection interval
163  const Real fb = a_impFunc->value(curPos +
164  dxStep); // Value of the implicit function at the end of the bisection interval
165 
166  if (fa * fb <= 0.0) {
167 
168  // If this triggered we happen to know that f(pos+dxStep) > 0.0 and f(pos) < 0.0 and so we must have a root on the interval. We now compute the precise location
169  // where the particle crossed the EB. For that we use a Brent root finder on the interval [pos, pos+dxStep]. This is a 1D problem.
170  const RealVect intersectionPos = PolyUtils::brentRootFinder(a_impFunc, curPos, curPos + dxStep);
171  a_s = (intersectionPos - a_oldPos).vectorLength() / pathLen;
172  crossedEB = true;
173 
174  break;
175  }
176  else { // Move to next interval
177  curPos += dxStep;
178  }
179  }
180 
181  return crossedEB;
182 }

184 inline bool
185 ParticleOps::ebIntersectionRaycast(const RefCountedPtr<BaseIF>& a_impFunc,
186  const RealVect& a_oldPos,
187  const RealVect& a_newPos,
188  const Real& a_tolerance,
189  Real& a_s)
190 {
191 
192  a_s = std::numeric_limits<Real>::max();
193 
194  bool ret = false;
195 
196  // Absolute distance to EB.
197  auto dist = [&](const RealVect& x) -> Real {
198  return std::abs(a_impFunc->value(x));
199  };
200 
201  const Real D = (a_newPos - a_oldPos).vectorLength(); // Total particle path length
202  const Real D0 = dist(a_oldPos); // Distance to EB from starting position
203 
204  // If the distance to the EB from the starting position is smaller than the total path length, we need to check for intersections.
205  if (D > D0) {
206 
207  const RealVect t = (a_newPos - a_oldPos) / D; // Particle trajectory.
208 
209  // Move a_oldPos along +t. If we end up too close to the boundary the particle has intersected the BC. Note that this does NOT check for whether or not
210  // the particle moves tangential to the EB surface. The length of each step is the distance to the EB, so if the particle is close to the EB but moves
211  // tangentially to it, this routine will be EXTREMELY slow.
212  RealVect xa = a_oldPos;
213  Real r = D;
214  Real d = dist(xa);
215 
216  while (d < r) {
217 
218  if (d < a_tolerance) { // We collided.
219  a_s = (xa - a_oldPos).vectorLength() / D;
220  ret = true;
221  break;
222  }
223  else { // We did not collide.
224  xa += t * d;
225  r -= d;
226  d = dist(xa);
227  }
228  }
229  }
230 
231  return ret;
232 }
