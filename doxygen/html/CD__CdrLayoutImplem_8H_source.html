526 template <class T, class S>
527 RefCountedPtr<CdrLayout<T>>
528 CdrFactory<T, S>::newLayout(const Vector<RefCountedPtr<CdrSpecies>>& a_species) const
529 {
530  CH_TIME("CdrFactory<T, S>::newLayout(Vector<RefCountedPtr<CdrSpecies> >)");
531 
532  // Instatiate the return CdrLayout object first.
533  auto cdr = RefCountedPtr<CdrLayout<T>>(new CdrLayout<T>(a_species));
534 
535  // Cast solvers and instantiate them
536  for (int i = 0; i < a_species.size(); i++) {
537 
538  auto solver = RefCountedPtr<T>(static_cast<T*>(new S()));
539 
540  solver->setSpecies(a_species[i]);
541  solver->setPhase(phase::gas);
542  solver->setVerbosity(-1);
543 
544  cdr->addSolver(solver);
545  }
546 
547  return cdr;
548 }
