454 template <class T, class S>
455 RefCountedPtr<RtLayout<T>>
456 RtFactory<T, S>::newLayout(const Vector<RefCountedPtr<RtSpecies>>& a_species) const
457 {
458  // Build rte layout
459  auto rte = RefCountedPtr<RtLayout<T>>(new RtLayout<T>(a_species));
460  auto spe = a_species;
461 
462  // Cast solvers and instantiate them
463  for (int i = 0; i < a_species.size(); i++) {
464  auto solver = RefCountedPtr<T>(static_cast<T*>(new S()));
465  solver->setRtSpecies(spe[i]);
466  solver->setPhase(phase::gas);
467  solver->setVerbosity(-1);
468  rte->addSolver(solver);
469  }
470 
471  return rte;
472 }
