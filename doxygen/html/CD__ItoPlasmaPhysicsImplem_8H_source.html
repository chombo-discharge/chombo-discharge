93 {
94  const Real EPS = 1.E-8;
95 #if CH_SPACEDIM == 2
96  Real x1 = 2.0;
97  Real x2 = 2.0;
98  Real r = x1 * x1 + x2 * x2;
99  while (r >= 1.0 || r < EPS) {
100  x1 = m_udist11(m_rng);
101  x2 = m_udist11(m_rng);

103  }
104 
105  return RealVect(x1, x2) / sqrt(r);
106 #elif CH_SPACEDIM == 3
107  Real x1 = 2.0;
108  Real x2 = 2.0;
109  Real r = x1 * x1 + x2 * x2;

112  x2 = m_udist11(m_rng);
113  r = x1 * x1 + x2 * x2;
114  }

116  const Real x = 2 * x1 * sqrt(1 - r);

433 ItoPlasmaPhysics::removeWeights(Vector<long long>& a_particleWeights, const long long a_numParticlesToRemove) const
434 {
435 
436  constexpr long long zero = 0;
437 
438  if (a_numParticlesToRemove > zero) {
439  const long long numCompParticles = (long long)a_particleWeights.size();
440  const long long weight = a_numParticlesToRemove / numCompParticles;
441  const long long remainder = a_numParticlesToRemove % numCompParticles;
442 
443  // Remove weight from each particle.
444  if (weight > zero) {
445  for (int i = 0; i < a_particleWeights.size(); i++) {
446  a_particleWeights[i] -= weight;
447  }
448  }
449 
450  // Remove the remainder by cycling through the particles.
451  if (remainder > zero) {
452  long long W = zero;
453  for (int i = 0; i < a_particleWeights.size(); i++) {
454  const long long w = std::min(a_particleWeights[i], remainder - W);
455 
456  a_particleWeights[i] -= w;
457  W += w;
458 
459  if (W == remainder) { // OK, we have removed the remainder. Break out.
460  break;
461  }
462  }
463  }
464  }

552 inline void
553 ItoPlasmaPhysics::reconcilePhotoionization(Vector<List<ItoParticle>*>& a_particles,
554  Vector<Real>& a_mean_energies,
555  const Vector<long long>& a_numParticles,
556  const Vector<List<Photon>*>& a_Photons) const
557 {
558 
559  // TLDR: This routine runs through all the photoreactions. For each reaction we look up the target species (e.g. electrons, ions)
560  // and compute the total energy for those species. We then add the photoreaction products, including the added energy, and then
561  // we finally recompute the mean energy.
562 
563  //
564  Vector<long long> newNumParticles = a_numParticles;
565  Vector<Real> energies(a_mean_energies);
566  for (int i = 0; i < energies.size(); i++) {
567  energies[i] *= newNumParticles[i];
568  }
569 
570  for (const auto& r : m_photoReactions) {
571 
572  // Source and targets
573  const int& src = r.second.source();
574  const std::deque<int>& targets = r.second.targets();
575 
576  const std::deque<std::pair<int, Real>>& target_energies = r.second.getTargetEnergies();
577 
578  for (ListIterator<Photon> lit(*a_Photons[src]); lit.ok(); ++lit) {
579  const Photon& phot = lit();
580  const RealVect pos = phot.position();
581  const Real mass = phot.mass(); // Basically the number of reactions, in real format.
582 
583  // Increment energies and update total particles.
584  for (const auto& t : targets) {
585  a_particles[t]->add(ItoParticle(mass, pos));
586 
587  newNumParticles[t] += llround(mass); // Now have this many particles in the cell.
588  }
589 
590  // Increment energies
591  for (const auto& te : target_energies) {
592  const int idx = te.first;
593  const Real dE = te.second;
594  energies[idx] += mass * dE;
595  }
596  }
597  }
598 
599  // Recompute the mean energies. If dE < a_mean_energies then the addition of Photons lead to cooling, pretty cool :)
600  for (int i = 0; i < a_mean_energies.size(); i++) {
601  if (a_numParticles[i] > 0) {
602  a_mean_energies[i] = Max(0.0, energies[i] / newNumParticles[i]);
603  }
604  else {
605  a_mean_energies[i] = 0.0;
606  }
607  }
608 }

908 {

911  ReactionPtrs nonCriticalReactions; // std::deque<shared_ptr<ItoPlasmaReaction> > under the hood

933

942  }

951  }

956  ret.second = nonCriticalReactions;
