393 void ItoPlasmaPhysics::removeWeights(Vector<long long>& a_particleWeights,
394  const long long a_numParticlesToRemove) const {
395 
396  constexpr long long zero = 0;
397 
398  if(a_numParticlesToRemove > zero){
399  const long long numCompParticles = (long long) a_particleWeights.size();
400  const long long weight = a_numParticlesToRemove/numCompParticles;
401  const long long remainder = a_numParticlesToRemove%numCompParticles;
402 
403  // Remove weight from each particle.
404  if(weight > zero){
405  for (int i = 0; i < a_particleWeights.size(); i++){
406  a_particleWeights[i] -= weight;
407  }
408  }
409 
410  // Remove the remainder by cycling through the particles.
411  if(remainder > zero){
412  long long W = zero;
413  for (int i = 0; i < a_particleWeights.size(); i++){
414  const long long w = std::min(a_particleWeights[i], remainder-W);
415 
416  a_particleWeights[i] -= w;
417  W += w;
418 
419  if(W == remainder) { // OK, we have removed the remainder. Break out.
420  break;
421  }
422  }
423  }
424  }

499 void ItoPlasmaPhysics::reconcilePhotoionization(Vector<List<ItoParticle>* >& a_particles,
500  Vector<Real>& a_mean_energies,
501  const Vector<long long>& a_numParticles,
502  const Vector<List<Photon>* >& a_Photons) const {
503 
504  // TLDR: This routine runs through all the photoreactions. For each reaction we look up the target species (e.g. electrons, ions)
505  // and compute the total energy for those species. We then add the photoreaction products, including the added energy, and then
506  // we finally recompute the mean energy.
507 
508  //
509  Vector<long long> newNumParticles = a_numParticles;
510  Vector<Real> energies(a_mean_energies);
511  for (int i = 0; i < energies.size(); i++){
512  energies[i] *= newNumParticles[i];
513  }
514 
515  for (const auto& r : m_photoReactions){
516 
517  // Source and targets
518  const int& src = r.second.source();
519  const std::deque<int>& targets = r.second.targets();
520 
521  const std::deque<std::pair<int, Real> >& target_energies = r.second.getTargetEnergies();
522 
523  for (ListIterator<Photon> lit(*a_Photons[src]); lit.ok(); ++lit){
524  const Photon& phot = lit();
525  const RealVect pos = phot.position();
526  const Real mass = phot.mass(); // Basically the number of reactions, in real format.
527 
528  // Increment energies and update total particles.
529  for (const auto& t : targets){
530  a_particles[t]->add(ItoParticle(mass, pos));
531 
532  newNumParticles[t] += llround(mass); // Now have this many particles in the cell.
533  }
534 
535  // Increment energies
536  for (const auto& te : target_energies){
537  const int idx = te.first;
538  const Real dE = te.second;
539  energies[idx] += mass*dE;
540  }
541  }
542  }
543 
544  // Recompute the mean energies. If dE < a_mean_energies then the addition of Photons lead to cooling, pretty cool :)
545  for (int i = 0; i < a_mean_energies.size(); i++){
546  if(a_numParticles[i] > 0){
547  a_mean_energies[i] = Max(0.0, energies[i]/newNumParticles[i]);
548  }
549  else{
550  a_mean_energies[i] = 0.0;
551  }
552  }
553 }
