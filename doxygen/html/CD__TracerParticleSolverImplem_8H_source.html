92 TracerParticleSolver<P>::parseDeposition()
93 {
94  CH_TIME("TracerParticleSolver::parseDeposition()");
95  if (m_verbosity > 5) {
96  pout() << m_name + "::parseDeposition()" << endl;
97  }
98 
99  ParmParse pp(m_className.c_str());
100 
101  std::string str;
102 
103  // Deposition for particle-mesh operations
104  pp.get("deposition", str);
105  if (str == "ngp") {
106  m_deposition = DepositionType::NGP;

108  else if (str == "cic") {

111  else {
112  MayDay::Error("TracerParticleSolver::parseDeposition - unknown deposition method requested.");
113  }
114 
115  pp.get("deposition_cf", str);
116  if (str == "interp") {
117  m_coarseFineDeposition = CoarseFineDeposition::Interp;
118  }
119  else if (str == "halo") {
120  m_coarseFineDeposition = CoarseFineDeposition::Halo;
121  }
122  else if (str == "halo_ngp") {
123  m_coarseFineDeposition = CoarseFineDeposition::HaloNGP;
124  }
125  else {
126  MayDay::Error("TracerParticleSolver::parseDeposition - unknown coarse-fine deposition method requested.");
127  }
128 
129  // Interpolation type particle-mesh operations
130  pp.get("interpolation", str);
131  if (str == "ngp") {
132  m_interpolation = DepositionType::NGP;
133  }
134  else if (str == "cic") {
135  m_interpolation = DepositionType::CIC;
136  }
137  else {
138  MayDay::Error("TracerParticleSolver::parseDeposition - unknown interpolation method requested.");
139  }

566 template <typename P>
567 Real
568 TracerParticleSolver<P>::computeDt() const
569 {
570  CH_TIME("TracerParticleSolver::computeDt()");
571  if (m_verbosity > 5) {
572  pout() << m_name + "::computeDt()" << endl;
573  }
574 
575  Real dt = std::numeric_limits<Real>::infinity();
576 
577  for (int lvl = 0; lvl <= m_amr->getFinestLevel(); lvl++) {
578  const DisjointBoxLayout& dbl = m_amr->getGrids(m_realm)[lvl];
579  const Real dx = m_amr->getDx()[lvl];
580 
581  for (DataIterator dit(dbl); dit.ok(); ++dit) {
582  const List<P>& particles = m_particles[lvl][dit()].listItems();
583 
584  for (ListIterator<P> lit(particles); lit.ok(); ++lit) {
585  const RealVect& v = lit().velocity();
586 
587  for (int dir = 0; dir < SpaceDim; dir++) {
588  dt = std::min(dt, dx / std::abs(v[dir]));
589  }
590  }
591  }
592  }
593 
594  // Get the minimum of dt over all MPI ranks.
595  dt = ParallelOps::min(dt);
596 
597  return dt;
598 }

638 template <typename P>
639 template <typename T, const Real& (T::*particleScalarFunction)() const>
640 void
641 TracerParticleSolver<P>::depositParticles(EBAMRCellData& a_phi,
642  ParticleContainer<T>& a_particles,
643  const DepositionType a_baseDeposition,
644  const CoarseFineDeposition a_coarseFineDeposition)
645 {
646  CH_TIME("TracerParticleSolver::depositParticles()");
647  if (m_verbosity > 5) {
648  pout() << m_name + "::depositParticles()" << endl;
649  }
650 
651  CH_assert(a_phi[0]->nComp() == 1);
652 
653  switch (a_coarseFineDeposition) {
654  case CoarseFineDeposition::Interp: {
655  m_amr->depositParticles<T, particleScalarFunction>(a_phi,
656  m_realm,
657  m_phase,
658  a_particles,
659  a_baseDeposition,
660  CoarseFineDeposition::Interp,
661  false);
662 
663  break;
664  }
665  case CoarseFineDeposition::Halo: {
666 
667  // Copy particles living on the mask.
668  const AMRMask& mask = m_amr->getMask(s_particle_halo, m_haloBuffer, m_realm);
669  a_particles.copyMaskParticles(mask);
670 
671  m_amr->depositParticles<T, particleScalarFunction>(a_phi,
672  m_realm,
673  m_phase,
674  a_particles,
675  a_baseDeposition,
676  CoarseFineDeposition::Halo,
677  false);
678 
679  // Clear out the mask particles.
680  a_particles.clearMaskParticles();
681 
682  break;
683  }
684  case CoarseFineDeposition::HaloNGP: {
