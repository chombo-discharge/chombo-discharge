34template <typename P, typename Traits>
35inline typename ParticleContainer<P, Traits>::Destination
36ParticleContainer<P, Traits>::findDestination(const RealVect& a_pos) const
37{
38 // Finest level whose tile contains the particle wins.
39 for (int lvl = m_finestLevel; lvl >= 0; lvl--) {
40 IntVect tile;
41
42 for (int dir = 0; dir < SpaceDim; dir++) {
43 tile[dir] = static_cast<int>(std::floor((a_pos[dir] - m_probLo[dir]) / (m_minBlockSize * m_dx[lvl][dir])));
44 }
45
46 const LevelTiles& tiles = *m_levelTiles[lvl];
47
48 const auto& myTiles = tiles.getMyTiles();
49 const auto mit = myTiles.find(tile);
50
51 if (mit != myTiles.end()) {
52 return Destination{lvl, mit->second, procID(), true};
53 }
54
55#ifdef CH_MPI
56 const auto& otherTiles = tiles.getOtherTiles();
57 const auto oit = otherTiles.find(tile);
58
59 if (oit != otherTiles.end()) {
60 return Destination{lvl, oit->second.first, static_cast<int>(oit->second.second), true};
61 }
62#endif
63 }
64
65 return Destination{-1, 0u, -1, false};
66}

68template <typename P, typename Traits>
69inline typename ParticleContainer<P, Traits>::MoverPool
70ParticleContainer<P, Traits>::gatherToPool(const AMRParticlesSoA<P, Traits>& a_source,
71 const Vector<DisjointBoxLayout>& a_sourceGrids)
72{
73 CH_TIME("ParticleContainer::gatherToPool");
74
75 MoverPool pool(numProc());
76
77 // Bin every particle of one leaf into a destination pool, returning the number of off-domain
78 // (outcast) particles. Shared by the serial and OpenMP paths so the binning logic lives once.
79 // findDestination() is const and reads only the (immutable-during-gather) tile maps, so it is
80 // safe to call concurrently; each thread bins into its own thread-local pool.
81 auto binLeaf = [this](MoverPool& a_target, const Leaf& a_leaf) -> unsigned long long {
82 unsigned long long outcast = 0;
83
84 for (std::size_t i = 0; i < a_leaf.size(); i++) {
85 const RealVect pos = a_leaf.position(i);
86 const Destination dst = this->findDestination(pos);
87
88 if (!dst.found) {
89 outcast++;
90 continue;
91 }
92
93 Leaf& bin = a_target[dst.rank][PoolKey(dst.level, dst.gridIndex)];
94
95 bin.append(pos, a_leaf.weight(i), a_leaf.gather(i));
96
97 const std::size_t k = bin.size() - 1;
98
99 bin.particleID(k) = a_leaf.particleID(i); // preserve id
100 bin.rankID(k) = dst.rank; // new owner
101 }
102
103 return outcast;
104 };
105
106 unsigned long long numOutcast = 0;
107
108 for (int lvl = 0; lvl < a_source.size(); lvl++) {
109 const LevelParticles& level = *a_source[lvl];
110 const DataIterator& dit = a_sourceGrids[lvl].dataIterator();
111 const int nbox = dit.size();
112
113#ifdef _OPENMP
114#pragma omp parallel reduction(+ : numOutcast)
115 {
116 MoverPool threadPool(numProc()); // thread-local; no locking while binning
117
118#pragma omp for schedule(runtime)
119 for (int mybox = 0; mybox < nbox; mybox++) {
120 numOutcast += binLeaf(threadPool, level[dit[mybox]]);
121 }
122
123 // Merge the thread-local pool into the shared pool. First writer to a (rank,key) slot gets the
124 // O(1) arena swap inside catenate(); later writers append.
125#pragma omp critical(ParticleContainer_gatherMerge)
126 {
127 for (int r = 0; r < numProc(); r++) {
128 for (auto& kv : threadPool[r]) {
129 pool[r][kv.first].catenate(kv.second);
130 }
131 }
132 }
133 }
134#else
135 for (int mybox = 0; mybox < nbox; mybox++) {
136 numOutcast += binLeaf(pool, level[dit[mybox]]);
137 }
138#endif
139 }
140
141 m_numOutcastLocal = numOutcast;
142
143 return pool;
144}

146template <typename P, typename Traits>
147inline typename ParticleContainer<P, Traits>::MoverPool
148ParticleContainer<P, Traits>::gatherMoversToPool()
149{
150 CH_TIME("ParticleContainer::gatherMoversToPool");
151
152 CH_assert(m_validCells != nullptr);
153
154 MoverPool pool(numProc());
155 unsigned long long numOutcast = 0;
156
157 for (int lvl = 0; lvl <= m_finestLevel; lvl++) {
158 LevelParticles& level = *m_particles[lvl];
159 const LevelData<BaseFab<bool>>* maskLevel = (*m_validCells)[lvl].isNull() ? nullptr : &(*(*m_validCells)[lvl]);
160 const DataIterator& dit = m_grids[lvl].dataIterator();
161 const int nbox = dit.size();
162
163 // Partition one leaf in place: keep stayers (still inside their box AND in a cell flagged valid =
164 // not covered by a finer level), pool the movers and swap-pop them out. a_mask == nullptr forces
165 // every particle to be treated as a mover (no usable mask for this leaf). Returns outcast movers.
166 // findDestination()/cellIndex() are const reads, so this is safe to call from many threads as long
167 // as each leaf is touched by a single thread (the omp-for partitions boxes disjointly).
168 auto partitionLeaf = [this, lvl](MoverPool& a_target,
169 [[maybe_unused]] const DataIndex& a_din,
170 Leaf& a_leaf,
171 const BaseFab<bool>* a_mask) -> unsigned long long {
172 unsigned long long outcast = 0;
173
174 const Box maskBox = (a_mask != nullptr) ? a_mask->box() : Box();
175
176 std::size_t i = 0;
177
178 while (i < a_leaf.size()) {
179 const RealVect pos = a_leaf.position(i);
180 if (a_mask != nullptr) {
181 const IntVect iv = this->cellIndex(lvl, pos);
182
183 if (maskBox.contains(iv) && (*a_mask)(iv, 0)) {
184 // Stayer: in its own box and not under a finer grid, so it cannot have changed owner.
185 // Debug self-check of that invariant -- the destination the gather-all path would compute
186 // must be exactly this (level, grid, rank). If this ever fires, the skip is unsound.
187#ifndef NDEBUG
188 const Destination chk = this->findDestination(pos);
189 CH_assert(chk.found && chk.level == lvl && chk.rank == procID() &&
190 m_levelTiles[lvl]->getMyGrids().at(chk.gridIndex) == a_din);
191#endif
192 i++;
193
194 continue;
195 }
196 }
197
198 const Destination dst = this->findDestination(pos); // mover: route it
199
200 if (dst.found) {
201 Leaf& bin = a_target[dst.rank][PoolKey(dst.level, dst.gridIndex)];
202
203 bin.append(pos, a_leaf.weight(i), a_leaf.gather(i));
204
205 const std::size_t k = bin.size() - 1;
206
207 bin.particleID(k) = a_leaf.particleID(i); // preserve id
208 bin.rankID(k) = dst.rank; // new owner
209 }
210 else {
211 outcast++;
212 }
213
214 a_leaf.remove(i); // swap-pop; do NOT advance i (a new particle now occupies slot i)
215 }
216
217 return outcast;
218 };
219
220#ifdef _OPENMP
221#pragma omp parallel reduction(+ : numOutcast)
222 {
223 MoverPool threadPool(numProc());
224
225#pragma omp for schedule(runtime)
226 for (int mybox = 0; mybox < nbox; mybox++) {
227 const DataIndex& din = dit[mybox];
228 const BaseFab<bool>* mask = (maskLevel != nullptr) ? &(*maskLevel)[din] : nullptr;
229
230 if (mask != nullptr && !mask->isUsable()) {
231 mask = nullptr;
232 }
233
234 numOutcast += partitionLeaf(threadPool, din, level[din], mask);
235 }
236
237#pragma omp critical(ParticleContainer_gatherMerge)
238 {
239 for (int r = 0; r < numProc(); r++) {
240 for (auto& kv : threadPool[r]) {
241 pool[r][kv.first].catenate(kv.second);
242 }
243 }
244 }
245 }
246#else
247 for (int mybox = 0; mybox < nbox; mybox++) {
248 const DataIndex& din = dit[mybox];
249 const BaseFab<bool>* mask = (maskLevel != nullptr) ? &(*maskLevel)[din] : nullptr;
250
251 if (mask != nullptr && !mask->isUsable()) {
252 mask = nullptr;
253 }
254
255 numOutcast += partitionLeaf(pool, din, level[din], mask);
256 }
257#endif
258 }
259
260 m_numOutcastLocal = numOutcast;
261
262 return pool;
263}

265template <typename P, typename Traits>
266inline void
267ParticleContainer<P, Traits>::distributeFromPool(MoverPool& a_pool)
268{
269 CH_TIME("ParticleContainer::distributeFromPool");
270
271 const int myRank = procID();
272 const int numRanks = numProc();
273
274 // ---- same-rank movers: catenate each pooled leaf into its destination leaf ----
275 // On the gather-all path the destination leaves are empty (remap clears them; regrid allocates them
276 // fresh) so catenate() takes the O(1) arena-swap fast path. On the mask fast path the destinations
277 // still hold their stayers, so catenate() appends (O(movers)); a mover never returns to its own
278 // origin leaf, so there is no self-append.
279 for (auto& kv : a_pool[myRank]) {
280 const int lvl = kv.first.first;
281 const unsigned int gidx = kv.first.second;
282 const DataIndex din = m_levelTiles[lvl]->getMyGrids().at(gidx);
283
284 (*m_particles[lvl])[din].catenate(kv.second);
285 }
286
287#ifdef CH_MPI
288 // ---- cross-rank movers: one Alltoallv of linearized particles ----
289 constexpr std::size_t pp = Leaf::bytesPerParticle();
290 constexpr std::size_t headerLen = 2 * sizeof(std::uint32_t) + sizeof(std::uint64_t);
291
292 std::vector<std::vector<char>> sendBuf(numRanks);
293 // Each destination rank's buffer and pool map are disjoint, and linearizeParticle is a const read,
294 // so packing different ranks in parallel is race-free.
295#ifdef _OPENMP
296#pragma omp parallel for schedule(runtime)
297#endif
298 for (int r = 0; r < numRanks; r++) {
299 if (r == myRank) {
300 continue;
301 }
302
303 std::vector<char>& buf = sendBuf[r];
304
305 for (auto& kv : a_pool[r]) {
306 Leaf& src = kv.second;
307
308 const std::uint32_t lvl = static_cast<std::uint32_t>(kv.first.first);
309 const std::uint32_t gidx = static_cast<std::uint32_t>(kv.first.second);
310 const std::uint64_t count = static_cast<std::uint64_t>(src.size());
311 const std::size_t off = buf.size();
312
313 buf.resize(off + headerLen + count * pp);
314
315 char* p = buf.data() + off;
316
317 std::memcpy(p, &lvl, sizeof(std::uint32_t));
318 p += sizeof(std::uint32_t);
319
320 std::memcpy(p, &gidx, sizeof(std::uint32_t));
321 p += sizeof(std::uint32_t);
322
323 std::memcpy(p, &count, sizeof(std::uint64_t));
324 p += sizeof(std::uint64_t);
325
326 for (std::size_t i = 0; i < src.size(); i++) {
327 src.linearizeParticle(p, i);
328 p += pp;
329 }
330 }
331 }
332
333 // Exchange byte counts, then the byte buffers.
334 std::vector<int> sendCounts(numRanks, 0);
335 std::vector<int> recvCounts(numRanks, 0);
336
337 for (int r = 0; r < numRanks; r++) {
338 CH_assert(sendBuf[r].size() <= static_cast<std::size_t>(std::numeric_limits<int>::max()));
339 sendCounts[r] = static_cast<int>(sendBuf[r].size());
340 }
341
342 MPI_Alltoall(sendCounts.data(), 1, MPI_INT, recvCounts.data(), 1, MPI_INT, Chombo_MPI::comm);
343
344 std::vector<int> sdispl(numRanks, 0);
345 std::vector<int> rdispl(numRanks, 0);
346
347 std::int64_t stot = 0;
348 std::int64_t rtot = 0;
349
350 for (int r = 0; r < numRanks; r++) {
351 sdispl[r] = static_cast<int>(stot);
352 stot += sendCounts[r];
353
354 rdispl[r] = static_cast<int>(rtot);
355 rtot += recvCounts[r];
356 }
357
358 // The accumulators are int64 so the cross-rank sum cannot overflow; MPI displacement arrays are
359 // int, so assert the totals fit before allocating the flat buffers and calling MPI_Alltoallv.
360 CH_assert(stot <= static_cast<std::int64_t>(std::numeric_limits<int>::max()));
361 CH_assert(rtot <= static_cast<std::int64_t>(std::numeric_limits<int>::max()));
362
363 std::vector<char> sflat(stot);
364 std::vector<char> rflat(rtot);
365
366 for (int r = 0; r < numRanks; r++) {
367 if (sendCounts[r] > 0) {
368 std::memcpy(sflat.data() + sdispl[r], sendBuf[r].data(), static_cast<std::size_t>(sendCounts[r]));
369 }
370 }
371
372 MPI_Alltoallv(sflat.data(),
373 sendCounts.data(),
374 sdispl.data(),
375 MPI_BYTE,
376 rflat.data(),
377 recvCounts.data(),
378 rdispl.data(),
379 MPI_BYTE,
380 Chombo_MPI::comm);
381
382 // ---- unpack: group received chunks by destination leaf, then append in parallel ----
383 // Pass 1 (serial, header-only walk): record each chunk's payload pointer + count, grouped by its
384 // destination (level, gridIndex). Pass 2 appends in parallel with one destination leaf per thread,
385 // so no two threads ever touch the same leaf (delinearizeAndAppend mutates the leaf arena).
386 struct RecvChunk
387 {
388 const char* payload; // first particle of the chunk within rflat
389 std::uint64_t count;
390 };
391 struct DestGroup
392 {
393 int level;
394 unsigned int gridIndex;
395 std::uint64_t total;
396 std::vector<RecvChunk> chunks;
397 };
398
399 std::vector<DestGroup> groups;
400 std::map<PoolKey, std::size_t> keyToGroup;
401 {
402 const char* p = rflat.data();
403 const char* end = p + rtot;
404
405 while (p < end) {
406 std::uint32_t lvl;
407 std::uint32_t gidx;
408 std::uint64_t count;
409
410 std::memcpy(&lvl, p, sizeof(std::uint32_t));
411 p += sizeof(std::uint32_t);
412
413 std::memcpy(&gidx, p, sizeof(std::uint32_t));
414 p += sizeof(std::uint32_t);
415
416 std::memcpy(&count, p, sizeof(std::uint64_t));
417 p += sizeof(std::uint64_t);
418
419 const PoolKey key(static_cast<int>(lvl), gidx);
420 const auto it = keyToGroup.find(key);
421
422 if (it == keyToGroup.end()) {
423 keyToGroup.emplace(key, groups.size());
424 groups.push_back(DestGroup{static_cast<int>(lvl), gidx, count, {RecvChunk{p, count}}});
425 }
426 else {
427 DestGroup& g = groups[it->second];
428 g.total += count;
429 g.chunks.push_back(RecvChunk{p, count});
430 }
431
432 p += count * pp;
433 }
434 }
435
436 const int numGroups = static_cast<int>(groups.size());
437#ifdef _OPENMP
438#pragma omp parallel for schedule(runtime)
439#endif
440 for (int d = 0; d < numGroups; d++) {
441 const DestGroup& g = groups[d];
442 const DataIndex din = m_levelTiles[g.level]->getMyGrids().at(g.gridIndex);
443
444 Leaf& tgt = (*m_particles[g.level])[din];
445
446 tgt.reserve(tgt.size() + g.total); // one allocation; avoids geometric regrow during the appends
447 for (const RecvChunk& chunk : g.chunks) {
448 const char* q = chunk.payload;
449
450 for (std::uint64_t c = 0; c < chunk.count; c++) {
451 tgt.delinearizeAndAppend(q);
452 q += pp;
453 tgt.rankID(tgt.size() - 1) = myRank; // owner is the receiving rank
454 }
455 }
456 }
457#endif
458}

460template <typename P, typename Traits>
461inline void
462ParticleContainer<P, Traits>::remap()
463{
464 CH_TIME("ParticleContainer::remap");
465
466 CH_assert(m_isDefined);
467
468 if (m_validCells != nullptr && static_cast<int>(m_validCells->size()) >= m_finestLevel + 1) {
469 // Mask fast path: a particle still in its box and not under a finer grid cannot change owner, so we
470 // leave those "stayers" in place and pool only the movers. Destinations therefore keep their stayers
471 // -- do NOT clearParticles() here; distributeFromPool() appends the movers onto the existing leaves.
472 MoverPool pool = this->gatherMoversToPool();
473
474 this->distributeFromPool(pool);
475 }
476 else {
477 // Fallback: re-pool every valid particle (used when no mask is available, e.g. addParticlesDestructive
478 // staging, or before the realm's valid-cell mask has been built).
479 MoverPool pool = this->gatherToPool(m_particles, m_grids);
480
481 this->clearParticles();
482 this->distributeFromPool(pool);
483 }
484
485 m_isOrganizedByCell = false;
486}

502template <typename P, typename Traits>
503inline void
504ParticleContainer<P, Traits>::regrid(const Vector<DisjointBoxLayout>& a_grids,
505 const Vector<ProblemDomain>& a_domains,
506 const Vector<Real>& a_dx,
507 const Vector<int>& a_refRat,
508 const int a_minBlockSize,
509 const Vector<RefCountedPtr<LevelTiles>>& a_levelTiles,
510 const int a_newFinestLevel)
511{
512 CH_TIME("ParticleContainer::regrid");
513
514 CH_assert(m_isDefined);
515 CH_assert(a_levelTiles.size() > a_newFinestLevel);
516
517 // ---- adopt the new layout ----
518 m_grids = a_grids;
519 m_domains = a_domains;
520 m_refRat = a_refRat;
521 m_minBlockSize = a_minBlockSize;
522 m_finestLevel = a_newFinestLevel;
523
524 // Alias the Realm's per-level tile->box maps (the single source of truth); never build our own.
525 m_levelTiles = a_levelTiles;
526
527 m_dx.resize(a_newFinestLevel + 1);
528 for (int lvl = 0; lvl <= a_newFinestLevel; lvl++) {
529 m_dx[lvl] = a_dx[lvl] * RealVect::Unit;
530 }
531
532 // ---- re-allocate the (empty) holders over the new grids ----
533 m_particles.resize(a_newFinestLevel + 1);
534 m_maskParticles.resize(a_newFinestLevel + 1);
535
536 for (int lvl = 0; lvl <= a_newFinestLevel; lvl++) {
537 m_particles[lvl] = RefCountedPtr<LevelParticles>(new LevelParticles(a_grids[lvl]));
538 m_maskParticles[lvl] = RefCountedPtr<LevelParticles>(new LevelParticles(a_grids[lvl]));
539 }
540
541 this->setupGrownGrids();
542
543 m_bufferParticles.resize(a_newFinestLevel + 1);
544
545 for (int lvl = 0; lvl <= a_newFinestLevel; lvl++) {
546 m_bufferParticles[lvl] = RefCountedPtr<LevelParticles>(new LevelParticles(m_grownGrids[lvl]));
547 }
548
549 // ---- redistribute the cached particles onto the new layout ----
550 MoverPool pool = this->gatherToPool(m_cacheParticles, m_cacheGrids);
551 this->distributeFromPool(pool);
552
553 // ---- release the cache ----
554 m_cacheParticles.resize(0);
555 m_cacheGrids.resize(0);
556
557 m_isOrganizedByCell = false;
558}

560template <typename P, typename Traits>
561inline void
562ParticleContainer<P, Traits>::setupGrownGrids()
563{
564 CH_TIME("ParticleContainer::setupGrownGrids");
565
566 m_grownGrids.resize(m_finestLevel + 1);
567 for (int lvl = 0; lvl <= m_finestLevel; lvl++) {
568 const DisjointBoxLayout& dbl = m_grids[lvl];
569
570 Vector<Box> boxes = dbl.boxArray();
571
572 if (lvl > 0) {
573 const Box domainBox = m_domains[lvl].domainBox();
574 for (int i = 0; i < boxes.size(); i++) {
575 boxes[i].grow(m_refRat[lvl - 1]);
576 boxes[i] &= domainBox;
577 }
578 }
579
580 m_grownGrids[lvl] = BoxLayout(boxes, dbl.procIDs());
581 }
582}

599template <typename P, typename Traits>
600inline void
601ParticleContainer<P, Traits>::copyMaskParticles(const int a_level, const LevelData<BaseFab<bool>>& a_mask)
602{
603 CH_TIME("ParticleContainer::copyMaskParticles(level)");
604
605 CH_assert(m_isDefined);
606
607 LevelParticles& maskLevel = *m_maskParticles[a_level];
608 const LevelParticles& srcLevel = *m_particles[a_level];
609
610 const DisjointBoxLayout& dbl = m_grids[a_level];
611 const DataIterator& dit = dbl.dataIterator();
612
613 const int nbox = dit.size();
614
615#pragma omp parallel for schedule(runtime)
616 for (int mybox = 0; mybox < nbox; mybox++) {
617 const DataIndex& din = dit[mybox];
618
619 maskLevel[din].clear();
620
621 const BaseFab<bool>& mask = a_mask[din];
622 if (!mask.isUsable()) {
623 continue;
624 }
625 const Box maskBox = mask.box();
626
627 const Leaf& src = srcLevel[din];
628 Leaf& dst = maskLevel[din];
629
630 for (std::size_t i = 0; i < src.size(); i++) {
631 const IntVect iv = this->cellIndex(a_level, src.position(i));
632
633 if (maskBox.contains(iv) && mask(iv, 0)) {
634 dst.append(src.position(i), src.weight(i), src.gather(i));
635
636 const std::size_t k = dst.size() - 1;
637
638 dst.particleID(k) = src.particleID(i);
639 dst.rankID(k) = src.rankID(i);
640 }
641 }
642 }

660template <typename P, typename Traits>
661inline void
662ParticleContainer<P, Traits>::transferMaskParticles(const int a_level, const LevelData<BaseFab<bool>>& a_mask)
663{
664 CH_TIME("ParticleContainer::transferMaskParticles(level)");
665
666 CH_assert(m_isDefined);
667
668 LevelParticles& maskLevel = *m_maskParticles[a_level];
669 LevelParticles& srcLevel = *m_particles[a_level];
670
671 const DisjointBoxLayout& dbl = m_grids[a_level];
672 const DataIterator& dit = dbl.dataIterator();
673
674 const int nbox = dit.size();
675
676#pragma omp parallel for schedule(runtime)
677 for (int mybox = 0; mybox < nbox; mybox++) {
