38template <typename P, typename Traits>
39inline typename ParticleContainer<P, Traits>::Destination
40ParticleContainer<P, Traits>::findDestination(const RealVect& a_pos) const
41{
42 // Finest level whose tile contains the particle wins.
43 for (int lvl = m_finestLevel; lvl >= 0; lvl--) {
44 IntVect tile;
45
46 for (int dir = 0; dir < SpaceDim; dir++) {
47 tile[dir] = static_cast<int>(std::floor((a_pos[dir] - m_probLo[dir]) / (m_minBlockSize * m_dx[lvl][dir])));
48 }
49
50 const LevelTiles& tiles = *m_levelTiles[lvl];
51
52 const auto& myTiles = tiles.getMyTiles();
53 const auto mit = myTiles.find(tile);
54
55 if (mit != myTiles.end()) {
56 return Destination{lvl, mit->second, procID(), true};
57 }
58
59#ifdef CH_MPI
60 const auto& otherTiles = tiles.getOtherTiles();
61 const auto oit = otherTiles.find(tile);
62
63 if (oit != otherTiles.end()) {
64 return Destination{lvl, oit->second.first, static_cast<int>(oit->second.second), true};
65 }
66#endif
67 }
68
69 return Destination{-1, 0u, -1, false};
70}

72template <typename P, typename Traits>
73inline typename ParticleContainer<P, Traits>::MoverPool
74ParticleContainer<P, Traits>::gatherToPool(const AMRParticlesSoA<P, Traits>& a_source,
75 const Vector<DisjointBoxLayout>& a_sourceGrids)
76{
77 CH_TIME("ParticleContainer::gatherToPool");
78
79 MoverPool pool(numProc());
80
81 // Bin every particle of one leaf into a destination pool, returning the number of off-domain
82 // (outcast) particles. Shared by the serial and OpenMP paths so the binning logic lives once.
83 // findDestination() is const and reads only the (immutable-during-gather) tile maps, so it is
84 // safe to call concurrently; each thread bins into its own thread-local pool.
85 auto binLeaf = [this](MoverPool& a_target, const Leaf& a_leaf) -> unsigned long long {
86 unsigned long long outcast = 0;
87
88 for (std::size_t i = 0; i < a_leaf.size(); i++) {
89 const RealVect pos = a_leaf.position(i);
90 const Destination dst = this->findDestination(pos);
91
92 if (!dst.found) {
93 outcast++;
94 continue;
95 }
96
97 Leaf& bin = a_target[dst.rank][PoolKey(dst.level, dst.gridIndex)];
98
99 bin.append(pos, a_leaf.weight(i), a_leaf.gather(i));
100
101 const std::size_t k = bin.size() - 1;
102
103 bin.particleID(k) = a_leaf.particleID(i); // preserve id
104 bin.rankID(k) = dst.rank; // new owner
105 }
106
107 return outcast;
108 };
109
110 unsigned long long numOutcast = 0;
111
112 for (int lvl = 0; lvl < a_source.size(); lvl++) {
113 const LevelParticles& level = *a_source[lvl];
114 const DataIterator& dit = a_sourceGrids[lvl].dataIterator();
115 const int nbox = dit.size();
116
117#ifdef _OPENMP
118#pragma omp parallel reduction(+ : numOutcast)
119 {
120 MoverPool threadPool(numProc()); // thread-local; no locking while binning
121
122#pragma omp for schedule(runtime)
123 for (int mybox = 0; mybox < nbox; mybox++) {
124 numOutcast += binLeaf(threadPool, level[dit[mybox]]);
125 }
126
127 // Merge the thread-local pool into the shared pool. First writer to a (rank,key) slot gets the
128 // O(1) arena swap inside catenate(); later writers append.
129#pragma omp critical(ParticleContainer_gatherMerge)
130 {
131 for (int r = 0; r < numProc(); r++) {
132 for (auto& kv : threadPool[r]) {
133 pool[r][kv.first].catenate(kv.second);
134 }
135 }
136 }
137 }
138#else
139 for (int mybox = 0; mybox < nbox; mybox++) {
140 numOutcast += binLeaf(pool, level[dit[mybox]]);
141 }
142#endif
143 }
144
145 m_numOutcastLocal = numOutcast;
146
147 return pool;
148}

150template <typename P, typename Traits>
151inline typename ParticleContainer<P, Traits>::MoverPool
152ParticleContainer<P, Traits>::gatherMoversToPool()
153{
154 CH_TIME("ParticleContainer::gatherMoversToPool");
155
156 CH_assert(m_validCells != nullptr);
157
158 MoverPool pool(numProc());
159 unsigned long long numOutcast = 0;
160
161 for (int lvl = 0; lvl <= m_finestLevel; lvl++) {
162 LevelParticles& level = *m_particles[lvl];
163 const LevelData<BaseFab<bool>>* maskLevel = (*m_validCells)[lvl].isNull() ? nullptr : &(*(*m_validCells)[lvl]);
164 const DataIterator& dit = m_grids[lvl].dataIterator();
165 const int nbox = dit.size();
166
167 // Partition one leaf in place: keep stayers (still inside their box AND in a cell flagged valid =
168 // not covered by a finer level), pool the movers and swap-pop them out. a_mask == nullptr forces
169 // every particle to be treated as a mover (no usable mask for this leaf). Returns outcast movers.
170 // findDestination()/cellIndex() are const reads, so this is safe to call from many threads as long
171 // as each leaf is touched by a single thread (the omp-for partitions boxes disjointly).
172 auto partitionLeaf = [this, lvl](MoverPool& a_target,
173 [[maybe_unused]] const DataIndex& a_din,
174 Leaf& a_leaf,
175 const BaseFab<bool>* a_mask) -> unsigned long long {
176 unsigned long long outcast = 0;
177
178 const Box maskBox = (a_mask != nullptr) ? a_mask->box() : Box();
179
180 std::size_t i = 0;
181
182 while (i < a_leaf.size()) {
183 const RealVect pos = a_leaf.position(i);
184 if (a_mask != nullptr) {
185 const IntVect iv = this->cellIndex(lvl, pos);
186
187 if (maskBox.contains(iv) && (*a_mask)(iv, 0)) {
188 // Stayer: in its own box and not under a finer grid, so it cannot have changed owner.
189 // Debug self-check of that invariant -- the destination the gather-all path would compute
190 // must be exactly this (level, grid, rank). If this ever fires, the skip is unsound.
191#ifndef NDEBUG
192 const Destination chk = this->findDestination(pos);
193 CH_assert(chk.found && chk.level == lvl && chk.rank == procID() &&
194 m_levelTiles[lvl]->getMyGrids().at(chk.gridIndex) == a_din);
195#endif
196 i++;
197
198 continue;
199 }
200 }
201
202 const Destination dst = this->findDestination(pos); // mover: route it
203
204 if (dst.found) {
205 Leaf& bin = a_target[dst.rank][PoolKey(dst.level, dst.gridIndex)];
206
207 bin.append(pos, a_leaf.weight(i), a_leaf.gather(i));
208
209 const std::size_t k = bin.size() - 1;
210
211 bin.particleID(k) = a_leaf.particleID(i); // preserve id
212 bin.rankID(k) = dst.rank; // new owner
213 }
214 else {
215 outcast++;
216 }
217
218 a_leaf.remove(i); // swap-pop; do NOT advance i (a new particle now occupies slot i)
219 }
220
221 return outcast;
222 };
223
224#ifdef _OPENMP
225#pragma omp parallel reduction(+ : numOutcast)
226 {
227 MoverPool threadPool(numProc());
228
229#pragma omp for schedule(runtime)
230 for (int mybox = 0; mybox < nbox; mybox++) {
231 const DataIndex& din = dit[mybox];
232 const BaseFab<bool>* mask = (maskLevel != nullptr) ? &(*maskLevel)[din] : nullptr;
233
234 if (mask != nullptr && !mask->isUsable()) {
235 mask = nullptr;
236 }
237
238 numOutcast += partitionLeaf(threadPool, din, level[din], mask);
239 }
240
241#pragma omp critical(ParticleContainer_gatherMerge)
242 {
243 for (int r = 0; r < numProc(); r++) {
244 for (auto& kv : threadPool[r]) {
245 pool[r][kv.first].catenate(kv.second);
246 }
247 }
248 }
249 }
250#else
251 for (int mybox = 0; mybox < nbox; mybox++) {
252 const DataIndex& din = dit[mybox];
253 const BaseFab<bool>* mask = (maskLevel != nullptr) ? &(*maskLevel)[din] : nullptr;
254
255 if (mask != nullptr && !mask->isUsable()) {
256 mask = nullptr;
257 }
258
259 numOutcast += partitionLeaf(pool, din, level[din], mask);
260 }
261#endif
262 }
263
264 m_numOutcastLocal = numOutcast;
265
266 return pool;
267}

269template <typename P, typename Traits>
270inline void
271ParticleContainer<P, Traits>::distributeFromPool(MoverPool& a_pool, const bool a_setOwnerToReceiver)
272{
273 CH_TIME("ParticleContainer::distributeFromPool");
274
275 const int myRank = procID();
276 const int numRanks = numProc();
277
278 // ---- same-rank movers: catenate each pooled leaf into its destination leaf ----
279 // On the gather-all path the destination leaves are empty (remap clears them; regrid allocates them
280 // fresh) so catenate() takes the O(1) arena-swap fast path. On the mask fast path the destinations
281 // still hold their stayers, so catenate() appends (O(movers)); a mover never returns to its own
282 // origin leaf, so there is no self-append.
283 for (auto& kv : a_pool[myRank]) {
284 const int lvl = kv.first.first;
285 const unsigned int gidx = kv.first.second;
286 const DataIndex din = m_levelTiles[lvl]->getMyGrids().at(gidx);
287
288 (*m_particles[lvl])[din].catenate(kv.second);
289 }
290
291#ifdef CH_MPI
292 // ---- cross-rank movers: one Alltoallv of linearized particles ----
293 constexpr std::size_t pp = Leaf::bytesPerParticle();
294 constexpr std::size_t headerLen = 2 * sizeof(std::uint32_t) + sizeof(std::uint64_t);
295
296 std::vector<std::vector<char>> sendBuf(numRanks);
297 // Each destination rank's buffer and pool map are disjoint, and linearizeParticle is a const read,
298 // so packing different ranks in parallel is race-free.
299#ifdef _OPENMP
300#pragma omp parallel for schedule(runtime)
301#endif
302 for (int r = 0; r < numRanks; r++) {
303 if (r == myRank) {
304 continue;
305 }
306
307 std::vector<char>& buf = sendBuf[r];
308
309 for (auto& kv : a_pool[r]) {
310 Leaf& src = kv.second;
311
312 const std::uint32_t lvl = static_cast<std::uint32_t>(kv.first.first);
313 const std::uint32_t gidx = static_cast<std::uint32_t>(kv.first.second);
314 const std::uint64_t count = static_cast<std::uint64_t>(src.size());
315 const std::size_t off = buf.size();
316
317 buf.resize(off + headerLen + count * pp);
318
319 char* p = buf.data() + off;
320
321 std::memcpy(p, &lvl, sizeof(std::uint32_t));
322 p += sizeof(std::uint32_t);
323
324 std::memcpy(p, &gidx, sizeof(std::uint32_t));
325 p += sizeof(std::uint32_t);
326
327 std::memcpy(p, &count, sizeof(std::uint64_t));
328 p += sizeof(std::uint64_t);
329
330 for (std::size_t i = 0; i < src.size(); i++) {
331 src.linearizeParticle(p, i);
332 p += pp;
333 }
334 }
335 }
336
337 // Exchange byte counts, then the byte buffers.
338 std::vector<int> sendCounts(numRanks, 0);
339 std::vector<int> recvCounts(numRanks, 0);
340
341 for (int r = 0; r < numRanks; r++) {
342 CH_assert(sendBuf[r].size() <= static_cast<std::size_t>(std::numeric_limits<int>::max()));
343 sendCounts[r] = static_cast<int>(sendBuf[r].size());
344 }
345
346 MPI_Alltoall(sendCounts.data(), 1, MPI_INT, recvCounts.data(), 1, MPI_INT, Chombo_MPI::comm);
347
348 std::vector<int> sdispl(numRanks, 0);
349 std::vector<int> rdispl(numRanks, 0);
350
351 std::int64_t stot = 0;
352 std::int64_t rtot = 0;
353
354 for (int r = 0; r < numRanks; r++) {
355 sdispl[r] = static_cast<int>(stot);
356 stot += sendCounts[r];
357
358 rdispl[r] = static_cast<int>(rtot);
359 rtot += recvCounts[r];
360 }
361
362 // The accumulators are int64 so the cross-rank sum cannot overflow; MPI displacement arrays are
363 // int, so assert the totals fit before allocating the flat buffers and calling MPI_Alltoallv.
364 CH_assert(stot <= static_cast<std::int64_t>(std::numeric_limits<int>::max()));
365 CH_assert(rtot <= static_cast<std::int64_t>(std::numeric_limits<int>::max()));
366
367 std::vector<char> sflat(stot);
368 std::vector<char> rflat(rtot);
369
370 for (int r = 0; r < numRanks; r++) {
371 if (sendCounts[r] > 0) {
372 std::memcpy(sflat.data() + sdispl[r], sendBuf[r].data(), static_cast<std::size_t>(sendCounts[r]));
373 }
374 }
375
376 MPI_Alltoallv(sflat.data(),
377 sendCounts.data(),
378 sdispl.data(),
379 MPI_BYTE,
380 rflat.data(),
381 recvCounts.data(),
382 rdispl.data(),
383 MPI_BYTE,
384 Chombo_MPI::comm);
385
386 // ---- unpack: group received chunks by destination leaf, then append in parallel ----
387 // Pass 1 (serial, header-only walk): record each chunk's payload pointer + count, grouped by its
388 // destination (level, gridIndex). Pass 2 appends in parallel with one destination leaf per thread,
389 // so no two threads ever touch the same leaf (delinearizeAndAppend mutates the leaf arena).
390 struct RecvChunk
391 {
392 const char* payload; // first particle of the chunk within rflat
393 std::uint64_t count;
394 };
395 struct DestGroup
396 {
397 int level;
398 unsigned int gridIndex;
399 std::uint64_t total;
400 std::vector<RecvChunk> chunks;
401 };
402
403 std::vector<DestGroup> groups;
404 std::map<PoolKey, std::size_t> keyToGroup;
405 {
406 const char* p = rflat.data();
407 const char* end = p + rtot;
408
409 while (p < end) {
410 std::uint32_t lvl;
411 std::uint32_t gidx;
412 std::uint64_t count;
413
414 std::memcpy(&lvl, p, sizeof(std::uint32_t));
415 p += sizeof(std::uint32_t);
416
417 std::memcpy(&gidx, p, sizeof(std::uint32_t));
418 p += sizeof(std::uint32_t);
419
420 std::memcpy(&count, p, sizeof(std::uint64_t));
421 p += sizeof(std::uint64_t);
422
423 const PoolKey key(static_cast<int>(lvl), gidx);
424 const auto it = keyToGroup.find(key);
425
426 if (it == keyToGroup.end()) {
427 keyToGroup.emplace(key, groups.size());
428 groups.push_back(DestGroup{static_cast<int>(lvl), gidx, count, {RecvChunk{p, count}}});
429 }
430 else {
431 DestGroup& g = groups[it->second];
432 g.total += count;
433 g.chunks.push_back(RecvChunk{p, count});
434 }
435
436 p += count * pp;
437 }
438 }
439
440 const int numGroups = static_cast<int>(groups.size());
441
442#ifdef _OPENMP
443#pragma omp parallel for schedule(runtime)
444#endif
445 for (int d = 0; d < numGroups; d++) {
446 const DestGroup& g = groups[d];
447 const DataIndex din = m_levelTiles[g.level]->getMyGrids().at(g.gridIndex);
448
449 Leaf& tgt = (*m_particles[g.level])[din];
450
451 tgt.reserve(tgt.size() + g.total); // one allocation; avoids geometric regrow during the appends
452 for (const RecvChunk& chunk : g.chunks) {
453 const char* q = chunk.payload;
454
455 for (std::uint64_t c = 0; c < chunk.count; c++) {
456 tgt.delinearizeAndAppend(q);
457 q += pp;
458 if (a_setOwnerToReceiver) {
459 tgt.rankID(tgt.size() - 1) = myRank; // remap/regrid: owner is the receiving rank
460 }
461 // else: keep the rankID carried in the buffer (ghost particles record their true owner)
462 }
463 }
464 }
465#endif
466}

468template <typename P, typename Traits>
469inline void
470ParticleContainer<P, Traits>::remap()
471{
472 CH_TIME("ParticleContainer::remap");
473
474 CH_assert(m_isDefined);
475
476 // Ghost particles are transient, non-owned copies; drop them so they are never routed as if owned.
477 this->clearGhostParticles();
478
479 if (m_validCells != nullptr && static_cast<int>(m_validCells->size()) >= m_finestLevel + 1) {
480 // Mask fast path: a particle still in its box and not under a finer grid cannot change owner, so we
481 // leave those "stayers" in place and pool only the movers. Destinations therefore keep their stayers
482 // -- do NOT clearParticles() here; distributeFromPool() appends the movers onto the existing leaves.
483 MoverPool pool = this->gatherMoversToPool();
484
485 this->distributeFromPool(pool, true); // remap: the receiving rank becomes the new owner
486 }
487 else {
488 // Fallback: re-pool every valid particle (used when no mask is available, e.g. addParticlesDestructive
489 // staging, or before the realm's valid-cell mask has been built).
490 MoverPool pool = this->gatherToPool(m_particles, m_grids);
491
492 this->clearParticles();
493 this->distributeFromPool(pool, true); // remap: the receiving rank becomes the new owner
494 }
495
496 m_isOrganizedByCell = false;
497}

499template <typename P, typename Traits>
500inline void
501ParticleContainer<P, Traits>::preRegrid()
502{
503 CH_TIME("ParticleContainer::preRegrid");
504
505 CH_assert(m_isDefined);
506
507 // Ghost particles are transient, non-owned copies; drop them so they are never cached/regridded.
508 this->clearGhostParticles();
509
510 // Share the current holders (RefCountedPtr) so regrid() can rebuild m_particles while the old

512 m_cacheParticles = m_particles;
513 m_cacheGrids = m_grids;
514}

516template <typename P, typename Traits>
517inline void
518ParticleContainer<P, Traits>::regrid(const Vector<DisjointBoxLayout>& a_grids,
519 const Vector<ProblemDomain>& a_domains,
520 const Vector<Real>& a_dx,
521 const Vector<int>& a_refRat,
522 const int a_minBlockSize,
523 const Vector<RefCountedPtr<LevelTiles>>& a_levelTiles,
524 const int a_newFinestLevel)
525{
526 CH_TIME("ParticleContainer::regrid");
527
528 CH_assert(m_isDefined);
529 CH_assert(a_levelTiles.size() > a_newFinestLevel);
530
531 // ---- adopt the new layout ----
532 m_grids = a_grids;
533 m_domains = a_domains;
534 m_refRat = a_refRat;
535 m_minBlockSize = a_minBlockSize;
536 m_finestLevel = a_newFinestLevel;
537
538 // Alias the Realm's per-level tile->box maps (the single source of truth); never build our own.
539 m_levelTiles = a_levelTiles;
540
541 m_dx.resize(a_newFinestLevel + 1);
542 for (int lvl = 0; lvl <= a_newFinestLevel; lvl++) {
543 m_dx[lvl] = a_dx[lvl] * RealVect::Unit;
544 }
545
546 // ---- re-allocate the (empty) holders over the new grids ----
547 m_particles.resize(a_newFinestLevel + 1);
548 m_maskParticles.resize(a_newFinestLevel + 1);
549
550 for (int lvl = 0; lvl <= a_newFinestLevel; lvl++) {
551 m_particles[lvl] = RefCountedPtr<LevelParticles>(new LevelParticles(a_grids[lvl]));
552 m_maskParticles[lvl] = RefCountedPtr<LevelParticles>(new LevelParticles(a_grids[lvl]));
553 }
554
555 this->setupGrownGrids();
556
557 m_bufferParticles.resize(a_newFinestLevel + 1);
558
559 for (int lvl = 0; lvl <= a_newFinestLevel; lvl++) {
560 m_bufferParticles[lvl] = RefCountedPtr<LevelParticles>(new LevelParticles(m_grownGrids[lvl]));
561 }
562
563 // ---- redistribute the cached particles onto the new layout ----
564 MoverPool pool = this->gatherToPool(m_cacheParticles, m_cacheGrids);
565 this->distributeFromPool(pool, true); // regrid: the receiving rank becomes the new owner
566
567 // ---- release the cache ----
568 m_cacheParticles.resize(0);
569 m_cacheGrids.resize(0);
570
571 m_isOrganizedByCell = false;
572}

574template <typename P, typename Traits>
575inline void
576ParticleContainer<P, Traits>::clearGhostParticles()
577{
578 CH_TIME("ParticleContainer::clearGhostParticles");
579
580 CH_assert(m_isDefined);
581
582 for (int lvl = 0; lvl <= m_finestLevel; lvl++) {
583 LevelParticles& level = *m_particles[lvl];
584 const DisjointBoxLayout& dbl = m_grids[lvl];
585 const DataIterator& dit = dbl.dataIterator();
586
587 const int nbox = dit.size();
588
589#pragma omp parallel for schedule(runtime)
590 for (int mybox = 0; mybox < nbox; mybox++) {
591 const DataIndex& din = dit[mybox];
592 Leaf& leaf = level[din];
593
594 // Swap-and-pop every ghost; the swapped-in particle now occupies index i, so do NOT advance on a hit.
595 std::size_t i = 0;
596 while (i < leaf.size()) {
597 if (leaf.isGhost(i)) {
598 leaf.remove(i);
599 }
600 else {
601 i++;
602 }
603 }
604 }
605 }
606
607 m_isOrganizedByCell = false; // swap-pop reorders the surviving valid particles
608}

610template <typename P, typename Traits>
611inline void
612ParticleContainer<P, Traits>::claimOwnership(const bool a_onlyValidRegion)
613{
614 CH_TIME("ParticleContainer::claimOwnership");
615
616 CH_assert(m_isDefined);
617
618 const RankID myRank = static_cast<RankID>(procID());
619
620 // Ghosts are left untouched (they carry their true owner rank). Each non-ghost particle in the valid
621 // region (or all non-ghost particles when !a_onlyValidRegion) is stamped with this rank. Per-particle
622 // and box-local -> embarrassingly parallel.
623 for (int lvl = 0; lvl <= m_finestLevel; lvl++) {
624 LevelParticles& level = *m_particles[lvl];
625 const DisjointBoxLayout& dbl = m_grids[lvl];
626 const DataIterator& dit = dbl.dataIterator();
627 const int nbox = dit.size();
628
629#pragma omp parallel for schedule(runtime)
630 for (int mybox = 0; mybox < nbox; mybox++) {
631 const DataIndex& din = dit[mybox];
632 const Box box = dbl[din];
633 Leaf& leaf = level[din];
634
635 for (std::size_t i = 0; i < leaf.size(); i++) {
636 if (leaf.isGhost(i)) {
637 continue;
638 }
639 if (!a_onlyValidRegion || this->inValidBox(lvl, box, leaf.position(i))) {
640 leaf.rankID(i) = myRank;
641 }
642 }
643 }
644 }
645}

647template <typename P, typename Traits>
648inline long long
649ParticleContainer<P, Traits>::resetParticleIDs(const bool a_onlyValidRegion, const ParticleID a_startID)
650{
651 CH_TIME("ParticleContainer::resetParticleIDs");
652
653 CH_assert(m_isDefined);
654
655 // A non-ghost particle qualifies for a new id iff (!a_onlyValidRegion) or its cell lies in the leaf's
656 // grid box. Ghosts never qualify (and are left untouched); non-qualifying non-ghost strays get
657 // s_invalidID. The predicate must be identical in the count and assign passes below.
658
659 // Flatten the local boxes in (level, box) order -- this fixes the local id ordering.
660 std::vector<std::pair<int, DataIndex>> boxes;
661 for (int lvl = 0; lvl <= m_finestLevel; lvl++) {
662 const DataIterator& dit = m_grids[lvl].dataIterator();
663 const int nbox = dit.size();
664 for (int mybox = 0; mybox < nbox; mybox++) {
665 boxes.emplace_back(lvl, dit[mybox]);
666 }
667 }
668 const int numBoxes = static_cast<int>(boxes.size());
669
670 // Pass 1: count qualifying particles per box (parallel; each box independent).
671 std::vector<long long> boxCount(numBoxes, 0);
672#pragma omp parallel for schedule(runtime)
673 for (int b = 0; b < numBoxes; b++) {
674 const int lvl = boxes[b].first;
675 const DataIndex& din = boxes[b].second;
676 const Box box = m_grids[lvl][din];
677 const Leaf& leaf = (*m_particles[lvl])[din];
678
679 long long count = 0;
680 for (std::size_t i = 0; i < leaf.size(); i++) {
681 if (!leaf.isGhost(i) && (!a_onlyValidRegion || this->inValidBox(lvl, box, leaf.position(i)))) {
682 count++;
683 }
684 }
685 boxCount[b] = count;
686 }
687
688 // Local exclusive prefix over boxes + local total.
689 std::vector<long long> boxLocalOffset(numBoxes, 0);
690 long long localTotal = 0;
691 for (int b = 0; b < numBoxes; b++) {
692 boxLocalOffset[b] = localTotal;
693 localTotal += boxCount[b];
694 }
695
696 // Per-rank exclusive offset: all-gather each rank's local total, then sum the ranks before this one.
697 const Vector<long long> perRankCount = ParallelOps::gather(localTotal);
698
699 ParticleID rankOffset = a_startID;
700 for (int r = 0; r < procID(); r++) {
701 rankOffset += perRankCount[r];
702 }
703
704 // Pass 2: assign ids (parallel; each box starts at its own offset). Ghosts untouched; qualifying get a
705 // contiguous id; non-qualifying non-ghost strays get s_invalidID.
706#pragma omp parallel for schedule(runtime)
707 for (int b = 0; b < numBoxes; b++) {
708 const int lvl = boxes[b].first;
709 const DataIndex& din = boxes[b].second;
710 const Box box = m_grids[lvl][din];
711 Leaf& leaf = (*m_particles[lvl])[din];
712
713 ParticleID id = rankOffset + static_cast<ParticleID>(boxLocalOffset[b]);
714 for (std::size_t i = 0; i < leaf.size(); i++) {
715 if (leaf.isGhost(i)) {
716 continue;
717 }
718 if (!a_onlyValidRegion || this->inValidBox(lvl, box, leaf.position(i))) {
719 leaf.particleID(i) = id++;
720 }
721 else {
722 leaf.particleID(i) = ParticleSoA<P, Traits>::s_invalidID;
723 }
724 }
725 }
726
727 return ParallelOps::sum(localTotal);
728}

730template <typename P, typename Traits>
731inline typename ParticleContainer<P, Traits>::MoverPool
732ParticleContainer<P, Traits>::gatherGhostsFromMasks(const AMRParticleGhostMask& a_maskSame,
733 const AMRParticleGhostMask& a_maskCoarToFine,
734 const AMRParticleGhostMask& a_maskFineToCoar)
735{
736 CH_TIME("ParticleContainer::gatherGhostsFromMasks");
737
738 MoverPool pool(numProc());
739
740 // Bin one source leaf's valid particles as ghosts of every target box listed by its cell's masks. The
741 // three masks are all indexed by the SOURCE cell on the source level, so a single cell lookup drives all
742 // directions. GhostType is from the RECEIVER's view: same level -> SameLevel, coarse source scattered to
743 // a finer patch -> Coarse, fine source scattered to a coarser patch -> Fine.
744 auto binLeaf = [&](MoverPool& a_target, const Leaf& a_leaf, const int a_srcLvl, const DataIndex& a_din) {
745 const bool haveSame = (a_srcLvl < a_maskSame.size()) && !a_maskSame[a_srcLvl].isNull();
746 const bool haveC2F = (a_srcLvl < m_finestLevel) && (a_srcLvl < a_maskCoarToFine.size()) &&
747 !a_maskCoarToFine[a_srcLvl].isNull();
748 const bool haveF2C = (a_srcLvl >= 1) && (a_srcLvl < a_maskFineToCoar.size()) &&
749 !a_maskFineToCoar[a_srcLvl].isNull();
750
751 const ParticleGhostMask* same = haveSame ? &((*a_maskSame[a_srcLvl])[a_din]) : nullptr;
752 const ParticleGhostMask* c2f = haveC2F ? &((*a_maskCoarToFine[a_srcLvl])[a_din]) : nullptr;
753 const ParticleGhostMask* f2c = haveF2C ? &((*a_maskFineToCoar[a_srcLvl])[a_din]) : nullptr;
754
755 if (same == nullptr && c2f == nullptr && f2c == nullptr) {
756 return;
757 }
758
759 for (std::size_t i = 0; i < a_leaf.size(); i++) {
760 const RealVect pos = a_leaf.position(i);
761 const IntVect iv = this->cellIndex(a_srcLvl, pos);
762
763 auto scatter =
764 [&](const ParticleGhostMask* a_mask, const int a_dstLvl, const GhostType a_gt, const bool a_pruneByTargetBox) {
765 if (a_mask == nullptr || !a_mask->box().contains(iv)) {
766 return;
767 }
768 const int nt = a_mask->numTargets(iv);
769
770 // Coarse-to-fine only: the source (coarse) cell is coarser than the fine ghost width, so a whole
771 // coarse cell would over-scatter. Keep the particle for a target only if its DESTINATION (fine)
772 // cell lies in that target's acceptance box (the exact fine ghost shell).
773 const IntVect dstCell = a_pruneByTargetBox ? this->cellIndex(a_dstLvl, pos) : IntVect::Zero;
774
775 for (int k = 0; k < nt; k++) {
776 if (a_pruneByTargetBox && !a_mask->targetBox(iv, k).contains(dstCell)) {
777 continue;
778 }
779
780 Leaf& bin = a_target[a_mask->targetRank(iv, k)][PoolKey(a_dstLvl, a_mask->targetGridIndex(iv, k))];
781
782 bin.append(pos, a_leaf.weight(i), a_leaf.gather(i));
783
784 const std::size_t j = bin.size() - 1;
785
786 bin.particleID(j) = a_leaf.particleID(i); // preserve id
787 bin.rankID(j) = procID(); // owner is this rank (it owns its valid particles)
788 bin.ghost(j) = a_gt;
789 }
790 };
791
792 scatter(same, a_srcLvl, GhostType::SameLevel, false);
793 scatter(c2f, a_srcLvl + 1, GhostType::Coarse, true);
794 scatter(f2c, a_srcLvl - 1, GhostType::Fine, false);
795 }
796 };
797
798 for (int lvl = 0; lvl <= m_finestLevel; lvl++) {
799 const LevelParticles& level = *m_particles[lvl];
800 const DataIterator& dit = m_grids[lvl].dataIterator();
801 const int nbox = dit.size();
802
803#ifdef _OPENMP
804#pragma omp parallel
805 {
806 MoverPool threadPool(numProc()); // thread-local; no locking while binning
807
808#pragma omp for schedule(runtime)
809 for (int mybox = 0; mybox < nbox; mybox++) {
810 binLeaf(threadPool, level[dit[mybox]], lvl, dit[mybox]);
811 }
812
813#pragma omp critical(ParticleContainer_gatherGhostsFromMasksMerge)
814 {
815 for (int r = 0; r < numProc(); r++) {
816 for (auto& kv : threadPool[r]) {
817 pool[r][kv.first].catenate(kv.second);
818 }
819 }
820 }
821 }
822#else
823 for (int mybox = 0; mybox < nbox; mybox++) {
824 binLeaf(pool, level[dit[mybox]], lvl, dit[mybox]);
825 }
826#endif
827 }
828
829 return pool;
830}
