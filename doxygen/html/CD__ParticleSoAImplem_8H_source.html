27template <typename P, typename Traits>
28inline void
29ParticleSoA<P, Traits>::reallocate(const std::size_t a_capacity)
30{
31 std::array<std::size_t, s_numColumns> off{};
32 const std::size_t total = this->computeOffsets(a_capacity, off, std::make_index_sequence<s_numColumns>{});
33 void* const newBuf = std::aligned_alloc(s_align, alignUp(total));
34 if (newBuf == nullptr) {
35 MayDay::Abort("ParticleSoA::reallocate - std::aligned_alloc returned null (out of memory)");
36 }
37 if (m_size > 0) {
38 this->moveColumns(newBuf, off, std::make_index_sequence<s_numColumns>{});
39 }
40 std::free(m_buffer);
41 m_buffer = newBuf;
42 m_capacity = a_capacity;
43 this->setColPtrs(off, std::make_index_sequence<s_numColumns>{});
44}

46template <typename P, typename Traits>
47inline void
48ParticleSoA<P, Traits>::permute(const std::vector<std::size_t>& a_dest)
49{
50 std::array<std::size_t, s_numColumns> off{};
51 const std::size_t total = this->computeOffsets(m_capacity, off, std::make_index_sequence<s_numColumns>{});
52 void* const newBuf = std::aligned_alloc(s_align, alignUp(total));
53 if (newBuf == nullptr) {
54 MayDay::Abort("ParticleSoA::permute - std::aligned_alloc returned null (out of memory)");
55 }
56 this->scatterColumns(newBuf, off, a_dest, std::make_index_sequence<s_numColumns>{});
57 std::free(m_buffer);
58 m_buffer = newBuf;
59 this->setColPtrs(off, std::make_index_sequence<s_numColumns>{});
60}

81template <typename P, typename Traits>
82inline ParticleSoA<P, Traits>
83ParticleSoA<P, Traits>::deepCopy() const
84{
85 ParticleSoA<P, Traits> dst;
86 dst.reserve(m_capacity); // identical arena layout: reallocate() sets the capacity exactly
87 if (m_size > 0) {
88 std::memcpy(dst.m_buffer, m_buffer, this->byteSpan()); // one contiguous copy of the live region
89 }
90 dst.m_size = m_size;
91 dst.m_sorted = m_sorted;
92 dst.m_cellStart = m_cellStart;
93 return dst;
94}

96template <typename P, typename Traits>
97inline void
98ParticleSoA<P, Traits>::deepCopyTo(ParticleSoA<P, Traits>& a_dst) const
99{
100 if (this == &a_dst) {
101 return; // copying onto self is a no-op
102 }
103
104 a_dst.clear(); // drop a_dst's particles (keeps its arena) so the grow below moves nothing
105 a_dst.reserve(m_size); // reuse a_dst's allocation if it is already large enough; grow only if not
106 if (m_size > 0) {
107 this->copyLiveColumnsTo(a_dst, std::make_index_sequence<s_numColumns>{});
108 }
109 a_dst.m_size = m_size;
110 a_dst.m_sorted = m_sorted;
111 a_dst.m_cellStart = m_cellStart;
112}

114template <typename P, typename Traits>
115inline void
116ParticleSoA<P, Traits>::append(const ParticleSoA<P, Traits>& a_other)
117{
118 const std::size_t n = a_other.m_size;
119 if (n == 0) {
120 return;
121 }
122 const std::size_t oldSize = m_size;
123 this->reserve(oldSize + n); // may reallocate this (and a_other, when a_other == this); pointers refreshed below
124 this->appendColumnsFrom(a_other, oldSize, std::make_index_sequence<s_numColumns>{});
125 m_size = oldSize + n;
126 m_sorted = false;
127}

144ParticleSoA<P, Traits>::shrinkToFit()
145{
146 if (m_capacity <= m_size) {
147 return; // already compact
148 }
149 if (m_size == 0) {
150 std::free(m_buffer);
151 m_buffer = nullptr;
152 m_capacity = 0;

154 return;

159template <typename P, typename Traits>
160inline void
161ParticleSoA<P, Traits>::append(const RealVect& a_position, const double a_weight, const P& a_payload)
162{
163 this->growIfFull();
164 for (int dir = 0; dir < SpaceDim; dir++) {
165 this->positionColumn(dir)[m_size] = a_position[dir]; // write the reserved slot directly (m_size not yet bumped)
166 }
167 this->template columnByIndex<s_weightCol>()[m_size] = a_weight;
168 this->template columnByIndex<s_idCol>()[m_size] = s_invalidID;
169 this->template columnByIndex<s_rankCol>()[m_size] = static_cast<RankID>(-1);
170 this->writePayload(a_payload, std::make_index_sequence<s_numPayloadColumns>{});
171 m_size++;
172 m_sorted = false;
173}

210template <typename P, typename Traits>
211inline void
212ParticleSoA<P, Traits>::h5DelinearizeAndAppend(const void* a_buffer)
213{
214 this->growIfFull();
215 const auto* p = static_cast<const unsigned char*>(a_buffer);
216 for (int dir = 0; dir < SpaceDim; dir++) {
217 detail::pullBytes(p, this->positionColumn(dir)[m_size]);
218 }
219 detail::pullBytes(p, this->template columnByIndex<s_weightCol>()[m_size]);
220 this->template columnByIndex<s_idCol>()[m_size] = s_invalidID;
221 this->template columnByIndex<s_rankCol>()[m_size] = static_cast<RankID>(-1);
222
223 // Non-checkpointed payload columns default-construct; then overwrite the checkpointed subset.
224 const P def{};
225 this->writePayload(def, std::make_index_sequence<s_numPayloadColumns>{});
226 this->h5DelinearizePayload(p, H5PayloadColumns{});
227
228 m_size++;
229 m_sorted = false;
230}

232template <typename P, typename Traits>
233inline void
234ParticleSoA<P, Traits>::sortByCell(const Box& a_box, const RealVect& a_dx, const RealVect& a_probLo)
235{
236 // No CH_TIME here: this leaf container is the per-particle hot path; timing of bulk
237 // operations like this one is done by the container/mesh drivers above.
238 const std::size_t nCells = a_box.numPts();
239 const IntVect loEnd = a_box.smallEnd();
240 const IntVect boxSize = a_box.size();
241
242 m_cellStart.assign(nCells + 1, 0);
243 if (m_size == 0) {
244 m_sorted = true;
245 return;
246 }
247
248 // Pass 1: compute each particle's Fortran cell key and tally per-cell counts.
249 std::vector<std::size_t> key(m_size);
250 for (std::size_t i = 0; i < m_size; i++) {
251 const RealVect x = this->position(i);
252 std::size_t lin = 0;
253 std::size_t stride = 1;
254 for (int d = 0; d < SpaceDim; d++) {
255 int c = static_cast<int>(std::floor((x[d] - a_probLo[d]) / a_dx[d])) - loEnd[d];
256 if (c < 0) {
257 c = 0;
258 }
259 if (c >= boxSize[d]) {
260 c = boxSize[d] - 1;
261 }
262 lin += static_cast<std::size_t>(c) * stride;
263 stride *= static_cast<std::size_t>(boxSize[d]);
264 }
265 key[i] = lin;
266 m_cellStart[lin + 1]++;
267 }
268
269 // Prefix-sum the counts into CSR start offsets (m_cellStart[c] = start of cell c).
270 for (std::size_t c = 1; c <= nCells; c++) {
271 m_cellStart[c] += m_cellStart[c - 1];
272 }
273
274 // Pass 2: stable destination slot per particle, then physically reorder all columns.
275 std::vector<std::size_t> cursor(m_cellStart.begin(), m_cellStart.begin() + nCells);
276 std::vector<std::size_t> dest(m_size);
277 for (std::size_t i = 0; i < m_size; i++) {
278 dest[i] = cursor[key[i]]++;
279 }
280 this->permute(dest);
281
282 m_sorted = true;
283}
