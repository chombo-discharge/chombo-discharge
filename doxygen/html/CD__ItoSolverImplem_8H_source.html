31inline RealVect
32ItoSolver::randomGaussian() const
33{
34 // TLDR: We draw a random number from a Gaussian distribution for each coordinate, and truncate the distribution at m_normalDistributionTruncation.
35
36 auto sign = [](const Real& a) -> Real {
37 return (a > 0) - (a < 0);
38 };
39
40 RealVect r = RealVect::Zero;
41 for (int i = 0; i < SpaceDim; i++) {
42 r[i] = Random::getNormal01();
43
44 r[i] = sign(r[i]) * std::min(std::abs(r[i]), m_normalDistributionTruncation);
45 }
46
47 return r;
48}

50template <typename Gather>
51void
52ItoSolver::depositGatheredNGP(LevelData<EBCellFAB>& a_output,
53 const ParticleContainer<ItoParticle>& a_particles,
54 const int a_level,
55 Gather a_gather) const noexcept
56{
57 CH_TIME("ItoSolver::depositGatheredNGP");
58 if (m_verbosity > 5) {
59 pout() << m_name + "::depositGatheredNGP" << endl;
60 }
61
62 CH_assert(a_level >= 0);
63 CH_assert(a_level <= m_amr->getFinestLevel());
64
65 const ProblemDomain& domain = m_amr->getDomains()[a_level];
66 const DisjointBoxLayout& dbl = m_amr->getGrids(a_particles.getRealm())[a_level];
67 const DataIterator& dit = dbl.dataIterator();
68 const EBISLayout& ebisl = m_amr->getEBISLayout(a_particles.getRealm(), m_phase)[a_level];
69 const Real dx = m_amr->getDx()[a_level];
70 const RealVect probLo = m_amr->getProbLo();
71
72 CH_assert(a_output.disjointBoxLayout() == dbl);
73
74 const int nbox = dit.size();
75
76#pragma omp parallel for schedule(runtime)
77 for (int mybox = 0; mybox < nbox; mybox++) {
78 const DataIndex& din = dit[mybox];
79
80 const Box cellBox = dbl[din];
81 const EBISBox& ebisbox = ebisl[din];
82
83 EBParticleMesh particleMesh(domain, cellBox, ebisbox, dx * RealVect::Unit, probLo);
84
85 EBCellFAB& output = a_output[din];
86 const ParticleSoA<ItoParticle>& leaf = a_particles[a_level][din];
87
88 // The per-patch deposit INCREMENTS, so start from a clean slate.
89 output.setVal(0.0);
90
91 particleMesh
92 .depositGathered(output, 0, leaf, DepositionType::NGP, 1.0, true, [&](const std::size_t a_i, Real* a_out) {
93 a_out[0] = a_gather(leaf, a_i);
94 });
95 }
96}

98template <typename Gather>
99void
100ItoSolver::depositGathered(EBAMRCellData& a_phi,
101 const ParticleContainer<ItoParticle>& a_particles,
102 const DepositionType a_deposition,
103 const CoarseFineDeposition a_coarseFineDeposition,
104 Gather a_gather) const
105{
106 CH_TIME("ItoSolver::depositGathered");
107 if (m_verbosity > 5) {
108 pout() << m_name + "::depositGathered" << endl;
109 }
110
111 CH_assert(a_phi[0]->nComp() == 1);
112 CH_assert(!a_particles.isOrganizedByCell());
113
114 // Deposit the gathered quantity onto the mesh (resets a_phi internally), then run the shared
115 // redistribution + average-down + ghost-interpolation tail.
116 m_amr->depositGathered(a_phi,
117 m_realm,
118 m_phase,
119 a_particles,
120 a_deposition,
121 a_coarseFineDeposition,
122 m_forceIrregDepositionNGP,
123 a_gather);
124
125 this->finalizeDeposit(a_phi);
126}
