29 inline RealVect
30 ItoSolver::randomGaussian() const
31 {
32  // TLDR: We draw a random number from a Gaussian distribution for each coordinate, and truncate the distribution at m_normalDistributionTruncation.
33 
34  auto sign = [](const Real& a) -> Real {
35  return (a > 0) - (a < 0);
36  };
37 
38  RealVect r = RealVect::Zero;
39  for (int i = 0; i < SpaceDim; i++) {
40  r[i] = Random::getNormal01();
41 
42  r[i] = sign(r[i]) * std::min(std::abs(r[i]), m_normalDistributionTruncation);
43  }
44 
45  return r;
46 }

127 ItoSolver::depositKappaConservative(EBAMRCellData& a_phi,
128  ParticleContainer<P>& a_particles,
129  const DepositionType a_deposition,
130  const CoarseFineDeposition a_coarseFineDeposition) const
131 {
132  CH_TIME("ItoSolver::depositKappaConservative");
133  if (m_verbosity > 5) {
134  pout() << m_name + "::depositKappaConservative" << endl;
135  }
136 
137  CH_assert(a_phi[0]->nComp() == 1);
138 
139  // Now do the deposition. Recall that when we deposit with "halos", we need to fetch the subset of coarse-level particles that surround the
140  // refinement boundary.
141  switch (a_coarseFineDeposition) {
142  case CoarseFineDeposition::Interp: {
143  m_amr->depositParticles<P, particleScalarField>(a_phi,
144  m_realm,
145  m_phase,
146  a_particles,
147  a_deposition,
148  CoarseFineDeposition::Interp,
149  m_forceIrregDepositionNGP);
150  break;
151  }
152  case CoarseFineDeposition::Halo: {
153 
154  // Copy the mask particles.
155  const AMRMask& mask = m_amr->getMask(s_particle_halo, m_haloBuffer, m_realm);
156  a_particles.copyMaskParticles(mask);
157 
158  m_amr->depositParticles<P, particleScalarField>(a_phi,
159  m_realm,
160  m_phase,
161  a_particles,
162  a_deposition,
163  CoarseFineDeposition::Halo,
164  m_forceIrregDepositionNGP);
165 
166  a_particles.clearMaskParticles();
167 
168  break;
169  }
170  case CoarseFineDeposition::HaloNGP: {
171 
172  // Transfer mask particles
173  const AMRMask& mask = m_amr->getMask(s_particle_halo, m_haloBuffer, m_realm);
174  a_particles.transferMaskParticles(mask);
175 
176  m_amr->depositParticles<P, particleScalarField>(a_phi,
177  m_realm,
178  m_phase,
179  a_particles,
180  a_deposition,
181  CoarseFineDeposition::HaloNGP,
182  m_forceIrregDepositionNGP);
183 
184  a_particles.transferParticles(a_particles.getMaskParticles());
185 
186  break;
187  }
188  default: {
189  MayDay::Error("ItoSolverImplem.H in function ItoSolver::depositKappaConservative -- logic bust!");
190 
191  break;
192  }
