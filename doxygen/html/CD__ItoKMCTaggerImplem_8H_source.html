185 template <typename S>
186 bool
187 ItoKMCTagger<S>::tagCells(EBAMRTags& a_tags) noexcept
188 {
189  CH_TIME("ItoKMCTagger::tagCells");
190  if (m_verbosity > 5) {
191  pout() << m_name + "::tagCells" << endl;
192  }
193 
194  CH_assert(m_isDefined);
195 
196  bool gotNewTags = false;
197 
198  const RealVect probLo = m_amr->getProbLo();
199  const Real curTime = m_timeStepper->getTime();
200  const int finestLevel = m_amr->getFinestLevel();
201  const int maxAmrDepth = m_amr->getMaxAmrDepth();
202  const int finestTagLevel = (finestLevel == maxAmrDepth) ? maxAmrDepth - 1 : finestLevel;
203 
204  if (m_numTagFields > 0) {
205  this->computeTagFields();
206 
207  for (int lvl = 0; lvl <= finestTagLevel; lvl++) {
208  const DisjointBoxLayout& dbl = m_amr->getGrids(m_realm)[lvl];
209  const EBISLayout& ebisl = m_amr->getEBISLayout(m_realm, m_phase)[lvl];
210  const Real dx = m_amr->getDx()[lvl];
211 
212  for (DataIterator dit(dbl); dit.ok(); ++dit) {
213  const Box box = dbl.get(dit());
214  const EBISBox& ebisbox = ebisl[dit()];
215 
216  const IntVectSet irregCells = ebisbox.getIrregIVS(box);
217  const IntVectSet backupTags = IntVectSet((*a_tags[lvl])[dit()]);
218 
219  DenseIntVectSet coarsenCells(box, false);
220  DenseIntVectSet refinedCells(box, false);
221 
222  Vector<EBCellFAB*> tagFields;
223  Vector<EBCellFAB*> gradTagFields;
224 
225  for (int i = 0; i < m_numTagFields; i++) {
226  tagFields.push_back(&((*m_tagFields[i][lvl])[dit()]));
227  gradTagFields.push_back(&((*m_gradTagFields[i][lvl])[dit()]));
228  }
229 
230  this->tagCellsBox(refinedCells,
231  coarsenCells,
232  tagFields,
233  gradTagFields,
234  lvl,
235  dit(),
236  box,
237  ebisbox,
238  curTime,
239  dx,
240  probLo);
241 
242  // Check if we got any new tags, or we are just recycling old tags.
243  // Basically we will check if (current_tags + refined_tags - coarsenCells) == current_tags
244  DenseIntVectSet& tags = (*a_tags[lvl])[dit()];
245 
246  DenseIntVectSet cpy1 = tags;
247  tags -= coarsenCells;
248  tags |= refinedCells;
249  DenseIntVectSet cpy2 = tags;
250 
251  cpy2 -= cpy1; // = new tags minus old tags. If nonzero, we got some new tags.
252  cpy1 -= tags; // = old_tags minus new tags. If nonzero, we got some new tags
253  if (cpy1.numPts() != 0 || cpy2.numPts() != 0) {
254  gotNewTags = true;
255  }
256  }
257  }
258  }
259 
260  // Some ranks may have gotten new tags while others have not. This little code snippet
261  // sets gotNewTags = true for all ranks if any rank probLoally had gotNewTags = true
262  const int foundTags = gotNewTags ? 1 : 0;
263 
264  return (ParallelOps::max(foundTags) > 0);
265 }

267 template <typename S>
268 void
269 ItoKMCTagger<S>::tagCellsBox(DenseIntVectSet& a_refinedCells,
270  DenseIntVectSet& a_coarsenedCells,
271  const Vector<EBCellFAB*>& a_tagFields,
272  const Vector<EBCellFAB*>& a_gradTagFields,
273  const int a_lvl,
274  const DataIndex a_dit,
275  const Box a_box,
276  const EBISBox& a_ebisbox,
277  const Real a_time,
278  const Real a_dx,
279  const RealVect a_probLo) const noexcept
280 {
281  CH_TIME("ItoKMCTagger::refineCellsBox");
282  if (m_verbosity > 5) {
283  pout() << "ItoKMCTagger::refineCellsBox" << endl;
284  }
285 
286  CH_assert(m_isDefined);
287 
288  Vector<FArrayBox*> tagFieldsReg;
289  Vector<FArrayBox*> gradTagFieldsReg;
290 
291  for (int i = 0; i < m_numTagFields; i++) {
292  tagFieldsReg.push_back(&(a_tagFields[i]->getFArrayBox()));
293  gradTagFieldsReg.push_back(&(a_gradTagFields[i]->getFArrayBox()));
294  }
295 
296  // Regular kernel.
297  auto regularKernel = [&](const IntVect& iv) -> void {
298  const RealVect pos = a_probLo + (0.5 * RealVect::Unit + RealVect(iv)) * a_dx;
299 
300  if (this->insideTagBox(pos) && a_ebisbox.isRegular(iv)) {
301 
302  Vector<Real> tr(m_numTagFields);
303  Vector<RealVect> gt(m_numTagFields);
304 
305  for (int i = 0; i < m_numTagFields; i++) {
306  tr[i] = (*tagFieldsReg[i])(iv, 0);
307  gt[i] = RealVect(
308  D_DECL((*gradTagFieldsReg[i])(iv, 0), (*gradTagFieldsReg[i])(iv, 1), (*gradTagFieldsReg[i])(iv, 2)));
309  }
310 
311  if (this->refineCell(pos, a_time, a_dx, a_lvl, tr, gt)) {
312  a_refinedCells |= iv;
313  }
314 
315  if (this->coarsenCell(pos, a_time, a_dx, a_lvl, tr, gt)) {
316  a_coarsenedCells |= iv;
317  }
318  }
319  };
320 
321  // Irregular box loop
322  auto irregularKernel = [&](const VolIndex& vof) -> void {
323  const RealVect pos = a_probLo + Location::position(Location::Cell::Center, vof, a_ebisbox, a_dx);
324 
325  if (this->insideTagBox(pos)) {
326 
327  Vector<Real> tr(m_numTagFields);
328  Vector<RealVect> gt(m_numTagFields);
329 
330  for (int i = 0; i < m_numTagFields; i++) {
331  tr[i] = (*a_tagFields[i])(vof, 0);
332  gt[i] = RealVect(
333  D_DECL((*a_gradTagFields[i])(vof, 0), (*a_gradTagFields[i])(vof, 1), (*a_gradTagFields[i])(vof, 2)));
334  }
335 
336  if (this->refineCell(pos, a_time, a_dx, a_lvl, tr, gt)) {
337  a_refinedCells |= vof.gridIndex();
338  }
339 
340  if (this->coarsenCell(pos, a_time, a_dx, a_lvl, tr, gt)) {
341  a_coarsenedCells |= vof.gridIndex();
342  }
343  }
344  };
345 
346  // Run the kernels.
347  VoFIterator& vofit = (*m_amr->getVofIterator(m_realm, m_phase)[a_lvl])[a_dit];
348 
349  BoxLoops::loop(a_box, regularKernel);
350  BoxLoops::loop(vofit, irregularKernel);
351 }
