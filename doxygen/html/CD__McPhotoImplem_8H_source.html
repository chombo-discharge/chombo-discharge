21 McPhoto::depositPhotons(EBAMRCellData& a_phi, ParticleContainer<P>& a_photons, const DepositionType& a_deposition)
22 {
23  CH_TIME("McPhoto::depositPhotons(ParticleContainer)");
24  if (m_verbosity > 5) {
25  pout() << m_name + "::depositPhotons(ParticleContainer)" << endl;
26  }
27 
28  // a_phi contains only weights, i.e. not divided by kappa
29  this->depositKappaConservative<P, particleScalarField>(a_phi, a_photons, a_deposition, m_coarseFineDeposition);
30 
31  // Compute m_depositionNC = sum(kappa*Wc)/sum(kappa)
32  this->depositNonConservative(m_depositionNC, a_phi);
33 
34  // Compute hybrid deposition, including mass differnce
35  this->depositHybrid(a_phi, m_massDiff, m_depositionNC);
36 
37  // Redistribute
38  if (m_blendConservation) {
39  Vector<RefCountedPtr<EBFluxRedistribution>>& redistOps = m_amr->getRedistributionOp(m_realm, m_phase);
40  for (int lvl = 0; lvl <= m_amr->getFinestLevel(); lvl++) {
41  const Real scale = 1.0;
42  const Interval variables = Interval(0, 0);
43  const bool hasCoar = lvl > 0;
44  const bool hasFine = lvl < m_amr->getFinestLevel();
45 
46  if (hasCoar) {
47  redistOps[lvl]->redistributeCoar(*a_phi[lvl - 1], *m_massDiff[lvl], scale, variables);
48  }
49 
50  redistOps[lvl]->redistributeLevel(*a_phi[lvl], *m_massDiff[lvl], scale, variables);
51 
52  if (hasFine) {
53  redistOps[lvl]->redistributeFine(*a_phi[lvl + 1], *m_massDiff[lvl], scale, variables);
54  }
55  }
56  }
57 
58  // Average down and interpolate
59  m_amr->conservativeAverage(a_phi, m_realm, m_phase);
60  m_amr->interpGhost(a_phi, m_realm, m_phase);
61 }

108  case CoarseFineDeposition::HaloNGP: {
