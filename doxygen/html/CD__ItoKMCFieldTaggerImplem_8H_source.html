95 ItoKMCFieldTagger<S>::computeTagFields() const noexcept
96 {
97  CH_TIME("ItoKMCFieldTagger::computeTagFields");
98  if (this->m_verbosity > 5) {
99  pout() << this->m_name + "::computeTagFields" << endl;
100  }
101

103 
104  this->allocateStorage();
105  this->computeElectricField(m_E, m_gradE);
106 
107  const RealVect probLo = this->m_amr->getProbLo();
108  const Real time = this->m_timeStepper->getTime();
109 
110  Real maxE, minE;
111  Real maxGradE, minGradE;
112 
113  DataOps::getMaxMinNorm(maxE, minE, m_E);
114  DataOps::getMaxMinNorm(maxGradE, minGradE, m_gradE);
115 
116  for (int lvl = 0; lvl <= this->m_amr->getFinestLevel(); lvl++) {
117  const DisjointBoxLayout& dbl = this->m_amr->getGrids(this->m_realm)[lvl];
118  const EBISLayout& ebisl = this->m_amr->getEBISLayout(this->m_realm, this->m_phase)[lvl];
119  const Real dx = this->m_amr->getDx()[lvl];
120 
121  for (DataIterator dit(dbl); dit.ok(); ++dit) {
122  const Box& box = dbl[dit()];
123  const EBISBox& ebisbox = ebisl[dit()];
124 
125  const EBCellFAB& electricField = (*m_E[lvl])[dit()];
126  const EBCellFAB& gradElectricField = (*m_gradE[lvl])[dit()];
127 
128  const FArrayBox& electricFieldReg = electricField.getFArrayBox();
129  const FArrayBox& gradElectricFieldReg = gradElectricField.getFArrayBox();
130 
131  Vector<EBCellFAB*> tagFields;
132  Vector<FArrayBox*> tagFieldsReg;
133  for (int i = 0; i < this->m_numTagFields; i++) {
134  tagFields.push_back(&((*this->m_tagFields[i][lvl])[dit()]));
135  tagFieldsReg.push_back(&(tagFields[i]->getFArrayBox()));
136  }
137 
138  auto regularKernel = [&](const IntVect& iv) -> void {
139  if (ebisbox.isRegular(iv)) {
140  const RealVect pos = probLo + RealVect(iv) * dx;
141 
142  const RealVect E = RealVect(
143  D_DECL(electricFieldReg(iv, 0), electricFieldReg(iv, 1), electricFieldReg(iv, 2)));
144  const RealVect gradE = RealVect(
145  D_DECL(gradElectricFieldReg(iv, 0), gradElectricFieldReg(iv, 1), gradElectricFieldReg(iv, 2)));
146 
147  const Vector<Real> compTagFields =
148  this->computeTagFields(pos, time, dx, E, minE, maxE, gradE, minGradE, maxGradE);
149 
150  for (int i = 0; i < this->m_numTagFields; i++) {
151  (*tagFieldsReg[i])(iv, 0) = compTagFields[i];
152  }
153  }
154  };
155 
156  // Irregular box loop
157  auto irregularKernel = [&](const VolIndex& vof) -> void {
158  const RealVect pos = probLo + Location::position(Location::Cell::Center, vof, ebisbox, dx);
159 
160  const RealVect E = RealVect(D_DECL(electricField(vof, 0), electricField(vof, 1), electricField(vof, 2)));
161  const RealVect gradE = RealVect(
162  D_DECL(gradElectricField(vof, 0), gradElectricField(vof, 1), gradElectricField(vof, 2)));
163 
164  const Vector<Real> compTagFields =
165  this->computeTagFields(pos, time, dx, E, minE, maxE, gradE, minGradE, maxGradE);
166 
167  for (int i = 0; i < this->m_numTagFields; i++) {
168  (*tagFields[i])(vof, 0) = compTagFields[i];
169  }
170  };
171 
172  // Run the kernels
173  VoFIterator& vofit = (*this->m_amr->getVofIterator(this->m_realm, this->m_phase)[lvl])[dit()];
174  BoxLoops::loop(box, regularKernel);
175  BoxLoops::loop(vofit, irregularKernel);
176  }
177 
178  for (int i = 0; i < this->m_numTagFields; i++) {
179  this->m_amr->conservativeAverage(this->m_tagFields[i], this->m_realm, this->m_phase);
180  this->m_amr->interpGhost(this->m_tagFields[i], this->m_realm, this->m_phase);
181  }
182 
183  // Compute gradient of tracers
184  for (int i = 0; i < this->m_numTagFields; i++) {
185  this->m_amr->computeGradient(this->m_gradTagFields[i], this->m_tagFields[i], this->m_realm, this->m_phase);
186  this->m_amr->conservativeAverage(this->m_gradTagFields[i], this->m_realm, this->m_phase);
187  }
188  }
189 
190  this->deallocateStorage();
191 }
