59template <class T, class BV, int K>
60Real
61SignedDistanceBVH<T, BV, K>::value(const RealVect& a_point) const
62{
63
64 // TLDR: In 2D we ignore z-variations, freezing that coordinate to some value. We then use the bounding volume hierarchy methodology for computing the signed distance.
65#if CH_SPACEDIM == 2
66 Vec3 p(a_point[0], a_point[1], m_zCoord);
67#else
68 Vec3 p(a_point[0], a_point[1], a_point[2]);
69#endif
70
71 high_resolution_clock::time_point t1 = high_resolution_clock::now();
72 auto d = m_root->signedDistance(p);
73 high_resolution_clock::time_point t2 = high_resolution_clock::now();
74 duration<double> time_span = duration_cast<duration<double>>(t2 - t1);
75
76 if (m_flipInside) {
77 d = -d;
78 }
79
80 m_timespan += time_span;
81 m_numCalled++;
82
83 return Real(d);
84}
