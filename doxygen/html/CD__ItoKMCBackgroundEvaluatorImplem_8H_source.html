91template <typename I, typename C, typename R, typename F>
92Real
93ItoKMCBackgroundEvaluator<I, C, R, F>::advance(const Real a_dt)
94{
95 CH_TIME("ItoKMCBackgroundEvaluator::advance");
96 if (this->m_verbosity > 5) {
97 pout() << "ItoKMCBackgroundEvaluator::advance" << endl;
98 }
99
100 const Real actualDt = ItoKMCGodunovStepper<I, C, R, F>::advance(a_dt);
101
102 const std::pair<Real, Real> fieldChange = this->evaluateSpaceChargeEffects();
103 const Real electrodeQ = this->integrateElectrodeSurfaceCharge();
104 const Real ohmicQ = this->integrateOhmicCharge();
105 const std::pair<Real, Real> opticalEmissions = this->integrateOpticalExcitations();
106
107 m_relFieldChange = fieldChange.first;
108 m_maxFieldChange = fieldChange.second;
109 m_electrodeCharge = electrodeQ;
110 m_ohmicCharge = ohmicQ;
111 m_sumOpticalPhi = opticalEmissions.first;
112 m_sumOpticalSrc = opticalEmissions.second;
113
114 return actualDt;

117template <typename I, typename C, typename R, typename F>
118Real
119ItoKMCBackgroundEvaluator<I, C, R, F>::computeDt()
120{
121 CH_TIME("ItoKMCBackgroundEvaluator::computeDt");
122 if (this->m_verbosity > 5) {
123 pout() << "ItoKMCBackgroundEvaluator::computeDt" << endl;
124 }
125
126 Real newDt = 0.0;
127
128 const bool noElectronsPresent = false;
129
130 bool abortCondition = false; // make sure all simultaneous abort reasons are stated in the log
131
132 if ((m_maxFieldChange > m_maxFieldExitCrit) && (m_maxFieldExitCrit > 0.0)) {
133 pout()
134 << "ItoKMCBackgroundEvaluator -- abort because the max field (in any cell) changed by more than specified threshold (max_field_exit_crit)"
135 << endl;
136 abortCondition = true;
137 }
138
139 if ((m_relFieldChange > m_relFieldExitCrit) && (m_relFieldExitCrit > 0.0)) {
140 pout()
141 << "ItoKMCBackgroundEvaluator -- abort because the relative field change in a cell is larger than specified threshold (rel_field_exit_crit)"
142 << endl;
143 abortCondition = true;
144 }
145
146 if (noElectronsPresent) { // Abort if we're out of electrons.
147 pout() << "ItoKMCBackgroundEvaluator -- abort because no electrons are present" << endl;
148 abortCondition = true;
149 }
150 if (abortCondition) {
151 newDt = 0.0;
152 }
153 else {
154 newDt = ItoKMCGodunovStepper<I, C, R, F>::computeDt();
155 }
156
157 if (this->m_timeStep == 0 && m_maxInitialTimeStep > 0.0 && newDt > m_maxInitialTimeStep)
158 newDt = m_maxInitialTimeStep;
159
160 return newDt;
161}

189template <typename I, typename C, typename R, typename F>
190void
191ItoKMCBackgroundEvaluator<I, C, R, F>::computeBackgroundField() noexcept
192{
193 CH_TIME("ItoKMCBackgroundEvaluator::computeBackgroundField");
194 if (this->m_verbosity > 5) {
195 pout() << "ItoKMCBackgroundEvaluator::computeBackgroundField" << endl;
196 }
197
198 // Allocate the background field, storage for the potential, storage for the space-charge, and storage for the
199 // surface charge.
200 MFAMRCellData phi;
201 MFAMRCellData rho;
202 EBAMRIVData sigma;
203
204 (this->m_amr)->allocate(phi, this->m_fluidRealm, 1);
205 (this->m_amr)->allocate(rho, this->m_fluidRealm, 1);
206 (this->m_amr)->allocate(sigma, this->m_fluidRealm, this->m_plasmaPhase, 1);
207 (this->m_amr)->allocate(m_backgroundField, this->m_fluidRealm, SpaceDim);
208
209 // Copy the potential over from the field solver so we have a better initial guess for the potential. The
210 // space-charge and surface charge must be zero.
211 (this->m_amr)->copyData(phi, (this->m_fieldSolver)->getPotential());
212
213 DataOps::setValue(rho, 0.0);
214 DataOps::setValue(sigma, 0.0);
215
216 // Reset the field solver permittivities. Note that this discards the semi-implicit coefficients for the field update,
217 // but these are refilled on the next solver time step anyways.
218 (this->m_fieldSolver)->setPermittivities();
219
220 const MFAMRCellData& permCell = (this->m_fieldSolver)->getPermittivityCell();
221 const MFAMRFluxData& permFace = (this->m_fieldSolver)->getPermittivityFace();
222 const MFAMRIVData& permEB = (this->m_fieldSolver)->getPermittivityEB();
223
224 (this->m_fieldSolver)->setSolverPermittivities(permCell, permFace, permEB);
225
226 // Solve the damn thing, and then compute the electric field onto m_backgroundField.
227 (this->m_fieldSolver)->solve(phi, rho, sigma, false);
228 (this->m_fieldSolver)->computeElectricField(m_backgroundField, phi);
229}

231template <typename I, typename C, typename R, typename F>
232std::pair<Real, Real>
233ItoKMCBackgroundEvaluator<I, C, R, F>::evaluateSpaceChargeEffects() noexcept
234{
235 CH_TIME("ItoKMCBackgroundEvaluator::evaluateSpaceChargeEffects");
236 if (this->m_verbosity > 5) {
237 pout() << "ItoKMCBackgroundEvaluator::evaluateSpaceChargeEffects" << endl;
238 }
239
240 EBAMRCellData poissonNorm;
241 EBAMRCellData laplaceNorm;
242 EBAMRCellData poissonField;
243 EBAMRCellData laplaceField;
244 EBAMRCellData deltaE;
245
246 (this->m_amr)->allocate(poissonNorm, this->m_fluidRealm, this->m_plasmaPhase, 1);
247 (this->m_amr)->allocate(laplaceNorm, this->m_fluidRealm, this->m_plasmaPhase, 1);
248 (this->m_amr)->allocate(poissonField, this->m_fluidRealm, this->m_plasmaPhase, SpaceDim);
249 (this->m_amr)->allocate(laplaceField, this->m_fluidRealm, this->m_plasmaPhase, SpaceDim);
250 (this->m_amr)->allocate(deltaE, this->m_fluidRealm, this->m_plasmaPhase, 1);
251
252 // Get the fields in the plasma phase -- the actual field and the background field.
253 const MFAMRCellData electricField = (this->m_fieldSolver)->getElectricField();
254
255 DataOps::copy(poissonField, (this->m_amr)->alias(this->m_plasmaPhase, electricField));
256 DataOps::copy(laplaceField, (this->m_amr)->alias(this->m_plasmaPhase, m_backgroundField));
257
258 (this->m_amr)->interpToCentroids(poissonField, this->m_fluidRealm, this->m_plasmaPhase);
259 (this->m_amr)->interpToCentroids(laplaceField, this->m_fluidRealm, this->m_plasmaPhase);
260
261 // Compute the field magnitude, and then the relative change in field magnitude.
262 DataOps::vectorLength(poissonNorm, poissonField);
263 DataOps::vectorLength(laplaceNorm, laplaceField);
264
265 DataOps::copy(deltaE, poissonNorm);
266 DataOps::incr(deltaE, laplaceNorm, -1.0);
267 DataOps::divideFallback(deltaE, laplaceNorm, 0.0);
268
269 (this->m_amr)->arithmeticAverage(deltaE, this->m_fluidRealm, this->m_plasmaPhase);
270 (this->m_amr)->interpGhost(deltaE, this->m_fluidRealm, this->m_plasmaPhase);
271
272 // Iterate through the grid cells and figure out where the field change the most.
273 //
274 // PS: I'm doing this with a direct loop because DataOps::getMaxMin will do ALL cells, including ones that are
275 // covered by a finer level. But we only want to evaluate the valid region.
276 Real relChange = -1.0;
277 Real maxChange = -1.0;
278
279 Real maxSpaceChargeField = 0.0;
280 Real minSpaceChargeField = 0.0;
281 Real maxBackgroundField = 0.0;
282 Real minBackgroundField = 0.0;
283
284 DataOps::getMaxMin(maxSpaceChargeField, minSpaceChargeField, poissonNorm, 0);
285 DataOps::getMaxMin(maxBackgroundField, minBackgroundField, laplaceNorm, 0);
286
287 for (int lvl = 0; lvl <= (this->m_amr)->getFinestLevel(); lvl++) {
288 const DisjointBoxLayout& dbl = (this->m_amr)->getGrids(this->m_fluidRealm)[lvl];
289 const DataIterator& dit = dbl.dataIterator();
290 const EBISLayout& ebisl = (this->m_amr)->getEBISLayout(this->m_fluidRealm, this->m_plasmaPhase)[lvl];
291
292 const int nbox = dit.size();
293
294#pragma omp parallel for schedule(runtime) reduction(max : relChange)
295 for (int mybox = 0; mybox < nbox; mybox++) {
296 const DataIndex& din = dit[mybox];
297 const Box box = dbl[din];
298 const EBISBox& ebisbox = ebisl[din];
299 const BaseFab<bool>& validCells = (*(this->m_amr)->getValidCells(this->m_fluidRealm)[lvl])[din];
300
301 const EBCellFAB& data = (*deltaE[lvl])[din];
302 const FArrayBox& dataReg = data.getFArrayBox();
303
304 auto regularKernel = [&](const IntVect& iv) -> void {
305 if (validCells(iv) && ebisbox.isRegular(iv)) {
306 relChange = std::max(relChange, std::abs(dataReg(iv, 0)));
307 }
308 };
309
310 auto irregularKernel = [&](const VolIndex& vof) -> void {
311 if (validCells(vof.gridIndex()) && ebisbox.isIrregular(vof.gridIndex())) {
312 relChange = std::max(relChange, std::abs(data(vof, 0)));
313 }
314 };
315
316 VoFIterator& vofit = (*(this->m_amr)->getVofIterator(this->m_fluidRealm, this->m_plasmaPhase)[lvl])[din];
317
318 BoxLoops::loop(box, regularKernel);
319 BoxLoops::loop(vofit, irregularKernel);
320 }
321 }
322
323 relChange = ParallelOps::max(relChange);
324 maxChange = maxSpaceChargeField / maxBackgroundField - 1.0;
325
326 return std::make_pair(relChange, maxChange);
327}

329template <typename I, typename C, typename R, typename F>
330Real
331ItoKMCBackgroundEvaluator<I, C, R, F>::integrateElectrodeSurfaceCharge() const noexcept
332{
333 CH_TIME("ItoKMCBackgroundEvaluator::integrateElectrodeSurfaceCharge");
334 if (this->m_verbosity > 5) {
335 pout() << "ItoKMCBackgroundEvaluator::integrateElectrodeSurfaceCharge" << endl;
336 }
337
338 Real Enorm = 0.0;
339
340 const Real eps = Units::eps0 * this->m_computationalGeometry->getGasPermittivity();
341
342 const MFAMRCellData& electricFieldCell = (this->m_fieldSolver)->getElectricField();
343 const MFAMRCellData& permittivityCell = (this->m_fieldSolver)->getPermittivityCell();
344 const EBAMRCellData permittivity = (this->m_amr)->alias(this->m_plasmaPhase, permittivityCell);
345
346 // Allocates data and computes the electric field on the cell centroid(s).
347 EBAMRCellData electricField;
348 (this->m_amr)->allocate(electricField, this->m_fluidRealm, this->m_plasmaPhase, SpaceDim);
349 DataOps::copy(electricField, (this->m_amr)->alias(this->m_plasmaPhase, electricFieldCell));
350 (this->m_amr)->interpToCentroids(electricField, this->m_fluidRealm, this->m_plasmaPhase);
351
352 for (int lvl = 0; lvl <= (this->m_amr)->getFinestLevel(); lvl++) {
353 const DisjointBoxLayout& dbl = (this->m_amr)->getGrids(this->m_fluidRealm)[lvl];
354 const EBISLayout& ebisl = (this->m_amr)->getEBISLayout(this->m_fluidRealm, this->m_plasmaPhase)[lvl];
355 const DataIterator& dit = dbl.dataIterator();
356 const Real dx = (this->m_amr)->getDx()[lvl];
357 const Real dA = std::pow(dx, SpaceDim - 1);
358
359 const int numBoxes = dit.size();
360
361#pragma omp parallel for schedule(runtime) reduction(+ : Enorm)
362 for (int mybox = 0; mybox < numBoxes; mybox++) {
363 const DataIndex& din = dit[mybox];
364 const EBISBox& ebisbox = ebisl[din];
365 const EBCellFAB& E = (*electricField[lvl])[din];
366 const EBCellFAB& eps = (*permittivity[lvl])[din];
367 const BaseFab<bool>& validCells = (*(this->m_amr)->getValidCells(this->m_fluidRealm)[lvl])[din];
368
369 auto kernel = [&](const VolIndex& vof) -> void {
370 if (ebisbox.isIrregular(vof.gridIndex()) && validCells(vof.gridIndex(), 0)) {
371 const RealVect normalVector = ebisbox.normal(vof);
372 const RealVect e = RealVect(D_DECL(E(vof, 0), E(vof, 1), E(vof, 2)));
373 const Real areaFrac = ebisbox.bndryArea(vof);
374
375 Enorm += eps(vof, 0) * e.dotProduct(normalVector) * areaFrac * dA;
376 }
377 };
378
379 VoFIterator& vofit = (*(this->m_amr)->getVofIterator(this->m_fluidRealm, this->m_plasmaPhase)[lvl])[din];
380
381 BoxLoops::loop(vofit, kernel);
382 }
383 }
384
385 return eps * ParallelOps::sum(Enorm);
386}

400template <typename I, typename C, typename R, typename F>
401std::pair<Real, Real>
402ItoKMCBackgroundEvaluator<I, C, R, F>::integrateOpticalExcitations() const noexcept
403{
404 CH_TIME("ItoKMCBackgroundEvaluator::integrateOpticalExcitations");
405 if (this->m_verbosity > 5) {
406 pout() << "ItoKMCBackgroundEvaluator::integrateOpticalExcitations" << endl;
407 }
408
409 Real sumPhi = 0.0;
410 Real sumSrc = 0.0;
411
412 for (CdrIterator<CdrSolver> solverIt = (this->m_cdr)->iterator(); solverIt.ok(); ++solverIt) {
413 const RefCountedPtr<CdrSolver>& solver = solverIt();
414
415 if (solver->getName() == m_opticalSolver) {
416 for (int lvl = 0; lvl <= (this->m_amr)->getFinestLevel(); lvl++) {
417 const DisjointBoxLayout& dbl = (this->m_amr)->getGrids(this->m_fluidRealm)[lvl];
418 const EBISLayout& ebisl = (this->m_amr)->getEBISLayout(this->m_fluidRealm, this->m_plasmaPhase)[lvl];
419 const DataIterator& dit = dbl.dataIterator();
420 const Real dx = (this->m_amr)->getDx()[lvl];
421 const Real dV = std::pow(dx, SpaceDim);
422
423 const int numBoxes = dit.size();
424
425#pragma omp parallel for schedule(runtime) reduction(+ : sumPhi, sumSrc)
426 for (int mybox = 0; mybox < numBoxes; mybox++) {
427 const DataIndex& din = dit[mybox];
428 const Box cellBox = dbl[din];
429 const EBISBox& ebisbox = ebisl[din];
430 const BaseFab<bool>& validCells = (*(this->m_amr)->getValidCells(this->m_fluidRealm)[lvl])[din];
431
432 const EBCellFAB& phi = (*(solver->getPhi()[lvl]))[din];
433 const EBCellFAB& src = (*(solver->getSource()[lvl]))[din];
434
435 const FArrayBox& phiReg = phi.getFArrayBox();
436 const FArrayBox& srcReg = src.getFArrayBox();
437
438 auto regularKernel = [&](const IntVect& iv) -> void {
439 if (ebisbox.isRegular(iv) && validCells(iv)) {
440 sumPhi += phiReg(iv, 0) * dV;
441 sumSrc += srcReg(iv, 0) * dV;
442 }
443 };
444
445 auto irregularKernel = [&](const VolIndex& vof) -> void {
446 if (ebisbox.isIrregular(vof.gridIndex()) && validCells(vof.gridIndex(), 0)) {
447 const Real kappa = ebisbox.volFrac(vof);
448
449 sumPhi += phi(vof, 0) * dV;
450 sumSrc += src(vof, 0) * dV;
451 }
452 };
453
454 VoFIterator& vofit = (*(this->m_amr)->getVofIterator(this->m_fluidRealm, this->m_plasmaPhase)[lvl])[din];
455
456 BoxLoops::loop(cellBox, regularKernel);
457 BoxLoops::loop(vofit, irregularKernel);
458 }
459 }
460 }
461 }
462
463 sumPhi = ParallelOps::sum(sumPhi);
464 sumSrc = ParallelOps::sum(sumSrc);
465
466 return std::make_pair(sumPhi, sumSrc);
467}
