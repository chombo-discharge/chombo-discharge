188 void
189 define(const Vector<DisjointBoxLayout>& a_grids,
190 const Vector<ProblemDomain>& a_domains,
191 const Vector<Real>& a_dx,
192 const Vector<int>& a_refRat,
193 const RealVect& a_probLo,
194 const int a_blockingFactor,
195 const int a_finestLevel,
196 const std::string& a_realm,
197 const Vector<RefCountedPtr<LevelData<BaseFab<bool>>>>* a_validCells)
198 {
199 CH_TIME("ParticleContainer::define");
200
201 CH_assert(a_grids.size() > a_finestLevel);
202 CH_assert(a_domains.size() > a_finestLevel);
203 CH_assert(a_dx.size() > a_finestLevel);
204 CH_assert(a_refRat.size() > a_finestLevel);
205
206 m_grids = a_grids;
207 m_domains = a_domains;
208 m_refRat = a_refRat;
209 m_probLo = a_probLo;
210 m_blockingFactor = a_blockingFactor;
211 m_finestLevel = a_finestLevel;
212 m_realm = a_realm;
213 m_validCells = a_validCells;
214 m_numOutcastLocal = 0;
215
216 m_dx.resize(a_finestLevel + 1);
217 for (int lvl = 0; lvl <= a_finestLevel; lvl++) {
218 m_dx[lvl] = a_dx[lvl] * RealVect::Unit;
219 }
220
221 m_particles.resize(a_finestLevel + 1);
222 m_maskParticles.resize(a_finestLevel + 1);
223 m_levelTiles.resize(a_finestLevel + 1);
224 for (int lvl = 0; lvl <= a_finestLevel; lvl++) {
225 m_particles[lvl] = RefCountedPtr<LevelParticles>(new LevelParticles(a_grids[lvl]));
226 m_maskParticles[lvl] = RefCountedPtr<LevelParticles>(new LevelParticles(a_grids[lvl]));
227 m_levelTiles[lvl] = RefCountedPtr<LevelTiles>(new LevelTiles(a_grids[lvl], a_blockingFactor));
228 }
229
230 // Buffer particles live on the grown grids (boxes grown by the refinement factor on finer levels).
231 this->setupGrownGrids();
232 m_bufferParticles.resize(a_finestLevel + 1);
233 for (int lvl = 0; lvl <= a_finestLevel; lvl++) {
234 m_bufferParticles[lvl] = RefCountedPtr<LevelParticles>(new LevelParticles(m_grownGrids[lvl]));
235 }
236
237 m_isOrganizedByCell = false;
238 m_isDefined = true;
239 }

470 void
471 addParticlesDestructive(ParticleSoA<P, Traits>& a_particles)
472 {
473 CH_TIME("ParticleContainer::addParticlesDestructive");
474
475 CH_assert(m_isDefined);
476
477 for (int lvl = 0; lvl <= m_finestLevel; lvl++) {
478 DataIterator dit = m_grids[lvl].dataIterator();
479 if (dit.ok()) {

484
485 this->remap(); // route every ingested particle to its owning patch/level/rank

501 {
502 CH_TIME("ParticleContainer::getNumberOfValidParticlesLocal");
503
504 CH_assert(m_isDefined);
505
506 unsigned long long n = 0;
507 for (int lvl = 0; lvl <= m_finestLevel; lvl++) {

509 const DisjointBoxLayout& dbl = m_grids[lvl];
510 const DataIterator& dit = dbl.dataIterator();

512
513#pragma omp parallel for schedule(runtime) reduction(+ : n)
514 for (int mybox = 0; mybox < nbox; mybox++) {
515 const DataIndex& din = dit[mybox];
516
517 n += static_cast<unsigned long long>(level[din].size());
518 }

718 void
719 transferParticles(AMRParticlesSoA<P, Traits>& a_source)
720 {
721 CH_TIME("ParticleContainer::transferParticles");
722
723 CH_assert(m_isDefined);
724
725 for (int lvl = 0; lvl <= m_finestLevel; lvl++) {
726 LevelParticles& src = *a_source[lvl];
727 LevelParticles& dst = *m_particles[lvl];
728
729 const DisjointBoxLayout& dbl = m_grids[lvl];
730 const DataIterator& dit = dbl.dataIterator();
731
732 const int nbox = dit.size();
733
734#pragma omp parallel for schedule(runtime)
735 for (int mybox = 0; mybox < nbox; mybox++) {
736 const DataIndex& din = dit[mybox];
737
738 dst[din].catenate(src[din]); // bulk move (id/rank preserved); empty dst -> O(1) swap
739 }
740 }
741
742 m_isOrganizedByCell = false;
743 }
