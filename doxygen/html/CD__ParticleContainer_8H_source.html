191 void
192 define(const Vector<DisjointBoxLayout>& a_grids,
193 const Vector<ProblemDomain>& a_domains,
194 const Vector<Real>& a_dx,
195 const Vector<int>& a_refRat,
196 const RealVect& a_probLo,
197 const int a_minBlockSize,
198 const Vector<RefCountedPtr<LevelTiles>>& a_levelTiles,
199 const int a_finestLevel,
200 const std::string& a_realm,
201 const Vector<RefCountedPtr<LevelData<BaseFab<bool>>>>* a_validCells)
202 {
203 CH_TIME("ParticleContainer::define");
204
205 CH_assert(a_grids.size() > a_finestLevel);
206 CH_assert(a_domains.size() > a_finestLevel);
207 CH_assert(a_dx.size() > a_finestLevel);
208 CH_assert(a_refRat.size() > a_finestLevel);
209 CH_assert(a_levelTiles.size() > a_finestLevel);
210
211 m_grids = a_grids;
212 m_domains = a_domains;
213 m_refRat = a_refRat;
214 m_probLo = a_probLo;
215 m_minBlockSize = a_minBlockSize;
216 m_finestLevel = a_finestLevel;
217 m_realm = a_realm;
218 m_validCells = a_validCells;
219 m_numOutcastLocal = 0;
220
221 m_dx.resize(a_finestLevel + 1);
222 for (int lvl = 0; lvl <= a_finestLevel; lvl++) {
223 m_dx[lvl] = a_dx[lvl] * RealVect::Unit;
224 }
225
226 // Alias the Realm's per-level tile->box maps (the single source of truth); never build our own.
227 m_levelTiles = a_levelTiles;
228
229 m_particles.resize(a_finestLevel + 1);
230 m_maskParticles.resize(a_finestLevel + 1);
231 for (int lvl = 0; lvl <= a_finestLevel; lvl++) {
232 m_particles[lvl] = RefCountedPtr<LevelParticles>(new LevelParticles(a_grids[lvl]));
233 m_maskParticles[lvl] = RefCountedPtr<LevelParticles>(new LevelParticles(a_grids[lvl]));
234 }
235
236 // Buffer particles live on the grown grids (boxes grown by the refinement factor on finer levels).
237 this->setupGrownGrids();
238 m_bufferParticles.resize(a_finestLevel + 1);
239 for (int lvl = 0; lvl <= a_finestLevel; lvl++) {
240 m_bufferParticles[lvl] = RefCountedPtr<LevelParticles>(new LevelParticles(m_grownGrids[lvl]));
241 }
242
243 m_isOrganizedByCell = false;
244 m_isDefined = true;
245 }

476 void
477 addParticlesDestructive(ParticleSoA<P, Traits>& a_particles)
478 {
479 CH_TIME("ParticleContainer::addParticlesDestructive");
480
481 CH_assert(m_isDefined);
482
483 for (int lvl = 0; lvl <= m_finestLevel; lvl++) {
484 DataIterator dit = m_grids[lvl].dataIterator();
485 if (dit.ok()) {
486 (*m_particles[lvl])[dit()].catenate(a_particles); // move the buffer onto a local leaf
487 break;
488 }
489 }
490
491 this->remap(); // route every ingested particle to its owning patch/level/rank

506 getNumberOfValidParticlesLocal() const
507 {
508 CH_TIME("ParticleContainer::getNumberOfValidParticlesLocal");
509
510 CH_assert(m_isDefined);
511
512 unsigned long long n = 0;
513 for (int lvl = 0; lvl <= m_finestLevel; lvl++) {
514 const LevelParticles& level = *m_particles[lvl];
515 const DisjointBoxLayout& dbl = m_grids[lvl];
516 const DataIterator& dit = dbl.dataIterator();
517 const int nbox = dit.size();
518
519#pragma omp parallel for schedule(runtime) reduction(+ : n)
520 for (int mybox = 0; mybox < nbox; mybox++) {
521 const DataIndex& din = dit[mybox];
522
523 n += static_cast<unsigned long long>(level[din].size());
524 }
525 }
526
527 return n;

727 void
728 transferParticles(AMRParticlesSoA<P, Traits>& a_source)
729 {
730 CH_TIME("ParticleContainer::transferParticles");
731
732 CH_assert(m_isDefined);
733
734 for (int lvl = 0; lvl <= m_finestLevel; lvl++) {
735 LevelParticles& src = *a_source[lvl];
736 LevelParticles& dst = *m_particles[lvl];
737
738 const DisjointBoxLayout& dbl = m_grids[lvl];
739 const DataIterator& dit = dbl.dataIterator();
740
741 const int nbox = dit.size();
742
743#pragma omp parallel for schedule(runtime)
744 for (int mybox = 0; mybox < nbox; mybox++) {
745 const DataIndex& din = dit[mybox];
746
747 dst[din].catenate(src[din]); // bulk move (id/rank preserved); empty dst -> O(1) swap
748 }
749 }
750
751 m_isOrganizedByCell = false;
752 }
