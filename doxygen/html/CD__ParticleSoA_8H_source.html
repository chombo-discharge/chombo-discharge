1445 void
1446 growIfFull()
1447 {
1448 if (m_size == m_capacity) {
1449 // Saturating geometric growth: 2 * m_capacity overflows std::size_t once m_capacity exceeds
1450 // SIZE_MAX/2, and reserve()'s "a_capacity <= m_capacity" guard would then silently no-op.
1451 const std::size_t newCap = (m_capacity == 0) ? 16
1452 : (m_capacity <= std::numeric_limits<std::size_t>::max() / 2
1453 ? 2 * m_capacity
1454 : std::numeric_limits<std::size_t>::max());
1455 this->reserve(newCap);
1456 }
