1594 void
1595 growIfFull()
1596 {
1597 if (m_size == m_capacity) {
1598 // Saturating geometric growth: 2 * m_capacity overflows std::size_t once m_capacity exceeds
1599 // SIZE_MAX/2, and reserve()'s "a_capacity <= m_capacity" guard would then silently no-op.
1600 const std::size_t newCap = (m_capacity == 0) ? 16
1601 : (m_capacity <= std::numeric_limits<std::size_t>::max() / 2
1602 ? 2 * m_capacity
1603 : std::numeric_limits<std::size_t>::max());
1604 this->reserve(newCap);
1605 }
