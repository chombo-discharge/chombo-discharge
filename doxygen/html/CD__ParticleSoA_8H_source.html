1536 void
1537 growIfFull()
1538 {
1539 if (m_size == m_capacity) {
1540 // Saturating geometric growth: 2 * m_capacity overflows std::size_t once m_capacity exceeds
1541 // SIZE_MAX/2, and reserve()'s "a_capacity <= m_capacity" guard would then silently no-op.
1542 const std::size_t newCap = (m_capacity == 0) ? 16
1543 : (m_capacity <= std::numeric_limits<std::size_t>::max() / 2
1544 ? 2 * m_capacity
1545 : std::numeric_limits<std::size_t>::max());
1546 this->reserve(newCap);
1547 }
