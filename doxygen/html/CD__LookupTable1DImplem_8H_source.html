237 template <int N>
238 template <int K>
239 inline Real
240 LookupTable1D<N>::getEntry(const Real a_x) const
241 {
242 
243  // TLDR: This is the main routine for fetching a column of data. This will switch between constant/linear interpolation outside of
244  // the stored data range.
245 
246  Real ret = 0.0;
247 
248  if (m_data.size() == 1) {
249  ret = m_data.front()[K];
250  }
251  else {
252 
253  if (!(std::get<Sorted>(m_state))) {
254  MayDay::Warning("LookupTable1D<N>::getEntry - table is not sorted");
255  }
256 
257  if (!(std::get<Uniform>(m_state))) {
258  MayDay::Warning("LookupTable1D<N>::getEntry - table is not uniform");
259  }
260 
261  // We have sorted our data from smallest to largest along some column -- this is that column.
262  const int sortedColumn = std::get<Column>(m_state);
263 
264  // This is the LOWER index among the two indices we will interpolate between.
265  const int idx = this->getIndexLo(a_x);
266 
267  // See how we should enforce fetching the data when retrieving data outside the stored data range.
268  if (idx < 0) { // Input point is outside the lower range
269 
270  // Figure out how we should interpolate on the low side.
271  switch (m_strategyLo) {
272  case OutOfRangeStrategy::Constant: {
273  ret = m_data.front()[K];
274 
275  break;
276  }
277  case OutOfRangeStrategy::Linear: {
278  const int size = m_data.size();
279 
280  const Real idxLo = 0;
281  const Real idxHi = 1;
282 
283  const Real xLo = m_data[idxLo][sortedColumn];
284  const Real xHi = m_data[idxHi][sortedColumn];
285 
286  // Figure out distance between the two points
287  const Real deltaX = xHi - xLo;
288 
289  // Figure out the distance between the low index and our own input point.
290  const Real dist = xLo - a_x;
291 
292  // Compute the slope.
293  const Real slope = (m_data[idxHi][K] - m_data[idxLo][K]) / deltaX;
294 
295  // Do the interpolation.
296  ret = m_data.front()[K] - dist * slope;
297 
298  break;
299  }
300  default:
301  MayDay::Error("LookupTable1D<N>::getEntry -- logic bust when fetching out-of-range endpoint on the low side");
302 
303  break;
304  }
305  }
306  else if (idx >= m_data.size() - 1) { // Input point is outside the range on the high end.
307 
308  // Figure out how to get the data outside the range.
309  switch (m_strategyHi) {
310  case OutOfRangeStrategy::Constant: {
311  ret = m_data.back()[K];
312 
313  break;
314  }
315  case OutOfRangeStrategy::Linear: {
316  const int size = m_data.size();
317 
318  const int idxLo = m_data.size() - 2;
319  const int idxHi = m_data.size() - 1;
320 
321  const Real xLo = m_data[idxLo][sortedColumn];
322  const Real xHi = m_data[idxHi][sortedColumn];
323 
324  // Compute the spacing between the two points
325  const Real deltaX = xHi - xLo;
326 
327  // Compute the slope.
328  const Real slope = (m_data[size - 1][K] - m_data[size - 2][K]) / deltaX;
329 
330  // Compute the distance to the endpoint.
331  const Real dist = a_x - xHi;
332 
333  // Linearly interpolate.
334  ret = m_data.back()[K] + dist * slope;
335 
336  break;
337  }
338  default:
339  MayDay::Error("LookupTable1D<N>::getEntry -- logic bust when fetching out-of-range endpoint on the high side");
340 
341  break;
342  }
343  }
344  else {
345 
346  // Compute spacing between the two points.
347  const Real d = (m_data[idx + 1][sortedColumn] - m_data[idx][sortedColumn]);
348 
349  // Compute the slope between the two points.
350  const Real t = (a_x - m_data[idx][sortedColumn]) / d;
351 
352  // Linearly interpolate.
353  ret = m_data[idx][K] + t * (m_data[idx + 1][K] - m_data[idx][K]);
354  }
355  }
356 
357  return ret;
358 }

360 template <int N>
361 inline std::array<Real, N>
362 LookupTable1D<N>::getData(const Real a_x) const
363 {
364 
365  std::array<Real, N> ret;
366 
367  if (m_data.size() == 1) {
368  ret = m_data.front();
369  }
370  else {
371  if (!(std::get<Sorted>(m_state))) {
372  MayDay::Error("LookupTable1D<N>::getEntry - table is not sorted");
373  }
374 
375  if (!(std::get<Uniform>(m_state))) {
376  MayDay::Error("LookupTable1D<N>::getEntry - table is not uniform");
377  }
378 
379  const int sortedColumn = std::get<Column>(m_state);
380 
381  // This is the LOWER index among the two indices we will interpolate between.
382  const int idx = this->getIndexLo(a_x);
383 
384  // See how we should enforce fetching the data when retrieving data outside the stored data range.
385  if (idx < 0) { // On the low side.
386 
387  // Figure out how we should interpolate on the low side.
388  switch (m_strategyLo) {
389  case OutOfRangeStrategy::Constant: {
390  ret = m_data.front();
391 
392  break;
393  }
394  case OutOfRangeStrategy::Linear: {
395  ret = m_data.front();
396 
397  // Distance from first point in the table to the interpolation point. This
398  // will be a positive number.
399  const Real dist = m_data.front()[sortedColumn] - a_x;
400 
401  // Interpolate each dependent and independent variable.
402  for (int i = 0; i < N; i++) {
403 
404  // This is the slope for variable i.
405  const Real slope = (m_data[1][i] - m_data[0][i]) / m_delta;
406 
407  // Extrapolate beyond endpoint
408  ret[i] -= dist * slope;
409  }
410 
411  break;
412  }
413  default:
414  MayDay::Error("LookupTable1D<N>::getData -- logic bust when fetching out-of-range endpoint on the low side");
415 
416  break;
417  }
418  }
419  else if (idx >= m_data.size() - 1) {
420 
421  // Figure out how to get the data outside the range.
422  switch (m_strategyHi) {
423  case OutOfRangeStrategy::Constant: {
424  ret = m_data.back();
425 
426  break;
427  }
428  case OutOfRangeStrategy::Linear: {
429  const int size = m_data.size();
430 
431  ret = m_data.back();
432 
433  // This is the distance from the interpolation to the endpoint in the table. This will be a positive number.
434  const Real dist = a_x - m_data.back()[sortedColumn];
435 
436  // Interpolate each dependent and independent variable.
437  for (int i = 0; i < N; i++) {
438 
439  // This is the slope for variable i
440  const Real slope = (m_data[size - 1][i] - m_data[size - 2][i]) / m_delta;
441 
442  ret[i] += dist * slope;
443  }
444 
445  break;
446  }
447  default:
448  MayDay::Error("LookupTable1D<N>::getEntry -- logic bust when fetching out-of-range endpoint on the high side");
449 
450  break;
451  }
452  }
453  else {
454  MayDay::Error("LookupTable1D<N>::getEntry (array version -- spacing is wrong");
455  const Real t = (a_x - m_data[idx][sortedColumn]) / m_delta;
456 
457  // Interpolate all variables.
458  for (int i = 0; i < N; i++) {
459  ret[i] = m_data[idx][i] + t * (m_data[idx + 1][i] - m_data[idx][i]);
460  }
461  }
462  }
463 
464  return ret;
465 }

467 template <int N>
468 inline std::array<Real, N>
469 LookupTable1D<N>::directLookup(const Real a_x) const
470 {
471 
472  std::array<Real, N> ret;
473  if (!(std::get<Sorted>(m_state))) {
474  std::cerr << "LookupTable1D<N>::directLookup -- logic bust. Data must be sorted along the input column!\n";
475  }
476  else {
477  const int column = std::get<Column>(m_state);
478 
479  if (a_x <= m_data.front()[column]) {
480  ret = m_data.front();
481  }
482  else if (a_x >= m_data.back()[column]) {
483  ret = m_data.back();
484  }
485  else { // Do linear interpolation. We look through the rows and find two entries that bracket a_x. We then linearly interpolate between those entries.
486  for (int irow = 0; irow < m_data.size() - 1; irow++) {
487  const std::array<Real, N>& row1 = m_data[irow];
488  const std::array<Real, N>& row2 = m_data[irow + 1];
489 
490  const Real x1 = row1[column];
491  const Real x2 = row2[column];
492 
493  if (a_x >= x1 && a_x <= x2) {
494  const Real t = (a_x - x1) / (x2 - x1);
495 
496  // Now interpolate data in each column.
497  for (int j = 0; j < N; j++) {
498  ret[j] = row1[j] + t * (row2[j] - row1[j]);
499  }
500  }
501  }
502  }
503  }
504 
505  return ret;
506 }

508 template <int N>
509 inline void
510 LookupTable1D<N>::makeUniform(const int a_numRows)
511 {
512 
513  if (std::get<Sorted>(m_state)) {
514  const int sortedColumn = std::get<Column>(m_state);
515 
516  if (a_numRows > 1) {
517 
518  // Find minimum and maximum values for the independent variable
519  const Real xMin = m_data.front()[sortedColumn];
520  const Real xMax = m_data.back()[sortedColumn];
521 
522  // Create a new vector of x-values
523  std::vector<Real> xValues(a_numRows, 0.0);
524 
525  switch (m_spacing) {
526  case TableSpacing::Uniform: {
527  // Figure out the spacing
528  m_delta = (xMax - xMin) / (a_numRows - 1);
529 
530  // Create uniformly spaced points
531  for (int i = 0; i < a_numRows; i++) {
532  xValues[i] = xMin + i * m_delta;
533  }
534 
535  break;
536  }
537  case TableSpacing::Exponential: {
538  // Figure out the spacing
539  m_delta = log(xMax / xMin) / a_numRows;
540 
541  // Create exponentially spaced points
542  for (int i = 0; i < a_numRows; i++) {
543  xValues[i] = xMin * exp(i * m_delta);
544  }
545 
546  break;
547  }
548  default: {
549  MayDay::Error("LookupTable1D<N>::makeUniform - logic bust");
550  }
551  }
552 
553  // Now linearly interpolate the data.
554  std::vector<std::array<Real, N>> newData;
555 
556  for (const auto& x : xValues) {
557  newData.emplace_back(this->directLookup(x));
558  }
559 
560  // Now replace the data with the freshly interpolated data.
561  m_data = newData;
562 
563  // Set the state flag for uniform spacing to true.
564  std::get<Uniform>(m_state) = true;
565  }
566  else {
567  std::cerr << "LookupTable1D<N>::makeUniform -- must have at least two rows!\n";
568  }
569  }
570  else {
571  std::cerr << "LookupTable1D<N>::makeUniform -- must sort the column along one of the coordinates first!\n";
572  }
573 }
