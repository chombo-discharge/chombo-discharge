72 writeH5PartScalarDataset(hid_t a_grp,

204template <typename P, typename Traits>
205void
206DischargeIO::writeH5Part(std::string a_filename,
207 const ParticleContainer<P, Traits>& a_particles,
208 RealVect a_shift,
209 Real a_time) noexcept
210{
211#ifdef CH_USE_HDF5
212 CH_TIME("DischargeIO::writeH5Part(SoA)");
213
214 // Figure out the number of particles on each rank
215 const unsigned long long numParticlesLocal = a_particles.getNumberOfValidParticlesLocal();
216 const unsigned long long numParticlesGlobal = a_particles.getNumberOfValidParticlesGlobal();
217
218 std::vector<unsigned long long> particlesPerRank;
219#ifdef CH_MPI
220 particlesPerRank.resize(numProc(), 0ULL);
221
222 std::vector<int> recv(numProc(), 1);
223 std::vector<int> displ(numProc(), 0);
224 for (int i = 0; i < numProc(); i++) {
225 displ[i] = i;
226 }
227 MPI_Allgatherv(&numParticlesLocal,
228 1,
229 MPI_UNSIGNED_LONG_LONG,
230 &particlesPerRank[0],
231 &recv[0],
232 &displ[0],
233 MPI_UNSIGNED_LONG_LONG,
234 Chombo_MPI::comm);
235#else
236 particlesPerRank.resize(1);
237 particlesPerRank[0] = numParticlesGlobal;
238#endif
239
240 // Set up file access and create the file.
241 hid_t fileAccess = 0;
242#ifdef CH_MPI
243 fileAccess = H5Pcreate(H5P_FILE_ACCESS);
244 H5Pset_fapl_mpio(fileAccess, Chombo_MPI::comm, MPI_INFO_NULL);
245#endif
246
247 hid_t fileID = H5Fcreate(a_filename.c_str(), H5F_ACC_TRUNC, H5P_DEFAULT, fileAccess);
248#ifdef CH_MPI
249 H5Pclose(fileAccess);
250#endif
251
252 // Define the top group necessary for the H5Part file format
253 hid_t grp = H5Gcreate2(fileID, "Step#0", H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);
254
255 // Write the time attribute
256 hid_t scal = H5Screate(H5S_SCALAR);
257 hid_t timeAttr = H5Acreate(grp, "time", H5T_NATIVE_DOUBLE, scal, H5P_DEFAULT, H5P_DEFAULT);
258 H5Awrite(timeAttr, H5T_NATIVE_DOUBLE, &a_time);
259 H5Aclose(timeAttr);
260
261 H5Sclose(scal);
262
263 // Define dataspace dimensions.
264 hsize_t dims[1];
265 dims[0] = numParticlesGlobal;
266
267 hid_t fileSpaceID = H5Screate_simple(1, dims, nullptr);
268
269 // Memory space
270 hsize_t memDims[1];
271 memDims[0] = numParticlesLocal;
272 hid_t memSpaceID = H5Screate_simple(1, memDims, nullptr);
273
274 // Set hyperslabs for file and memory
275 hsize_t fileStart[1];
276 hsize_t fileCount[1];
277
278 hsize_t memStart[1];
279 hsize_t memCount[1];
280
281 memStart[0] = 0;
282 fileStart[0] = 0;
283 for (int i = 0; i < procID(); i++) {
284 fileStart[0] += particlesPerRank[i];
285 }
286 fileCount[0] = particlesPerRank[procID()];
287 memCount[0] = particlesPerRank[procID()];
288
289 H5Sselect_hyperslab(fileSpaceID, H5S_SELECT_SET, fileStart, nullptr, fileCount, nullptr);
290 H5Sselect_hyperslab(memSpaceID, H5S_SELECT_SET, memStart, nullptr, memCount, nullptr);
291
292 // Create the ID and positional data sets
293 hid_t datasetID = H5Dcreate2(grp, "id", H5T_NATIVE_LLONG, fileSpaceID, H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);
294 hid_t datasetX = H5Dcreate2(grp, "x", H5T_NATIVE_DOUBLE, fileSpaceID, H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);
295 hid_t datasetY = H5Dcreate2(grp, "y", H5T_NATIVE_DOUBLE, fileSpaceID, H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);
296#if CH_SPACEDIM == 3
297 hid_t datasetZ = H5Dcreate2(grp, "z", H5T_NATIVE_DOUBLE, fileSpaceID, H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);
298#endif
299
300 std::vector<long long> id;
301 std::vector<double> x;
302 std::vector<double> y;
303 std::vector<double> z;
304
305 for (int lvl = 0; lvl <= a_particles.getFinestLevel(); lvl++) {
306 const DisjointBoxLayout& dbl = a_particles.getGrids()[lvl];
307 const DataIterator& dit = dbl.dataIterator();
308
309 const int nbox = dit.size();
310
311 for (int mybox = 0; mybox < nbox; mybox++) {
312 const DataIndex& din = dit[mybox];
313
314 const ParticleSoA<P, Traits>& leaf = a_particles[lvl][din];
315
316 for (std::size_t i = 0; i < leaf.size(); i++) {
317 const RealVect pos = leaf.position(i);
318
319 id.push_back(static_cast<long long>(leaf.particleID(i)));
320 x.push_back(pos[0] - a_shift[0]);
321 y.push_back(pos[1] - a_shift[1]);
322#if CH_SPACEDIM == 3
323 z.push_back(pos[2] - a_shift[2]);
324#endif
325 }
326 }
327 }
328
329 H5Dwrite(datasetID, H5T_NATIVE_LLONG, memSpaceID, fileSpaceID, H5P_DEFAULT, id.data());
330 H5Dwrite(datasetX, H5T_NATIVE_DOUBLE, memSpaceID, fileSpaceID, H5P_DEFAULT, x.data());
331 H5Dwrite(datasetY, H5T_NATIVE_DOUBLE, memSpaceID, fileSpaceID, H5P_DEFAULT, y.data());
332#if CH_SPACEDIM == 3
333 H5Dwrite(datasetZ, H5T_NATIVE_DOUBLE, memSpaceID, fileSpaceID, H5P_DEFAULT, z.data());
334#endif
335
336 H5Dclose(datasetID);
337 H5Dclose(datasetX);
338 H5Dclose(datasetY);
339#if CH_SPACEDIM == 3
340 H5Dclose(datasetZ);
341#endif
342
343 // Weight is a container-owned column and is always written.
344 writeH5PartScalarDataset(grp,
345 fileSpaceID,
346 memSpaceID,
347 a_particles,
348 "weight",
349 [](const ParticleSoA<P, Traits>& a_leaf, std::size_t a_i) -> double {
350 return a_leaf.weight(a_i);
351 });
352
353 // Payload datasets, derived declaratively from the optional Traits::h5PartColumns descriptor tuple.
354 if constexpr (HasH5PartColumns<Traits>::value) {
355 using H5PartTuple = std::remove_cv_t<std::remove_reference_t<decltype(Traits::h5PartColumns)>>;
356 writeH5PartColumns(grp,
357 fileSpaceID,
358 memSpaceID,
359 a_particles,
360 std::make_index_sequence<std::tuple_size<H5PartTuple>::value>{});
361 }
362
363 // Close dataspaces, top group and file
364 H5Sclose(fileSpaceID);
365 H5Sclose(memSpaceID);
366 H5Gclose(grp);
367 H5Fclose(fileID);
368#endif
369}

372template <typename P, typename Traits>
373void
374DischargeIO::writeCheckParticlesToHDF(HDF5Handle& a_handle,
375 const LayoutData<ParticleSoA<P, Traits>>& a_particles,
376 const std::string& a_dataType)
377{
378 CH_TIMERS("writeParticlesToHDF(SoA)");
379
380 const BoxLayout& grids = a_particles.boxLayout();
381
382 std::vector<unsigned long long> locParticlesPerBox(grids.size(), 0);
383 unsigned long long numLocalParticles = 0;
384
385 // serial: accumulates into the running counter numLocalParticles
386 const DataIterator& dit = grids.dataIterator();
387 const int nbox = dit.size();
388 for (int mybox = 0; mybox < nbox; mybox++) {
389 const DataIndex& din = dit[mybox];
390 const size_t numItems = a_particles[din].size();
391 numLocalParticles += (unsigned long long)numItems;
392 locParticlesPerBox[grids.index(din)] = (unsigned long long)numItems;
393 }
394
395 std::vector<unsigned long long> particlesPerBox(grids.size());
396
397#ifdef CH_MPI
398 int result = MPI_Allreduce(&locParticlesPerBox[0],
399 &particlesPerBox[0],
400 locParticlesPerBox.size(),
401 MPI_UNSIGNED_LONG_LONG,
402 MPI_SUM,
403 Chombo_MPI::comm);
404 if (result != MPI_SUCCESS) {
405 MayDay::Error("MPI communication error in ParticleIO (SoA)");
406 }
407#else
408 for (int i = 0; i < grids.size(); i++) {
409 particlesPerBox[i] = (unsigned long long)locParticlesPerBox[i];
410 }
411#endif
412
413 write_hdf_part_header(a_handle, grids, particlesPerBox, a_dataType);
414
415 // The SoA HDF5 record (position + weight + the payload h5 subset) is stored as raw bytes
416 // (H5T_NATIVE_CHAR). Payload columns may be ParticleReal (float), so the per-particle byte size need
417 // NOT be a multiple of sizeof(double); a double-typed dataset would truncate the extent
418 // (numComps = objSize/sizeof(double)) and overflow the hyperslab on write/read. Byte-based storage is
419 // exact for any payload and any SpaceDim.
420 const size_t objSize = ParticleSoA<P, Traits>::h5BytesPerParticle();
421
422 std::vector<unsigned long long> offsets;
423 unsigned long long totNumParticles = 0;
424 for (int i = 0; i < grids.size(); i++) {
425 offsets.push_back(objSize * totNumParticles);
426 totNumParticles += particlesPerBox[i];
427 }
428
429 if (totNumParticles == 0) {
430 return;
431 }
432
433 hsize_t dataSize = totNumParticles * objSize;
434 hid_t dataspace = H5Screate_simple(1, &dataSize, nullptr);
435 hid_t H5T_type = H5T_NATIVE_CHAR;
436 std::string dataname = a_dataType + ":data";
437
438#ifdef H516
439 hid_t dataset = H5Dcreate(a_handle.groupID(), dataname.c_str(), H5T_type, dataspace, H5P_DEFAULT);
440#else
441 hid_t dataset = H5Dcreate2(a_handle.groupID(),
442 dataname.c_str(),
443 H5T_type,
444 dataspace,
445 H5P_DEFAULT,
446 H5P_DEFAULT,
447 H5P_DEFAULT);
448#endif
449
450 if (numLocalParticles > 0) {
451 const size_t chunkSize = objSize * _CHUNK;
452 char* const chunkBase = new char[chunkSize];
453 char* chunk = chunkBase;
454
455 if (chunkBase == nullptr) {
456 MayDay::Error("WritePart(SoA)::Error: new returned NULL pointer ");
457 }
458
459 // serial: writes into the shared HDF5 dataset at running per-box offsets and shares the chunk buffer
460 const DataIterator& dit = grids.dataIterator();
461 const int nbox = dit.size();
462
463 for (int mybox = 0; mybox < nbox; mybox++) {
464 const DataIndex& din = dit[mybox];
465 size_t offset = offsets[grids.index(din)];
466
467 const ParticleSoA<P, Traits>& leaf = a_particles[din];
468 const size_t n = leaf.size();
469
470 for (size_t ip = 0; ip < n; ip++) {
471 leaf.h5LinearizeParticle((void*)chunk, ip);
472 chunk += objSize;
473
474 if ((ip + 1) % _CHUNK == 0) {
475 chunk -= chunkSize;
476 writeDataChunk(offset, dataspace, dataset, H5T_type, chunkSize, chunk);
477 }
478 }
479
480 const size_t nResidual = n % _CHUNK;
481 if (nResidual > 0) {
482 chunk -= (nResidual * objSize);
483 writeDataChunk(offset, dataspace, dataset, H5T_type, nResidual * objSize, chunk);
484 }
485
486 chunk = chunkBase;
487 }
488
489 delete[] chunkBase;
490 }
491
492 H5Sclose(dataspace);
493 H5Dclose(dataset);
494}

498template <typename P, typename Traits>
499void
500DischargeIO::readCheckParticlesFromHDF(HDF5Handle& a_handle,
501 LayoutData<ParticleSoA<P, Traits>>& a_particles,
502 const std::string& a_dataType)
503{
504 CH_TIMERS("readParticlesFromHDF(SoA)");
505
506 std::vector<unsigned long long> particlesPerBox;
507 Vector<Box> grids;
508 read_hdf_part_header(a_handle, grids, particlesPerBox, a_dataType, a_handle.getGroup());
509
510 const BoxLayout& bl = a_particles.boxLayout();
511
512 // Byte-based storage (matches writeCheckParticlesToHDF): the per-particle record may not be a multiple
513 // of sizeof(double) (ParticleReal payload columns), so everything is counted in bytes (H5T_NATIVE_CHAR).
514 const size_t objSize = ParticleSoA<P, Traits>::h5BytesPerParticle();
515
516 std::vector<unsigned long long> offsets;
517 unsigned long long totNumParticles = 0;
518 for (int i = 0; i < grids.size(); i++) {
519 offsets.push_back(objSize * totNumParticles);
520 totNumParticles += particlesPerBox[i];
521 }
522
523 if (totNumParticles == 0) {
524 return;
525 }
526
527 hid_t H5T_type = H5T_NATIVE_CHAR;
528 std::string dataname = a_dataType + ":data";
529
530#ifdef H516
531 hid_t dataset = H5Dopen(a_handle.groupID(), dataname.c_str());
532#else
533 hid_t dataset = H5Dopen2(a_handle.groupID(), dataname.c_str(), H5P_DEFAULT);
534#endif
535
536 hid_t dataspace = H5Dget_space(dataset);
537
538 const size_t chunkSize = objSize * _CHUNK;
539 char* chunk = new char[chunkSize];
540
541 // serial: reads from the shared HDF5 dataset at per-box offsets and shares the chunk buffer
542 const DataIterator& dit = bl.dataIterator();
543 const int nbox = dit.size();
544 for (int mybox = 0; mybox < nbox; mybox++) {
545 const DataIndex& din = dit[mybox];
546 size_t boxData = objSize * particlesPerBox[bl.index(din)]; // bytes
547 size_t offset = offsets[bl.index(din)]; // bytes
548
549 ParticleSoA<P, Traits>& leaf = a_particles[din];
550
551 size_t dataIn = 0;
552 while (dataIn < boxData) {
553 int size = ((boxData - dataIn) >= chunkSize) ? chunkSize : boxData - dataIn; // bytes
554 readDataChunk(offset, dataspace, dataset, H5T_type, size, chunk);
555
556 for (int ip = 0; ip < size / (int)objSize; ip++) {
557 leaf.h5DelinearizeAndAppend(chunk);
558 chunk += objSize;
559 }
560 dataIn += size;
561 chunk -= size;
562 }
563 }
564
565 delete[] chunk;
566 chunk = nullptr;
567
568 H5Sclose(dataspace);
569 H5Dclose(dataset);
570}
