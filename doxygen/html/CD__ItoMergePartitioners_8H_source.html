30  template <class T>
31  NodePartitioner<T> NodePartitionEqualMass = [](std::shared_ptr<Node<T> >& a_node, const int a_dir) {
32  CH_TIME("NodePartitionEqualMass<T>");
33 
34  CH_assert(a_node != nullptr);
35  CH_assert(a_dir >= 0 );
36  CH_assert(a_dir < SpaceDim);
37 
38  // Get a handle to the node data.
39  std::vector<T>& data = a_node->getData();
40  const Real mass = a_node->mass();
41 
42  // Some basic assertions -- the loop should evaluate to an empty loop when DEBUG=TRUE.
43  CH_assert(data.size() > 0 );
44  CH_assert(mass > 1.0);
45 
46  for (const auto& n : data){
47  CH_assert(n.mass() >= 1.0);
48  }
49 
50  // TLDR: This function splits the primitives in the node into two subvolumes. A splitting plane is made along the input coordinate and it is positioned
51  // such that the total mass on the left/right side of the splitting plane differ by at most one computational particle. The guts of the code mainly
52  // consists of tests for how to divide the "median" particle -- there are various cases that we need to cover here (such as the median particle being
53  // a physical particle or a superparticle).
54 
55  // 1. Sort data along the input coordinate. We need this because we need to have something sensible to iterate through
56  // when we place the splitting plane.
57  std::sort(std::begin(data), std::end(data), [a_dir](const T& p1, const T& p2){return p1[a_dir] < p2[a_dir];});
58 
59  // 2. Figure out where to place the splitting plane -- this is not the physical coordinate but index in data (the list of primitives). We are
60  // actually looking for the particle on some median where the masses between the two half-planes can be made approximately the same. We begin
61  // by setting the median on i = 0 in the primitives vector (data). We then go through the primitives and check if adding the median particle
62  // ends up placing MORE mass in the left node than the right node. Once that happens we have the inflection point where we need to split. The median
63  // particle will be assigned later because it may be a superparticle and we can equilibriate the masses even further.
64  //
65  //
66  // In all of the below. massLeft and massRight are the masses on each side of the median particle. As remarked above, the median particle could be split
67  // later.
68  int medianParticleIndex = 0;
69  Real massLeft = 0.0; // Zero mass, as it should be.
70  Real massRight = mass - data[medianParticleIndex].mass(); // So, entire mass minus the median particle mass.
71 
72  for (int i = 1; i < data.size(); i++){
73  const Real& medianParticleMass = data[medianParticleIndex].mass(); // This is the current median particle mass.
74 
75  // If adding the median particle to the mass in the left node still yields a lower mass than the right now, we just shift the median particle index. So, the previous
76  // "median particle" goes to the left node but we then have a new median particle.
77  if(massLeft + medianParticleMass < massRight){
78  massLeft += medianParticleMass;
79 
80  medianParticleIndex = i;
81 
82  massRight = mass - massLeft - data[medianParticleIndex].mass(); // Right mass = Total mass - left mass - median particle mass
83  }
84  else{
85  break;
86  }
87  }
88 
89  // 2. Make the two leaves and push data into them. Currently, we know that all particles to the left of the "median" particle go into the left half-plane
90  // and all particles to the right of the "median" particle go into the right half-plane. We do not yet know how to place the median particle -- it could
91  // go into either the left node or the right node, or it could be split into and placed into either node.
92  std::shared_ptr<Node<T> >& left = a_node->getLeft();
93  std::shared_ptr<Node<T> >& right = a_node->getRight();
94 
95  if(!left ) left = std::make_shared<Node<T> >();
96  if(!right) right = std::make_shared<Node<T> >();
97 
98  std::vector<T>& leftData = left ->getData();
99  std::vector<T>& rightData = right->getData();
100 
101  leftData. assign(data.begin() , data.begin() + medianParticleIndex);
102  rightData.assign(data.begin() + medianParticleIndex + 1, data.end() );
103 
104  // 3. We next need to assign the median particle. If we can split up the median point into several ones, we do it. This would be the case if the median point
105  // is a superparticle; in this case we divide the weights as best we can and then assign them to the left/right halves. If we can't then we just assign the
106  // particle to whichever left/right child node has the lowest mass.
107  const T& medianParticle = data[medianParticleIndex];
108 
109  if(medianParticle.canSplit()){
110  const RealVect& medianPos = medianParticle.pos (); // Physical position
111  const Real& medianMass = medianParticle.mass (); // Always positive
112  const Real& medianEnergy = medianParticle.energy(); // Note: This is average energy so it remain the same for subparticles.
113  const Real massDiff = massRight - massLeft ; // Not necessarily positive.
114 
115  // This hook is for when all mass goes into one part.
116  if(medianMass < std::abs(massDiff)){
117  if(massDiff > 0){ // Right mass is bigger than left so put entire particle in left mass.
118  massLeft += medianMass;
119  leftData.emplace_back(T(medianPos, medianMass, medianEnergy));
120  }
121  else{ // Left mass is bigger than right so put everything into the right node.
122  massRight += medianMass; // Goes into right
123  rightData.emplace_back(T(medianPos, medianMass, medianEnergy));
124  }
125  }
126  else {
127  // Distribute mass as best we can. This might seem weird because we can have massLeft > massRight in which
128  // case massDiff < 0. This still works since we are simply subtracting the difference from the left leaf first and
129  // then divide a larger total mass between the two leaves. Essentially, we are subtracting the part of the median
130  // particle that gives massLeft > massRight from the left particle first, and then we reassign a larger total mass.
131  Real tmpMassLeft = massDiff;
132  Real tmpMassRight = 0.0;
133  Real tmpMassRem = medianMass - massDiff;
134 
135  const long long N = llround(tmpMassRem);
136  if(N > 0LL){
137  const long long NR = N/2;
138  const long long NL = N - NR;
139 
140  tmpMassLeft += (tmpMassRem/N)*NL;
141  tmpMassRight += (tmpMassRem/N)*NR;
142  }
143 
144  if(tmpMassLeft > 0.0){
145  leftData.emplace_back(T(medianPos, tmpMassLeft, medianEnergy));
146  massLeft += tmpMassLeft;
147  }
148 
149  if(tmpMassRight > 0.0){
150  rightData.emplace_back(T(medianPos, tmpMassRight, medianEnergy));
151  massRight += tmpMassRight;
152  }
153  }
154  }
155  else{
156  // Not a computational particle -- put the particle in the child node with the lowest mass.
157  if(massLeft <= massRight){
158  leftData.emplace_back(medianParticle);
159  massLeft += medianParticle.mass();
160  }
161  else{
162  rightData.emplace_back(medianParticle);
163  massRight += medianParticle.mass();
164  }
165  }
166 
167  // Update the tree structure.
168  left ->setMass(massLeft );
169  right->setMass(massRight);
170 
171  left ->setParent(a_node);
172  right->setParent(a_node);
173 
174  data.resize(0);
175 
176  // Some assertions in case this code every breaks.
177  CH_assert((std::abs(mass - (massLeft + massRight)) <= 1.E-12 ));
178  CH_assert(massLeft >= std::numeric_limits<Real>::min());
179  CH_assert(massRight >= std::numeric_limits<Real>::min());
180 
181  // These should be evaluated to empty loopps by the compiler if DEBUG=FALSE.
182  for (const auto& l : leftData){
183  CH_assert(l.mass() >= std::numeric_limits<Real>::min());
184  }
185 
186  for (const auto& r : rightData){
187  CH_assert(r.mass() >= std::numeric_limits<Real>::min());
188  }
189  };
190 }
