391template <typename P>
392inline void
393TracerParticleStepper<P>::setVelocity()
394{
395 CH_TIME("TracerParticleStepper::setVelocity()");
396 if (m_verbosity > 5) {
397 pout() << "TracerParticleStepper::setVelocity()" << endl;
398 }
399
400 std::function<RealVect(const RealVect a_position)> velFunc;
401
402 switch (m_velocityField) {
403 case VelocityField::Diagonal: {
404 velFunc = [](const RealVect& a_position) -> RealVect {
405 return RealVect::Unit;
406 };
407
408 break;
409 }
410 case VelocityField::Rotational: {
411 velFunc = [](const RealVect pos) -> RealVect {
412 const Real r = pos.vectorLength();
413 const Real theta = atan2(pos[1], pos[0]);
414
415 return RealVect(D_DECL(-r * sin(theta), r * cos(theta), 0.));
416 };
417
418 break;
419 }
420 }
421
422 DataOps::setValue(m_velocity,
423 velFunc,
424 m_amr->getProbLo(),
425 m_amr->getDx(),
426 m_amr->getMultiCutVofIterator(m_realm, m_phase));
427
428 m_amr->conservativeAverage(m_velocity, m_realm, m_phase);
429 m_amr->interpGhost(m_velocity, m_realm, m_phase);
430}

432template <typename P>
433inline void
434TracerParticleStepper<P>::initialParticles()
435{
436 CH_TIME("TracerParticleStepper::initialParticles()");
437 if (m_verbosity > 5) {
438 pout() << "TracerParticleStepper::initialParticles()" << endl;
439 }
440
441 // TLDR: This code draws distributed particles. Thanks to our nifty ParticleManagement and Random static classes this is just
442 // a matter of defining a distribution function. We define this as a lambda that draws particles uniformly distributed
443 // inside a box.
444 //
445 // After creating the distribution we draw the particles (MPI rank distribution being taken care of under the hood) and
446 // set their weight to one. Finally, we add these particles to the ParticleContainer and remove the particles that lie inside
447 // the EB.
448
449 // Create a random distribution which draw particles uniformly distributed in the domain.
450 const RealVect probLo = m_amr->getProbLo();
451 const RealVect probHi = m_amr->getProbHi();
452
453 auto uniformDistribution = [probLo, probHi]() -> RealVect {
454 RealVect ret = probLo;
455
456 for (int dir = 0; dir < SpaceDim; dir++) {
457 ret[dir] += (probHi - probLo)[dir] * Random::getUniformReal01();
458 }
459
460 return ret;
461 };
462
463 // Draw particles (unit weight) into a free buffer (the List<P> analogue), then hand them to the
464 // container, which routes each particle to its owning patch/level/rank.
465 ParticleSoA<P> drawnParticles;
466 ParticleManagement::drawRandomParticles(drawnParticles, m_numInitialParticles, uniformDistribution);
467
468 ParticleContainer<P>& solverParticles = m_solver->getParticles();
469 solverParticles.clearParticles();
470 solverParticles.addParticlesDestructive(drawnParticles);
471
472 // Remove particles inside the EB.
473 m_amr->removeCoveredParticlesIF(solverParticles, m_phase, 0.0);
474}

476template <typename P>
477inline void
478TracerParticleStepper<P>::advanceParticlesEuler(const Real a_dt)
479{
480 CH_TIME("TracerParticleStepper::advanceParticlesEuler()");
481 if (m_verbosity > 5) {
482 pout() << "TracerParticleStepper::advanceParticlesEuler()" << endl;
483 }
484
485 // TLDR: The new position is just x^(k+1) = x^k + dt*v^k
486
487 ParticleContainer<P>& amrParticles = m_solver->getParticles();
488
489 for (int lvl = 0; lvl <= m_amr->getFinestLevel(); lvl++) {
490 const DisjointBoxLayout& dbl = m_amr->getGrids(m_realm)[lvl];
491 const DataIterator& dit = dbl.dataIterator();
492
493 const int nbox = dit.size();
494
495#pragma omp parallel for schedule(runtime)
496 for (int mybox = 0; mybox < nbox; mybox++) {
497 const DataIndex& din = dit[mybox];
498
499 ParticleSoA<P>& leaf = amrParticles[lvl][din];
500
501 double* const pos[SpaceDim] = {D_DECL(leaf.positionColumn(0), leaf.positionColumn(1), leaf.positionColumn(2))};
502 const ParticleReal* const vel[SpaceDim] = {
503 D_DECL(leaf.template column<&P::v_x>(), leaf.template column<&P::v_y>(), leaf.template column<&P::v_z>())};
504
505 ParticleLoops::loop(leaf, [&](const std::size_t i) {
506 for (int dir = 0; dir < SpaceDim; dir++) {
507 pos[dir][i] += vel[dir][i] * a_dt; // x^(k+1) = x^k + dt * v^k
508 }
509 });
510 }
511 }
512
513 amrParticles.remap();
514 m_amr->removeCoveredParticlesIF(amrParticles, m_phase, 0.0);
515
516 m_solver->interpolateVelocities();
517}

519template <typename P>
520inline void
521TracerParticleStepper<P>::advanceParticlesRK2(const Real a_dt)
522{
523 CH_TIME("TracerParticleStepper::advanceParticlesRK2()");
524 if (m_verbosity > 5) {
525 pout() << "TracerParticleStepper::advanceParticlesRK2()" << endl;
526 }
527
528 // TLDR: The new positions are x^(k+1) = x^k + 0.5*dt*[ v(x^k) + v(x^*) ] where x^* = x^k + dt*v^k.
529
530 ParticleContainer<P>& amrParticles = m_solver->getParticles();
531
532 // First step. Store old position (xk) and velocity (k1), then do the Euler advance.
533 for (int lvl = 0; lvl <= m_amr->getFinestLevel(); lvl++) {
534 const DisjointBoxLayout& dbl = m_amr->getGrids(m_realm)[lvl];
535 const DataIterator& dit = dbl.dataIterator();
536
537 const int nbox = dit.size();
538
539#pragma omp parallel for schedule(runtime)
540 for (int mybox = 0; mybox < nbox; mybox++) {
541 const DataIndex& din = dit[mybox];
542
543 ParticleSoA<P>& leaf = amrParticles[lvl][din];
544
545 double* const pos[SpaceDim] = {D_DECL(leaf.positionColumn(0), leaf.positionColumn(1), leaf.positionColumn(2))};
546 const ParticleReal* const vel[SpaceDim] = {
547 D_DECL(leaf.template column<&P::v_x>(), leaf.template column<&P::v_y>(), leaf.template column<&P::v_z>())};
548 ParticleReal* const xk[SpaceDim] = {
549 D_DECL(leaf.template column<&P::xk_x>(), leaf.template column<&P::xk_y>(), leaf.template column<&P::xk_z>())};
550 ParticleReal* const k1[SpaceDim] = {
551 D_DECL(leaf.template column<&P::k1_x>(), leaf.template column<&P::k1_y>(), leaf.template column<&P::k1_z>())};
552
553 ParticleLoops::loop(leaf, [&](const std::size_t i) {
554 for (int dir = 0; dir < SpaceDim; dir++) {
555 xk[dir][i] = pos[dir][i]; // store x^k
556 k1[dir][i] = vel[dir][i]; // store v(x^k)
557 pos[dir][i] += vel[dir][i] * a_dt; // x^* = x^k + dt * v(x^k)
558 }
559 });
560 }
561 }
562
563 // Remap and interpolate the velocities again. This puts the velocity v = v(x^*) into the particles
564 amrParticles.remap();
565 m_amr->removeCoveredParticlesIF(amrParticles, m_phase, 0.0);
566 m_solver->interpolateVelocities();
567
568 // Do the second RK2 stage.
569 for (int lvl = 0; lvl <= m_amr->getFinestLevel(); lvl++) {
570 const DisjointBoxLayout& dbl = m_amr->getGrids(m_realm)[lvl];
571 const DataIterator& dit = dbl.dataIterator();
572
573 const int nbox = dit.size();
574
575#pragma omp parallel for schedule(runtime)
576 for (int mybox = 0; mybox < nbox; mybox++) {
577 const DataIndex& din = dit[mybox];
578
579 ParticleSoA<P>& leaf = amrParticles[lvl][din];
580
581 double* const pos[SpaceDim] = {D_DECL(leaf.positionColumn(0), leaf.positionColumn(1), leaf.positionColumn(2))};
582 const ParticleReal* const vel[SpaceDim] = {
583 D_DECL(leaf.template column<&P::v_x>(), leaf.template column<&P::v_y>(), leaf.template column<&P::v_z>())};
584 const ParticleReal* const xk[SpaceDim] = {
585 D_DECL(leaf.template column<&P::xk_x>(), leaf.template column<&P::xk_y>(), leaf.template column<&P::xk_z>())};
586 const ParticleReal* const k1[SpaceDim] = {
587 D_DECL(leaf.template column<&P::k1_x>(), leaf.template column<&P::k1_y>(), leaf.template column<&P::k1_z>())};
588
589 ParticleLoops::loop(leaf, [&](const std::size_t i) {
590 for (int dir = 0; dir < SpaceDim; dir++) {
591 // x^(k+1) = x^k + dt/2 * [ v(x^k) + v(x^*) ]
592 pos[dir][i] = xk[dir][i] + 0.5 * a_dt * (k1[dir][i] + vel[dir][i]);
593 }
594 });
595 }
596 }
597
598 // Remap and interpolate the velocities again
599 amrParticles.remap();
600 m_amr->removeCoveredParticlesIF(amrParticles, m_phase, 0.0);
601 m_solver->interpolateVelocities();
602}

604template <typename P>
605inline void
606TracerParticleStepper<P>::advanceParticlesRK4(const Real a_dt)
607{
608 CH_TIME("TracerParticleStepper::advanceParticlesRK4()");
609 if (m_verbosity > 5) {
610 pout() << "TracerParticleStepper::advanceParticlesRK4()" << endl;
611 }
612
613 // TLDR: Just the standard RK4 method.
614
615 ParticleContainer<P>& amrParticles = m_solver->getParticles();
616
617 const Real dtHalf = a_dt / 2.0;
618 const Real dtSixth = a_dt / 6.0;
619
620 // k1 step.
621 {
622 for (int lvl = 0; lvl <= m_amr->getFinestLevel(); lvl++) {
623 const DisjointBoxLayout& dbl = m_amr->getGrids(m_realm)[lvl];
624 const DataIterator& dit = dbl.dataIterator();
625
626 const int nbox = dit.size();
627
628#pragma omp parallel for schedule(runtime)
629 for (int mybox = 0; mybox < nbox; mybox++) {
630 const DataIndex& din = dit[mybox];
631
632 ParticleSoA<P>& leaf = amrParticles[lvl][din];
633
634 double* const pos[SpaceDim] = {D_DECL(leaf.positionColumn(0), leaf.positionColumn(1), leaf.positionColumn(2))};
635 const ParticleReal* const vel[SpaceDim] = {
636 D_DECL(leaf.template column<&P::v_x>(), leaf.template column<&P::v_y>(), leaf.template column<&P::v_z>())};
637 ParticleReal* const xk[SpaceDim] = {
638 D_DECL(leaf.template column<&P::xk_x>(), leaf.template column<&P::xk_y>(), leaf.template column<&P::xk_z>())};
639 ParticleReal* const k1[SpaceDim] = {
640 D_DECL(leaf.template column<&P::k1_x>(), leaf.template column<&P::k1_y>(), leaf.template column<&P::k1_z>())};
641
642 ParticleLoops::loop(leaf, [&](const std::size_t i) {
643 for (int dir = 0; dir < SpaceDim; dir++) {
644 xk[dir][i] = pos[dir][i]; // store x^k
645 k1[dir][i] = vel[dir][i]; // k1 = v(x^k)
646 pos[dir][i] = xk[dir][i] + dtHalf * vel[dir][i]; // x = x^k + 0.5*dt*k1
647 }
648 });
649 }
650 }
651
652 // Remap and compute v = v(x)
653 amrParticles.remap();
654 m_solver->interpolateVelocities();
655 }
656
657 // k2 step.
658 {
659 for (int lvl = 0; lvl <= m_amr->getFinestLevel(); lvl++) {
660 const DisjointBoxLayout& dbl = m_amr->getGrids(m_realm)[lvl];
661 const DataIterator& dit = dbl.dataIterator();
662
663 const int nbox = dit.size();
664
665#pragma omp parallel for schedule(runtime)
666 for (int mybox = 0; mybox < nbox; mybox++) {
667 const DataIndex& din = dit[mybox];
668
669 ParticleSoA<P>& leaf = amrParticles[lvl][din];
670
671 double* const pos[SpaceDim] = {D_DECL(leaf.positionColumn(0), leaf.positionColumn(1), leaf.positionColumn(2))};

673 D_DECL(leaf.template column<&P::v_x>(), leaf.template column<&P::v_y>(), leaf.template column<&P::v_z>())};

675 D_DECL(leaf.template column<&P::xk_x>(), leaf.template column<&P::xk_y>(), leaf.template column<&P::xk_z>())};
676 ParticleReal* const k2[SpaceDim] = {
677 D_DECL(leaf.template column<&P::k2_x>(), leaf.template column<&P::k2_y>(), leaf.template column<&P::k2_z>())};
678
679 ParticleLoops::loop(leaf, [&](const std::size_t i) {
680 for (int dir = 0; dir < SpaceDim; dir++) {
681 k2[dir][i] = vel[dir][i]; // k2 = v(x^k + 0.5*dt*k1)
682 pos[dir][i] = xk[dir][i] + dtHalf * vel[dir][i]; // x = x^k + 0.5*dt*k2
683 }
684 });
685 }
686 }
687
688 // Remap and compute v = v(x)
689 amrParticles.remap();
690 m_solver->interpolateVelocities();
691 }
692
693 // k3 step.
694 {
695 for (int lvl = 0; lvl <= m_amr->getFinestLevel(); lvl++) {
696 const DisjointBoxLayout& dbl = m_amr->getGrids(m_realm)[lvl];
697 const DataIterator& dit = dbl.dataIterator();
698
699 const int nbox = dit.size();
700
701#pragma omp parallel for schedule(runtime)
702 for (int mybox = 0; mybox < nbox; mybox++) {
703 const DataIndex& din = dit[mybox];
704
705 ParticleSoA<P>& leaf = amrParticles[lvl][din];
706
707 double* const pos[SpaceDim] = {D_DECL(leaf.positionColumn(0), leaf.positionColumn(1), leaf.positionColumn(2))};
708 const ParticleReal* const vel[SpaceDim] = {
709 D_DECL(leaf.template column<&P::v_x>(), leaf.template column<&P::v_y>(), leaf.template column<&P::v_z>())};
710 const ParticleReal* const xk[SpaceDim] = {
711 D_DECL(leaf.template column<&P::xk_x>(), leaf.template column<&P::xk_y>(), leaf.template column<&P::xk_z>())};
712 ParticleReal* const k3[SpaceDim] = {
713 D_DECL(leaf.template column<&P::k3_x>(), leaf.template column<&P::k3_y>(), leaf.template column<&P::k3_z>())};
714
715 ParticleLoops::loop(leaf, [&](const std::size_t i) {
716 for (int dir = 0; dir < SpaceDim; dir++) {
717 k3[dir][i] = vel[dir][i]; // k3 = v(x^k + 0.5*dt*k2)
718 pos[dir][i] = xk[dir][i] + a_dt * vel[dir][i]; // x = x^k + dt*k3
719 }
720 });
721 }
722 }
723
724 // Remap and compute v = v(x)
725 amrParticles.remap();
726 m_solver->interpolateVelocities();
727 }
728
729 // Final step.
730 {
731 for (int lvl = 0; lvl <= m_amr->getFinestLevel(); lvl++) {
732 const DisjointBoxLayout& dbl = m_amr->getGrids(m_realm)[lvl];
733 const DataIterator& dit = dbl.dataIterator();
734
735 const int nbox = dit.size();
736
737#pragma omp parallel for schedule(runtime)
738 for (int mybox = 0; mybox < nbox; mybox++) {
739 const DataIndex& din = dit[mybox];
740
741 ParticleSoA<P>& leaf = amrParticles[lvl][din];
742
743 double* const pos[SpaceDim] = {D_DECL(leaf.positionColumn(0), leaf.positionColumn(1), leaf.positionColumn(2))};
744 const ParticleReal* const vel[SpaceDim] = {
745 D_DECL(leaf.template column<&P::v_x>(), leaf.template column<&P::v_y>(), leaf.template column<&P::v_z>())};
746 const ParticleReal* const xk[SpaceDim] = {
747 D_DECL(leaf.template column<&P::xk_x>(), leaf.template column<&P::xk_y>(), leaf.template column<&P::xk_z>())};
748 const ParticleReal* const k1[SpaceDim] = {
749 D_DECL(leaf.template column<&P::k1_x>(), leaf.template column<&P::k1_y>(), leaf.template column<&P::k1_z>())};
750 const ParticleReal* const k2[SpaceDim] = {
751 D_DECL(leaf.template column<&P::k2_x>(), leaf.template column<&P::k2_y>(), leaf.template column<&P::k2_z>())};
752 const ParticleReal* const k3[SpaceDim] = {
753 D_DECL(leaf.template column<&P::k3_x>(), leaf.template column<&P::k3_y>(), leaf.template column<&P::k3_z>())};
754
755 ParticleLoops::loop(leaf, [&](const std::size_t i) {
756 for (int dir = 0; dir < SpaceDim; dir++) {
757 pos[dir][i] = xk[dir][i] + dtSixth * k1[dir][i] + dtHalf * k2[dir][i] + dtHalf * k3[dir][i] +
758 dtSixth * vel[dir][i];
759 }
760 });
761 }
762 }
763
764 // Remap and compute v = v(x)
765 amrParticles.remap();
766 m_solver->interpolateVelocities();
767 }
768
769 m_amr->removeCoveredParticlesIF(amrParticles, m_phase, 0.0);
