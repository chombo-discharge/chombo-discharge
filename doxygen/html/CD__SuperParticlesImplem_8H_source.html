152  template <class P, Real& (P::*weight)(), const RealVect& (P::*position)() const>
153  typename KDNode<P>::Partitioner PartitionEqualWeight = [](KDNode<P>& a_node) -> void {
154  CH_assert(!(a_node.isInteriorNode()));
155 
156  constexpr Real splitThresh = 2.0 - std::numeric_limits<Real>::min();
157 
158  // Particles and node weight.
159  typename KDNode<P>::ParticleList& particles = a_node.getParticles();
160 
161  const Real W = a_node.weight();
162 
163  // A. Figure out which coordinate direction we should partition and sort
164  // the particles.
165  RealVect loCorner = +std::numeric_limits<Real>::max() * RealVect::Unit;
166  RealVect hiCorner = -std::numeric_limits<Real>::max() * RealVect::Unit;
167 
168  for (auto& p : particles) {
169  const RealVect& pos = (p.*position)();
170  for (int dir = 0; dir < SpaceDim; dir++) {
171  loCorner[dir] = std::min(pos[dir], loCorner[dir]);
172  hiCorner[dir] = std::max(pos[dir], hiCorner[dir]);
173  }
174  }
175 
176  const int splitDir = (hiCorner - loCorner).maxDir(true);
177 
178  auto sortCrit = [splitDir](const P& p1, const P& p2) -> bool {
179  return (p1.*position)()[splitDir] < (p2.*position)()[splitDir];
180  };
181 
182  std::sort(particles.begin(), particles.end(), sortCrit);
183 
184  // B. Determine the "median particle" and start computing the weight in the
185  // two halves.
186  size_t id = 0;
187  Real wl = 0.0;
188  Real wr = W - (particles[id].*weight)();
189 
190  for (size_t i = 1; i < particles.size(); i++) {
191  const Real& w = (particles[id].*weight)();
192 
193  if (wl + w < wr) {
194  id = i;
195  wl += w;
196  wr = W - wl - (particles[id].*weight)();
197  }
198  else {
199  break;
200  }
201  }
202 
203  // C. Copy the two particle halves to each subnode.
204  P p = particles[id];
205 
206  typename KDNode<P>::ParticleList pl;
207  typename KDNode<P>::ParticleList pr;
208 
209  std::move(particles.begin(), particles.begin() + id, std::back_inserter(pl));
210  std::move(particles.begin() + id + 1, particles.end(), std::back_inserter(pr));
211 
212  const Real& pw = (p.*weight)();
213  const Real dw = wr - wl;
214 
215  CH_assert(wl + wr + pw == W);
216 
217  // D. Assign the median particle; split the particle if we can.
218  if (pw >= splitThresh && pw >= std::abs(dw)) {
219  Real dwl = dw;
220  Real dwr = 0.0;
221  Real ddw = pw - dw;
222 
223  const long long N = llround(ddw);
224 
225  if (N > 0LL) {
226 
227  const long long Nr = N / 2;
228  const long long Nl = N - Nr;
229 
230  dwl += (ddw / N) * Nl;
231  dwr += (ddw / N) * Nr;
232  }
233 
234  if (dwl > 0.0) {
235  P il(p);
236 
237  CH_assert(dwl >= 1.0);
238 
239  wl += dwl;
240  (il.*weight)() = dwl;
241  pl.emplace_back(std::move(il));
242  }
243 
244  if (dwr > 0.0) {
245  P ir(p);
246 
247  CH_assert(dwr >= 1.0);
248 
249  wr += dwr;
250  (ir.*weight)() = dwr;
251  pr.emplace_back(std::move(ir));
252  }
253  }
254  else {
255  if (wl <= wr) {
256  wl += pw;
257  pl.emplace_back(std::move(p));
258  }
259  else {
260  wr += pw;
261  pr.emplace_back(std::move(p));
262  }
263  }
264 
265  // E. If this breaks, weight is not conserved or we broke the median particle splitting; the weight difference
266  // between the left/right node should be at most one physical particle.
267  CH_assert(wl + wr == W);
268  CH_assert(std::abs(wl - wr) <= 1.0);
269 
270  // F. Instantiate the child nodes.
271  particles.resize(0);
272 
273  a_node.getLeft() = std::make_shared<KDNode<P>>(pl);
274  a_node.getRight() = std::make_shared<KDNode<P>>(pr);
275 
276  a_node.getLeft()->weight() = wl;
277  a_node.getRight()->weight() = wr;
278 
279  // G. Debug code; make sure particle weights make sense
280 #ifndef NDEBUG
281  Real WL = 0.0;
282  Real WR = 0.0;
283 
284  // Note: Not auto& l/r : pl/pr because the particles were into the child node.
285  for (auto& l : a_node.getLeft()->getParticles()) {
286  CH_assert((l.*weight)() >= 1.0);
287 
288  WL += (l.*weight)();
289  }
290 
291  for (auto& r : a_node.getRight()->getParticles()) {
292  CH_assert((r.*weight)() >= 1.0);
293 
294  WR += (r.*weight)();
295  }
296 
297  CH_assert(WL == wl);
298  CH_assert(WR == wr);
299 #endif
300  };
