193 template <typename I, typename C, typename R, typename F>
194 Real
195 ItoKMCGodunovStepper<I, C, R, F>::advance(const Real a_dt)
196 {
197  CH_TIME("ItoKMCGodunovStepper::advance");
198  if (this->m_verbosity > 5) {
199  pout() << this->m_name + "::advance" << endl;
200  }
201 
202  // MayDay::Warning("ItoKMCGodunovStepper::advance -- missing hard conservation for extended conductivity");
203 
204  auto debugCharge = [this](const std::string a_message) -> void {
205  const Real Qtot = this->computeTotalCharge();
206  if (procID() == 0) {
207  std::cout << a_message << " = " << Qtot << "\n";
208  }
209  };
210 
211  // Special flag for telling the class that we have the necessary things in place for doing a regrid. This is
212  // an if-but-maybe situation where the user chose not to checkpoint the particles we need for regrids, yet tries
213  // to restart a simulation and regrid without all the prerequisites being in place. This flag is set to true
214  // because these requirements are checked during the regrid routine.
215  m_canRegridOnRestart = true;
216 
217  m_timer = Timer("ItoKMCGodunovStepper::advance");
218 
219  // debugCharge("advance");
220 
221  // Previous time step is needed when regridding.
222  m_prevDt = a_dt;
223 
224  // Done only so we can plot the absorbed photons (advanceReactionNetwork absorbs them)
225  m_timer.startEvent("Deposit photons");
226  for (auto solverIt = (this->m_rte)->iterator(); solverIt.ok(); ++solverIt) {
227  solverIt()->depositPhotons(solverIt()->getPhi(), solverIt()->getBulkPhotons(), DepositionType::NGP);
228  }
229  m_timer.stopEvent("Deposit photons");
230 
231  // Compute the number of particles per cell.
232  this->getPhysicalParticlesPerCell(this->m_oldPPC);
233 
234  // ====== BEGIN TRANSPORT STEP ======
235  // Semi-implicitly advance the particles and the field.
236  switch (m_algorithm) {
237  case WhichAlgorithm::EulerMaruyama: {
238  this->advanceEulerMaruyama(a_dt);
239 
240  break;
241  }
242  default: {
243  MayDay::Abort("ItoKMCGodunovStepper::advance - logic bust");
244 
245  break;
246  }
247  }
248 
249  // Do intersection test and remove particles that struck the EB or domain. Transfer them to appropriate containers. Then recompute the number of particles per cell.
250  this->barrier();
251  m_timer.startEvent("EB/Particle intersection");
252  if (m_extendConductivityEB) {
253 
254  // TLDR: This hook does a special intersection routine where instead of transferring the particles that were intersected, they
255  // are put in a separate data container. We want to do this in order to reduce artificial gradients in the particle
256  // densities near the EB. The way we do this is that we flag the particles during the intersection; particles that are flagged
257  // are transferred to a container which is added to the conductivity-related particles. The particles are later removed.
258  auto setFlag = [](ItoParticle& p) -> void {
259  p.tmpReal() = 1.0;
260  };
261  auto nonDeletionModifier = [](ItoParticle& p) -> void {
262  p.tmpReal() = -1.0;
263  };
264  auto removeCrit = [](const ItoParticle& p) -> bool {
265  return p.tmpReal() < 0.0;
266  };
267 
268  // Set flag for identifying which particles were intersected by not removed.
269  for (auto it = this->m_ito->iterator(); it.ok(); ++it) {
270  ParticleOps::setData<ItoParticle>(it()->getParticles(ItoSolver::WhichContainer::Bulk), setFlag);
271  }
272 
273  // Intersect particles, but don't remove them.
274  const bool deleteParticles = false;
275  this->intersectParticles(SpeciesSubset::AllMobileOrDiffusive, deleteParticles, nonDeletionModifier);
276 
277  // Particles that were not removed are copied to a separate data container.
278  for (auto it = this->m_ito->iterator(); it.ok(); ++it) {
279  const RefCountedPtr<ItoSolver>& solver = it();
280  const RefCountedPtr<ItoSpecies>& species = solver->getSpecies();
281 
282  const int idx = it.index();
283  const int Z = species->getChargeNumber();
284 
285  ParticleContainer<PointParticle>& irregParticles = *m_irregularParticles[idx];
286  ParticleContainer<ItoParticle>& ebParticles = solver->getParticles(ItoSolver::WhichContainer::EB);
287  ParticleContainer<ItoParticle>& bulkParticles = it()->getParticles(ItoSolver::WhichContainer::Bulk);
288 
289  irregParticles.clearParticles();
290 
291  if (Z != 0 && solver->isMobile()) {
292  for (int lvl = 0; lvl <= (this->m_amr)->getFinestLevel(); lvl++) {
293  const DisjointBoxLayout& dbl = (this->m_amr)->getGrids((this->m_particleRealm))[lvl];
294 
295  for (DataIterator dit(dbl); dit.ok(); ++dit) {
296  List<PointParticle>& pointParticles = irregParticles[lvl][dit()].listItems();
297  const List<ItoParticle>& patchParticles = bulkParticles[lvl][dit()].listItems();
298 
299  for (ListIterator<ItoParticle> lit(patchParticles); lit.ok(); ++lit) {
300  const ItoParticle& p = lit();
301 
302  if (p.tmpReal() < 0.0) {
303  const RealVect& pos = p.position();
304  const Real& weight = p.weight();
305  const Real& mobility = p.mobility();
306 
307  pointParticles.add(PointParticle(pos, weight * mobility));
308  }
309  }
310  }
311  }
312  }
313  }
314 
315  // Finally, delete the original particles
316  for (auto it = this->m_ito->iterator(); it.ok(); ++it) {
317  ParticleContainer<ItoParticle>& bulkParticles = it()->getParticles(ItoSolver::WhichContainer::Bulk);
318  ParticleOps::removeParticles<ItoParticle>(bulkParticles, removeCrit);
319  }
320  }
321  else {
322  const bool deleteParticles = true;
323 
324  this->intersectParticles(SpeciesSubset::AllMobileOrDiffusive, deleteParticles);
325  }
326 
327  // The intersection tests above may not have caught all particles -- do a cleanup sweep where particles on the wrong side of
328  // the EB are put on the EB.
329  for (auto it = this->m_ito->iterator(); it.ok(); ++it) {
330  const RefCountedPtr<ItoSolver>& solver = it();
331 
332  ParticleContainer<ItoParticle>& ebParticles = solver->getParticles(ItoSolver::WhichContainer::EB);
333  ParticleContainer<ItoParticle>& bulkParticles = solver->getParticles(ItoSolver::WhichContainer::Bulk);
334 
335  this->m_amr->transferIrregularParticles(ebParticles, bulkParticles, this->m_plasmaPhase);
336  }
337 
338  this->getPhysicalParticlesPerCell(this->m_newPPC);
339  m_timer.stopEvent("EB/Particle intersection");
340 
341  // Remove the run-time configurable particle storage. It is no longer needed.
342  // ====== END TRANSPORT STEP ======
343 
344  // Photon transport
345  this->barrier();
346  m_timer.startEvent("Photon transport");
347  this->advancePhotons(a_dt);
348  m_timer.stopEvent("Photon transport");
349 
350  // Resolve injection at the EB.
351  this->barrier();
352  m_timer.startEvent("EB particle injection");
353  this->fillSecondaryEmissionEB(a_dt);
354  this->resolveSecondaryEmissionEB(a_dt);
355  m_timer.stopEvent("EB particle injection");
356 
357  // Compute the gradients of the various species densities - this is used in the KMC kernels.
358  m_timer.startEvent("Gradient calculation");
359  (this->m_ito)->depositParticles();
360  this->computeDensityGradients();
361  m_timer.stopEvent("Gradient calculation");
362 
363  // Sort the particles and photons per cell so we can call reaction algorithms
364  this->barrier();
365  m_timer.startEvent("Sort by cell");
366  (this->m_ito)->organizeParticlesByCell(ItoSolver::WhichContainer::Bulk);
367  this->sortPhotonsByCell(McPhoto::WhichContainer::Bulk);
368  this->sortPhotonsByCell(McPhoto::WhichContainer::Source);
369  m_timer.stopEvent("Sort by cell");
370 
371  // Run the Kinetic Monte Carlo reaction kernels.
372  this->barrier();
373  m_timer.startEvent("Reaction network");
374  this->advanceReactionNetwork(a_dt);
375  m_timer.stopEvent("Reaction network");
376 
377  // Build superparticles.
378  if ((this->m_timeStep + 1) % this->m_mergeInterval == 0 && this->m_mergeInterval > 0) {
379  this->barrier();
380  m_timer.startEvent("Super-particle management");
381  (this->m_ito)->makeSuperparticles(ItoSolver::WhichContainer::Bulk, (this->m_particlesPerCell));
382  m_timer.stopEvent("Super-particle management");
383  }
384 
385  // Sort particles per patch.
386  this->barrier();
387  m_timer.startEvent("Sort by patch");
388  (this->m_ito)->organizeParticlesByPatch(ItoSolver::WhichContainer::Bulk);
389  this->sortPhotonsByPatch(McPhoto::WhichContainer::Bulk);
390  this->sortPhotonsByPatch(McPhoto::WhichContainer::Source);
391  m_timer.stopEvent("Sort by patch");
392 
393  // Remove particles that are inside the EB.
394  this->barrier();
395  m_timer.startEvent("Remove covered");
396  this->removeCoveredParticles(SpeciesSubset::AllMobileOrDiffusive, EBRepresentation::Discrete, this->m_toleranceEB);
397  m_timer.stopEvent("Remove covered");
398 
399  // Clear BC data holders for now.
400  for (auto solverIt = (this->m_ito)->iterator(); solverIt.ok(); ++solverIt) {
401  solverIt()->clear(ItoSolver::WhichContainer::EB);
402  solverIt()->clear(ItoSolver::WhichContainer::Domain);
403  }
404 
405  // Prepare for the next time step
406  this->barrier();
407  m_timer.startEvent("Post-compute v");
408  this->computeDriftVelocities();
409  m_timer.stopEvent("Post-compute v");
410 
411  this->barrier();
412  m_timer.startEvent("Post-compute D");
413  this->computeDiffusionCoefficients();
414  m_timer.stopEvent("Post-compute D");
415 
416  this->barrier();
417  m_timer.startEvent("Compute J");
418  this->computeCurrentDensity(this->m_currentDensity);
419  m_timer.stopEvent("Compute J");
420 
421  if ((this->m_profile)) {
422  m_timer.eventReport(pout(), false);
423  }
424 
425  m_timer.clear();
426 
427  return a_dt;
428 }

483  m_timer.stopEvent("Regrid ItoSolver");

487  m_timer.stopEvent("Regrid CdrSolver");

494  (this->m_rte)->regrid(a_lmin, a_oldFinestLevel, a_newFinestLevel);
495  m_timer.stopEvent("Regrid RTE");
496 
497  m_timer.startEvent("Regrid SurfaceODESolver");
498  this->m_sigmaSolver->regrid(a_lmin, a_oldFinestLevel, a_newFinestLevel);
499  m_timer.stopEvent("Regrid SurfaceODESolver");
500 
501  // Allocate internal memory for ItoKMCGodunovStepper now....
502  m_timer.startEvent("Allocate internals");
503  this->allocateInternals();
504  m_timer.stopEvent("Allocate internals");
505 
506  // We need to remap/regrid the stored data required for the semi-implicit update as well.
507  m_timer.startEvent("Remap algorithm-particles");
508  for (auto solverIt = (this->m_ito)->iterator(); solverIt.ok(); ++solverIt) {
509  const int idx = solverIt.index();
510  (this->m_amr)->remapToNewGrids(*m_rhoDaggerParticles[idx], a_lmin, a_newFinestLevel);
511  (this->m_amr)->remapToNewGrids(*m_conductivityParticles[idx], a_lmin, a_newFinestLevel);
512  (this->m_amr)->remapToNewGrids(*m_irregularParticles[idx], a_lmin, a_newFinestLevel);
513  }
514  m_timer.stopEvent("Remap algorithm-particles");
515 
516  // Also regrid the space charge density contribution from the CDR equations
517  this->m_amr->interpToNewGrids(m_semiImplicitRhoCDR,
518  m_scratchSemiImplicitRhoCDR,
519  this->m_plasmaPhase,
520  a_lmin,
521  a_oldFinestLevel,
522  a_newFinestLevel,
523  EBCoarseToFineInterp::Type::ConservativePWC);
524 
525  this->m_amr->interpToNewGrids(m_semiImplicitConductivityCDR,
526  m_scratchSemiImplicitConductivityCDR,
527  this->m_plasmaPhase,
528  a_lmin,
529  a_oldFinestLevel,
530  a_newFinestLevel,
531  EBCoarseToFineInterp::Type::ConservativePWC);
532 
533  // Set up the field solver with standard coefficients or with
534  // modified coefficients if we are reusing data from the last time step.
535  m_timer.startEvent("Setup field solver");
536  (this->m_fieldSolver)->setupSolver();
537  this->computeConductivities(m_conductivityParticles);
538  this->setupSemiImplicitPoisson(m_prevDt);
539  m_timer.stopEvent("Setup field solver");
540 
541  // Solve the Poisson equation.
542  m_timer.startEvent("Solve Poisson");
543  if (this->m_timeStep == 0) {
544  this->computeSpaceChargeDensity();
545  }
546  else {
547  this->depositPointParticles(m_rhoDaggerParticles, SpeciesSubset::All);
548  this->computeSemiImplicitRho();
549  }

551  if (!converged) {
552  MayDay::Warning("ItoKMCGodunovStepper::regrid - Poisson solve did not converge after regrid!!!");
553  }
554  m_timer.stopEvent("Solve Poisson");
555 
556  // Regrid superparticles.
557  if (this->m_regridSuperparticles) {
558  m_timer.startEvent("Make superparticles");
559  (this->m_ito)->organizeParticlesByCell(ItoSolver::WhichContainer::Bulk);
560  (this->m_ito)->makeSuperparticles(ItoSolver::WhichContainer::Bulk, (this->m_particlesPerCell));
561  (this->m_ito)->organizeParticlesByPatch(ItoSolver::WhichContainer::Bulk);
562  m_timer.stopEvent("Make superparticles");

943  EBAMRCellData& a_conductivityCell,
944  const Vector<RefCountedPtr<ParticleContainer<PointParticle>>>& a_particles) noexcept
945 {
946  CH_TIME("ItoKMCGodunovStepper::computeCellConductivity(EBAMRCellData, PointParticle");
947  if (this->m_verbosity > 5) {
948  pout() << this->m_name + "::computeCellConductivity(EBAMRCellData, PointParticle)" << endl;
949  }
950 
951  DataOps::setValue(a_conductivityCell, 0.0);
952 
953  // Contribution from Ito solvers.
954  for (auto solverIt = (this->m_ito)->iterator(); solverIt.ok(); ++solverIt) {
955  RefCountedPtr<ItoSolver>& solver = solverIt();
956  const RefCountedPtr<ItoSpecies>& species = solver->getSpecies();
957 
958  const int idx = solverIt.index();
959  const int Z = species->getChargeNumber();
960 
961  if (Z != 0 && solver->isMobile()) {
962  // Deposit on the particle realm.
963  DataOps::setValue(this->m_particleScratch1, 0.0);
964  solver->depositParticles<PointParticle, &PointParticle::weight>(this->m_particleScratch1, *a_particles[idx]);

966  // Copy to fluid realm and add to total conductivity.
967  (this->m_amr)->copyData(this->m_fluidScratch1, this->m_particleScratch1);
968  DataOps::incr(a_conductivityCell, this->m_fluidScratch1, 1.0 * std::abs(Z));
969  }
970  }
971 
972  // Contribution from CDR solvers.
973  DataOps::setValue(m_semiImplicitConductivityCDR, 0.0);
974 
975  for (auto solverIt = (this->m_cdr)->iterator(); solverIt.ok(); ++solverIt) {
976  const RefCountedPtr<CdrSolver>& solver = solverIt();
977  const RefCountedPtr<CdrSpecies>& species = solver->getSpecies();
978 
979  const int index = solverIt.index();
980  const int Z = species->getChargeNumber();
981 
982  if (Z != 0 && solver->isMobile()) {
983  const EBAMRCellData& phi = solver->getPhi();
984  const EBAMRCellData& mu = this->m_cdrMobilities[index];
985 
986  DataOps::copy(this->m_fluidScratch1, phi);
987  DataOps::multiply(this->m_fluidScratch1, mu);
988 
989  DataOps::incr(a_conductivityCell, this->m_fluidScratch1, 1.0 * std::abs(Z));
990  }
991  }
992 
993  // Conductivity is mobility * weight * Q
994  DataOps::scale(a_conductivityCell, Units::Qe);
995 
996  // Coarsen, update ghost cells and interpolate to centroids. User can ask for bi/tri-linear filtering.
997  (this->m_amr)->arithmeticAverage(a_conductivityCell, this->m_fluidRealm, (this->m_plasmaPhase));
998  (this->m_amr)->interpGhostPwl(a_conductivityCell, this->m_fluidRealm, (this->m_plasmaPhase));
999 
1000  if (this->m_smoothConductivity) {
1001  const Real alpha = 0.5;
1002  const int stride = 1;
1003 
1004  DataOps::filterSmooth(a_conductivityCell, alpha, stride);
1005 
1006  (this->m_amr)->arithmeticAverage(a_conductivityCell, this->m_fluidRealm, (this->m_plasmaPhase));
1007  (this->m_amr)->interpGhostPwl(a_conductivityCell, this->m_fluidRealm, (this->m_plasmaPhase));
1008  }
1009 
1010  (this->m_amr)->interpToCentroids(a_conductivityCell, this->m_fluidRealm, (this->m_plasmaPhase));
1011 }
