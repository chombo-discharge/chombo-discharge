106 ItoKMCStepper<I, C, R, F>::allocate();

108 // Now allocate for the conductivity particles and rho^dagger particles. This is only done in the 'allocate' routine

326template <typename I, typename C, typename R, typename F>
327Real
328ItoKMCGodunovStepper<I, C, R, F>::advance(const Real a_dt)
329{
330 CH_TIME("ItoKMCGodunovStepper::advance");
331 if (this->m_verbosity > 5) {
332 pout() << this->m_name + "::advance" << endl;
333 }
334
335 auto debugCharge = [this](const std::string a_message) -> void {
336 const Real Qtot = this->computeTotalCharge();
337 if (procID() == 0) {
338 std::cout << a_message << " = " << Qtot << "\n";
339 }
340 };
341
342 // Special flag for telling the class that we have the necessary things in place for doing a regrid. This is
343 // an if-but-maybe situation where the user chose not to checkpoint the particles we need for regrids, yet tries
344 // to restart a simulation and regrid without all the prerequisites being in place. This flag is set to true
345 // because these requirements are checked during the regrid routine.
346 m_canRegridOnRestart = true;
347
348 m_timer = Timer("ItoKMCGodunovStepper::advance");
349
350 // debugCharge("advance");
351
352 // Previous time step is needed when regridding.
353 this->m_prevDt = a_dt;
354
355 // Done only so we can plot the absorbed photons (advanceReactionNetwork absorbs them)
356 m_timer.startEvent("Deposit photons");
357 for (auto solverIt = (this->m_rte)->iterator(); solverIt.ok(); ++solverIt) {
358 RefCountedPtr<McPhoto> solver = solverIt();
359
360 EBAMRCellData& phi = solver->getPhi();
361 ParticleContainer<Photon>& photons = solver->getBulkPhotons();
362
363 solver->depositPhotons(phi, photons, DepositionType::NGP);
364 }
365 m_timer.stopEvent("Deposit photons");
366
367 // ====== BEGIN TRANSPORT STEP ======
368 // Semi-implicitly advance the particles and the field.
369 switch (m_algorithm) {
370 case WhichAlgorithm::EulerMaruyama: {
371 this->advanceEulerMaruyama(a_dt);
372
373 break;
374 }
375 default: {
376 MayDay::Abort("ItoKMCGodunovStepper::advance - logic bust");
377
378 break;
379 }
380 }
381
382 // Do intersection test and remove particles that struck the EB or domain. Transfer them to appropriate containers. Then recompute the number of particles per cell.
383 this->barrier();
384 m_timer.startEvent("EB/Particle intersection");
385 if (m_extendConductivityEB) {
386
387 // TLDR: This hook does a special intersection routine where instead of transferring the particles that were intersected, they
388 // are put in a separate data container. We want to do this in order to reduce artificial gradients in the particle
389 // densities near the EB. The way we do this is that we flag the particles during the intersection; particles that are flagged
390 // are transferred to a container which is added to the conductivity-related particles. The particles are later removed.
391 const std::function<void(ParticleSoA<ItoParticle>&, std::size_t)> setFlag = [](ParticleSoA<ItoParticle>& leaf,
392 const std::size_t i) -> void {
393 leaf.template get<&ItoParticle::scratch>(i) = 1.0;
394 };
395 const std::function<void(ParticleSoA<ItoParticle>&, std::size_t)> nonDeletionModifier =
396 [](ParticleSoA<ItoParticle>& leaf, const std::size_t i) -> void {
397 leaf.template get<&ItoParticle::scratch>(i) = -1.0;
398 };
399
400 // Set flag for identifying which particles were intersected by not removed.
401 for (auto it = this->m_ito->iterator(); it.ok(); ++it) {
402 ParticleOps::setData(it()->getParticles(ItoSolver::WhichContainer::Bulk), setFlag);
403 }
404
405 // Intersect particles, but don't remove them.
406 const bool deleteParticles = false;
407 this->intersectParticles(SpeciesSubset::AllMobileOrDiffusive, deleteParticles, nonDeletionModifier);
408
409 // Particles that were not removed are copied to a separate data container.
410 for (auto it = this->m_ito->iterator(); it.ok(); ++it) {
411 const RefCountedPtr<ItoSolver>& solver = it();
412 const RefCountedPtr<ItoSpecies>& species = solver->getSpecies();
413
414 const int idx = it.index();
415 const int Z = species->getChargeNumber();
416
417 ParticleContainer<NoPayload>& irregParticles = *m_irregularParticles[idx];
418 ParticleContainer<ItoParticle>& ebParticles = solver->getParticles(ItoSolver::WhichContainer::EB);
419 ParticleContainer<ItoParticle>& bulkParticles = it()->getParticles(ItoSolver::WhichContainer::Bulk);
420
421 irregParticles.clearParticles();
422
423 if (Z != 0 && solver->isMobile()) {
424 for (int lvl = 0; lvl <= (this->m_amr)->getFinestLevel(); lvl++) {
425 const DisjointBoxLayout& dbl = (this->m_amr)->getGrids((this->m_particleRealm))[lvl];
426 const DataIterator& dit = dbl.dataIterator();
427
428 const int nbox = dit.size();
429
430#pragma omp parallel for schedule(runtime)
431 for (int mybox = 0; mybox < nbox; mybox++) {
432 const DataIndex& din = dit[mybox];
433
434 ParticleSoA<NoPayload>& pointParticles = irregParticles[lvl][din];
435 const ParticleSoA<ItoParticle>& leaf = bulkParticles[lvl][din];
436
437 for (std::size_t i = 0; i < leaf.size(); i++) {
438 if (leaf.template get<&ItoParticle::scratch>(i) < 0.0) {
439 const RealVect pos = leaf.position(i);
440 const Real weight = leaf.weight(i);
441 const Real mobility = leaf.template get<&ItoParticle::mobility>(i);
442
443 pointParticles.append(pos, weight * mobility);
444 }
445 }
446 }
447 }
448 }
449 }
450
451 // Finally, delete the original particles that were flagged (scratch < 0) via swap-and-pop on the SoA leaves.
452 for (auto it = this->m_ito->iterator(); it.ok(); ++it) {
453 ParticleContainer<ItoParticle>& bulkParticles = it()->getParticles(ItoSolver::WhichContainer::Bulk);
454
455 for (int lvl = 0; lvl <= (this->m_amr)->getFinestLevel(); lvl++) {
456 const DisjointBoxLayout& dbl = (this->m_amr)->getGrids((this->m_particleRealm))[lvl];
457 const DataIterator& dit = dbl.dataIterator();
458
459 const int nbox = dit.size();
460
461#pragma omp parallel for schedule(runtime)
462 for (int mybox = 0; mybox < nbox; mybox++) {
463 const DataIndex& din = dit[mybox];
464
465 ParticleSoA<ItoParticle>& leaf = bulkParticles[lvl][din];
466
467 std::size_t i = 0;
468 while (i < leaf.size()) {
469 if (leaf.template get<&ItoParticle::scratch>(i) < 0.0) {
470 leaf.remove(i);
471 }
472 else {
473 i++;
474 }
475 }
476 }
477 }
478 }
479 }
480 else {
481 const bool deleteParticles = true;
482
483 this->intersectParticles(SpeciesSubset::AllMobileOrDiffusive, deleteParticles);
484 }
485
486 // The intersection tests above may not have caught all particles -- do a cleanup sweep where particles on the wrong side of
487 // the EB are put on the EB.
488 for (auto it = this->m_ito->iterator(); it.ok(); ++it) {
489 const RefCountedPtr<ItoSolver>& solver = it();
490
491 ParticleContainer<ItoParticle>& ebParticles = solver->getParticles(ItoSolver::WhichContainer::EB);
492 ParticleContainer<ItoParticle>& bulkParticles = solver->getParticles(ItoSolver::WhichContainer::Bulk);
493
494 this->m_amr->transferIrregularParticles(ebParticles, bulkParticles, this->m_plasmaPhase);
495 }
496 m_timer.stopEvent("EB/Particle intersection");
497 // ====== END TRANSPORT STEP ======
498
499 // Photon transport
500 this->barrier();
501 m_timer.startEvent("Photon transport");
502 this->advancePhotons(a_dt);
503 m_timer.stopEvent("Photon transport");
504
505 // Compute the gradients of the various species densities - this is used in the KMC kernels.
506 if ((this->m_physics)->needGradients()) {
507 m_timer.startEvent("Gradient calculation");
508 (this->m_ito)->depositParticles();
509 this->computeDensityGradients();
510 m_timer.stopEvent("Gradient calculation");
511 }
512
513 // Resolve secondary emission. We have filled the relevant particles in the transport step -- this can be done
514 // either before or after the reactions.
515 if (m_emitSecondaryParticlesBeforeReactions) {
516 this->barrier();
517 m_timer.startEvent("EB particle injection");
518 this->fillSecondaryEmissionEB(a_dt);
519 this->resolveSecondaryEmissionEB(a_dt);
520 m_timer.stopEvent("EB particle injection");
521 }
522
523 // Sort the particles and photons per cell so we can call reaction algorithms
524 this->barrier();
525 m_timer.startEvent("Sort by cell");
526 (this->m_ito)->organizeParticlesByCell(ItoSolver::WhichContainer::Bulk);
527 this->sortPhotonsByCell(McPhoto::WhichContainer::Bulk);
528 this->sortPhotonsByCell(McPhoto::WhichContainer::Source);
529 m_timer.stopEvent("Sort by cell");
530
531 // Run the Kinetic Monte Carlo reaction kernels.
532 this->barrier();
533 m_timer.startEvent("Reaction network");
534 this->advanceReactionNetwork(a_dt);
535 m_timer.stopEvent("Reaction network");
536
537 // Sort particles per patch.
538 this->barrier();
539 m_timer.startEvent("Sort by patch");
540 (this->m_ito)->organizeParticlesByPatch(ItoSolver::WhichContainer::Bulk);
541 this->sortPhotonsByPatch(McPhoto::WhichContainer::Bulk);
542 this->sortPhotonsByPatch(McPhoto::WhichContainer::Source);
543 m_timer.stopEvent("Sort by patch");
544
545 // Resolve secondary emission. We have filled the relevant particles in the transport step -- this can be done
546 // either before or after the reactions.
547 if (!m_emitSecondaryParticlesBeforeReactions) {
548 this->barrier();
549 m_timer.startEvent("EB particle injection");
550 this->fillSecondaryEmissionEB(a_dt);
551 this->resolveSecondaryEmissionEB(a_dt);
552 m_timer.stopEvent("EB particle injection");
553 }
554
555 // Remove particles that are inside the EB -- this is not a part of the algorithm, just a safety measure to make sure
556 // we don't do things with particles that lie inside the EB.
557 this->barrier();
558 m_timer.startEvent("Remove covered");
559 this->removeCoveredParticles(SpeciesSubset::AllMobileOrDiffusive, EBRepresentation::Discrete, this->m_toleranceEB);
560 m_timer.stopEvent("Remove covered");
561
562 // Clear BC data holders.
563 for (auto solverIt = (this->m_ito)->iterator(); solverIt.ok(); ++solverIt) {
564 solverIt()->clear(ItoSolver::WhichContainer::EB);
565 solverIt()->clear(ItoSolver::WhichContainer::Domain);
566 }
567
568 // Prepare for the next time step
569 this->barrier();
570 m_timer.startEvent("Post-compute v");
571 this->computeDriftVelocities();
572 m_timer.stopEvent("Post-compute v");
573
574 this->barrier();
575 m_timer.startEvent("Post-compute D");
576 this->computeDiffusionCoefficients();
577 m_timer.stopEvent("Post-compute D");
578
579 this->computePhysicsDt();
580
581 if ((this->m_profile)) {
582 m_timer.eventReport(pout(), false);
583 }
584
585 m_timer.clear();
586
587 // Compute the maximum field (in Townsend).
588 this->m_maxReducedField = this->computeMaxReducedElectricField(this->m_plasmaPhase);
589
590 return a_dt;
591}

593template <typename I, typename C, typename R, typename F>
594void
595ItoKMCGodunovStepper<I, C, R, F>::preRegrid(const int a_lmin, const int a_oldFinestLevel) noexcept
596{
597 CH_TIME("ItoKMCGodunovStepper::preRegrid");
598 if (this->m_verbosity > 5) {
599 pout() << "ItoKMCGodunovStepper::preRegrid" << endl;
600 }
601
602 const int numItoSpecies = (this->m_physics)->getNumItoSpecies();
603 const int numCdrSpecies = (this->m_physics)->getNumCdrSpecies();
604 const int numPlasmaSpecies = (this->m_physics)->getNumPlasmaSpecies();
605 const int numPhotonSpecies = (this->m_physics)->getNumPhotonSpecies();
606
607 ItoKMCStepper<I, C, R, F>::preRegrid(a_lmin, a_oldFinestLevel);
608
609 for (auto solverIt = (this->m_ito)->iterator(); solverIt.ok(); ++solverIt) {
610 const int idx = solverIt.index();
611
612 m_conductivityParticles[idx]->preRegrid();
613 m_irregularParticles[idx]->preRegrid();
614 m_rhoDaggerParticles[idx]->preRegrid();
615 }
616
617 this->m_amr->allocate(m_scratchSemiImplicitRhoCDR, this->m_fluidRealm, this->m_plasmaPhase, 1);
618 this->m_amr->allocate(m_scratchSemiImplicitConductivityCDR, this->m_fluidRealm, this->m_plasmaPhase, 1);

632template <typename I, typename C, typename R, typename F>
633void
634ItoKMCGodunovStepper<I, C, R, F>::regrid(const int a_lmin,
635 const int a_oldFinestLevel,
636 const int a_newFinestLevel) noexcept
637{
638 CH_TIME("ItoKMCGodunovStepper::regrid");
639 if (this->m_verbosity > 5) {
640 pout() << "ItoKMCGodunovStepper::regrid" << endl;
641 }
642
643 m_timer = Timer("ItoKMCGodunovStepper::regrid");
644
645 // A special flag for aborting the simulation if the user did NOT put checkpoint particles in the checkpoint
646 // file but still try to regrid on restart.
647 if (!m_canRegridOnRestart) {
648 const std::string baseErr = "ItoKMCGodunovStepper::regrid -- can't regrid because";
649 const std::string err1 = "checkpoint file does not contain particles. Set Driver.initial_regrids=0";
650
651 pout() << baseErr + err1 << endl;
652
653 MayDay::Error((baseErr + err1).c_str());
654 }
655
656 // Regrid solvers
657 m_timer.startEvent("Regrid ItoSolver");
658 (this->m_ito)->regrid(a_lmin, a_oldFinestLevel, a_newFinestLevel);
659 if (this->m_timeStep == 0) {
660 // Necessary because first time step is not semi-implicit, so it will use the densities from the
661 // Ito and CDr solvers. However, ItoSolver does not re-deposit the particles during regrid, so we
662 // enforce it here.
663 (this->m_ito)->depositParticles();
664 }
665 m_timer.stopEvent("Regrid ItoSolver");
666
667 m_timer.startEvent("Regrid CdrSolver");
668 (this->m_cdr)->regrid(a_lmin, a_oldFinestLevel, a_newFinestLevel);
669 m_timer.stopEvent("Regrid CdrSolver");
670
671 m_timer.startEvent("Regrid FieldSolver");
672 (this->m_fieldSolver)->regrid(a_lmin, a_oldFinestLevel, a_newFinestLevel);
673 m_timer.stopEvent("Regrid FieldSolver");
674
675 m_timer.startEvent("Regrid RTE");
676 (this->m_rte)->regrid(a_lmin, a_oldFinestLevel, a_newFinestLevel);
677 m_timer.stopEvent("Regrid RTE");
678
679 m_timer.startEvent("Regrid SurfaceODESolver");
680 this->m_sigmaSolver->regrid(a_lmin, a_oldFinestLevel, a_newFinestLevel);
681 m_timer.stopEvent("Regrid SurfaceODESolver");
682
683 // Allocate internal memory for ItoKMCGodunovStepper now....
684 m_timer.startEvent("Allocate internals");
685 this->allocateInternals();
686 m_timer.stopEvent("Allocate internals");
687
688 // We need to remap/regrid the stored data required for the semi-implicit update as well.
689 m_timer.startEvent("Remap algorithm-particles");
690 for (auto solverIt = (this->m_ito)->iterator(); solverIt.ok(); ++solverIt) {
691 const int idx = solverIt.index();
692 (this->m_amr)->remapToNewGrids(*m_rhoDaggerParticles[idx], a_lmin, a_newFinestLevel);
693 (this->m_amr)->remapToNewGrids(*m_conductivityParticles[idx], a_lmin, a_newFinestLevel);
694 (this->m_amr)->remapToNewGrids(*m_irregularParticles[idx], a_lmin, a_newFinestLevel);
695 }
696 m_timer.stopEvent("Remap algorithm-particles");
697
698 // Also regrid the space charge density contribution from the CDR equations
699 this->m_amr->interpToNewGrids(m_semiImplicitRhoCDR,
700 m_scratchSemiImplicitRhoCDR,
701 this->m_plasmaPhase,
702 a_lmin,
703 a_oldFinestLevel,
704 a_newFinestLevel,
705 EBCoarseToFineInterp::Type::ConservativePWC);
706
707 this->m_amr->interpToNewGrids(m_semiImplicitConductivityCDR,
708 m_scratchSemiImplicitConductivityCDR,
709 this->m_plasmaPhase,
710 a_lmin,
711 a_oldFinestLevel,
712 a_newFinestLevel,
713 EBCoarseToFineInterp::Type::ConservativePWC);
714
715 // Set up the field solver with standard coefficients or with
716 // modified coefficients if we are reusing data from the last time step.
717 m_timer.startEvent("Setup field solver");
718 (this->m_fieldSolver)->setupSolver();
719 this->computeConductivities(m_conductivityParticles);
720 this->setupSemiImplicitPoisson(this->m_prevDt);
721 m_timer.stopEvent("Setup field solver");
722
723 // Solve the Poisson equation.
724 m_timer.startEvent("Solve Poisson");
725 if (this->m_timeStep == 0) {
726 this->computeSpaceChargeDensity();
727 }
728 else {
729 this->depositPointParticles(m_rhoDaggerParticles, SpeciesSubset::All);
730 this->computeSemiImplicitRho();
731 }
732
733 const bool converged = this->solvePoisson();
734
735 if (!converged) {
736 const std::string errMsg = "ItoKMCGodunovStepper::regrid - Poisson solve did not converge after regrid";
737
738 pout() << errMsg << endl;
739
740 if (this->m_abortOnFailure) {
741 MayDay::Error(errMsg.c_str());
742 }
743 }
744 m_timer.stopEvent("Solve Poisson");
745
746 // Regrid superparticles.
747 if (this->m_regridSuperparticles) {
748 m_timer.startEvent("Make superparticles");
749 (this->m_ito)->organizeParticlesByCell(ItoSolver::WhichContainer::Bulk);
750 (this->m_ito)->makeSuperparticles(ItoSolver::WhichContainer::Bulk, (this->m_particlesPerCell));
751 (this->m_ito)->organizeParticlesByPatch(ItoSolver::WhichContainer::Bulk);
752 m_timer.stopEvent("Make superparticles");
753 }
754
755 // Now let Ihe ito solver deposit its actual particles... In the above it deposit m_rhoDaggerParticles.
756 m_timer.startEvent("Deposit particles");
757 (this->m_ito)->depositParticles();
758 m_timer.stopEvent("Deposit particles");
759
760 // Recompute new velocities and diffusion coefficients
761 m_timer.startEvent("Prepare next step");
762 this->computeDiffusionCoefficients();
763 this->computeDriftVelocities();
764 m_timer.stopEvent("Prepare next step");
765
766 m_timer.eventReport(pout(), false);
767
768 // No reason to have these lying around.
769 m_scratchSemiImplicitRhoCDR.clear();
770 m_scratchSemiImplicitConductivityCDR.clear();
771
772 // Fill the neutral density on the mesh
773 this->fillNeutralDensity();
774}

1197template <typename I, typename C, typename R, typename F>
1198void
1199ItoKMCGodunovStepper<I, C, R, F>::computeCellConductivity(
1200 EBAMRCellData& a_conductivityCell,
1201 const Vector<RefCountedPtr<ParticleContainer<NoPayload>>>& a_particles) noexcept
1202{
1203 CH_TIME("ItoKMCGodunovStepper::computeCellConductivity(EBAMRCellData, ParticleContainer)");
1204 if (this->m_verbosity > 5) {
1205 pout() << this->m_name + "::computeCellConductivity(EBAMRCellData, ParticleContainer)" << endl;
1206 }
1207
1208 DataOps::setValue(a_conductivityCell, 0.0);
1209
1210 // Contribution from Ito solvers.
1211 for (auto solverIt = (this->m_ito)->iterator(); solverIt.ok(); ++solverIt) {
1212 RefCountedPtr<ItoSolver>& solver = solverIt();
1213 const RefCountedPtr<ItoSpecies>& species = solver->getSpecies();
1214
1215 const int idx = solverIt.index();
1216 const int Z = species->getChargeNumber();
1217
1218 if (Z != 0 && solver->isMobile()) {
1219 // Deposit on the particle realm.
1220 DataOps::setValue(this->m_particleScratch1, 0.0);
1221 depositPointParticlesLikeSolver(this->m_amr,
1222 this->m_particleRealm,
1223 this->m_plasmaPhase,
1224 solver,
1225 this->m_particleScratch1,
1226 *a_particles[idx]);
1227
1228 // Copy to fluid realm and add to total conductivity.
1229 (this->m_amr)->copyData(this->m_fluidScratch1, this->m_particleScratch1);
1230 DataOps::incr(a_conductivityCell, this->m_fluidScratch1, 1.0 * std::abs(Z));
1231 }
1232 }
1233
1234 // Contribution from CDR solvers.
1235 DataOps::setValue(m_semiImplicitConductivityCDR, 0.0);
1236 for (auto solverIt = (this->m_cdr)->iterator(); solverIt.ok(); ++solverIt) {
1237 const RefCountedPtr<CdrSolver>& solver = solverIt();
1238 const RefCountedPtr<CdrSpecies>& species = solver->getSpecies();
1239
1240 const int index = solverIt.index();
1241 const int Z = species->getChargeNumber();
1242
1243 if (Z != 0 && solver->isMobile()) {
1244 const EBAMRCellData& phi = solver->getPhi();
1245 const EBAMRCellData& mu = this->m_cdrMobilities[index];
1246
1247 DataOps::copy(this->m_fluidScratch1, phi);
1248 DataOps::multiply(this->m_fluidScratch1, mu);
1249
1250 DataOps::incr(a_conductivityCell, this->m_fluidScratch1, 1.0 * std::abs(Z));
1251 }
1252 }
1253
1254 // Conductivity is mobility * weight * Q
1255 DataOps::scale(a_conductivityCell, Units::Qe);
1256
1257 // Coarsen, update ghost cells and interpolate to centroids. User can ask for bi/tri-linear filtering.
1258 (this->m_amr)->arithmeticAverage(a_conductivityCell, this->m_fluidRealm, (this->m_plasmaPhase));
1259 (this->m_amr)->interpGhostPwl(a_conductivityCell, this->m_fluidRealm, (this->m_plasmaPhase));
1260
1261 // User can choose to filter the conductivity. Need to sync after each smoothing.
1262 if (m_condFilterNum > 0 && m_condFilterMaxStride > 0) {
1263 for (int i = 0; i < m_condFilterNum; i++) {
1264 for (int curStride = 1; curStride <= m_condFilterMaxStride; curStride++) {
1265 DataOps::filterSmooth(a_conductivityCell, m_condFilterAlpha, curStride, true);
1266
1267 (this->m_amr)->arithmeticAverage(a_conductivityCell, this->m_fluidRealm, (this->m_plasmaPhase));
1268 (this->m_amr)->interpGhostPwl(a_conductivityCell, this->m_fluidRealm, (this->m_plasmaPhase));
1269 }
1270 }
1271 }
1272
1273 (this->m_amr)->interpToCentroids(a_conductivityCell, this->m_fluidRealm, (this->m_plasmaPhase));
1274}

1276template <typename I, typename C, typename R, typename F>
1277void
1278ItoKMCGodunovStepper<I, C, R, F>::computeFaceConductivity() noexcept
1279{
1280 CH_TIME("ItoKMCGodunovStepper::computeFaceConductivity");
1281 if (this->m_verbosity > 5) {
1282 pout() << this->m_name + "::computeFaceConductivity" << endl;
1283 }
1284
1285 DataOps::setValue((this->m_conductivityFace), 0.0);
1286 DataOps::setValue((this->m_conductivityEB), 0.0);
1287
1288 // Average the cell-centered conductivity to faces. Note that this includes one "ghost face", which we need
1289 // because the multigrid solver will interpolate face-centered conductivities to face centroids.
1290 const Average average = Average::Arithmetic;
1291 const int tanGhost = 1;
1292 const Interval interv(0, 0);
1293
1294 DataOps::averageCellToFace(
1295 (this->m_conductivityFace),
1296 (this->m_conductivityCell),
1297 (this->m_amr)->getDomains(),
1298 tanGhost,
1299 interv,
1300 interv,
1301 average,
1302 (this->m_amr)->getFaceIteratorWithTangentialGhosts(this->m_fluidRealm, this->m_plasmaPhase));
1303
1304 // Set the EB conductivity.
1305 DataOps::incr((this->m_conductivityEB),
1306 (this->m_conductivityCell),
1307 1.0,
1308 (this->m_amr)->getVofIterator(this->m_fluidRealm, this->m_plasmaPhase));

1311template <typename I, typename C, typename R, typename F>
1312void
1313ItoKMCGodunovStepper<I, C, R, F>::computeSemiImplicitRho() noexcept
1314{
1315 CH_TIMERS("ItoKMCGodunovStepper::computeSemiImplicitRho");
1316 CH_TIMER("ItoKMCGodunovStepper::computeSemiImplicitRho::plasma_phase", t1);
1317 CH_TIMER("ItoKMCGodunovStepper::computeSemiImplicitRho::solid_phase", t2);
1318 CH_TIMER("ItoKMCGodunovStepper::computeSemiImplicitRho::filter", t3);
1319 if (this->m_verbosity > 5) {
1320 pout() << this->m_name + "::computeSemiImplicitRho" << endl;
1321 }
1322
1323 // Soft requirement?
1324 CH_assert(this->m_plasmaPhase == phase::gas);
1325
1326 const RefCountedPtr<MultiFluidIndexSpace>& mfis = (this->m_computationalGeometry)->getMfIndexSpace();
1327 const Vector<Dielectric>& dielectrics = (this->m_computationalGeometry)->getDielectrics();
1328
1329 const bool hasDielectrics = (mfis->numPhases() > 1) && (dielectrics.size() > 0);
1330
1331 MFAMRCellData& rho = this->m_fieldSolver->getRho();
1332 EBAMRCellData rhoGas = (this->m_amr)->alias(phase::gas, rho);
1333 EBAMRCellData rhoSolid;
1334
1335 if (hasDielectrics) {
1336 rhoSolid = (this->m_amr)->alias(phase::solid, rho);
1337 }
1338
1339 DataOps::setValue(rho, 0.0);
1340
1341 // Contribution from Ito solvers to the gas-side space charge density.
1342 CH_START(t1);
1343 for (auto solverIt = this->m_ito->iterator(); solverIt.ok(); ++solverIt) {
1344 const RefCountedPtr<ItoSolver>& solver = solverIt();
1345 const RefCountedPtr<ItoSpecies>& species = solver->getSpecies();
1346 const int Z = species->getChargeNumber();
1347
1348 if (Z != 0) {
1349 (this->m_amr)->copyData(this->m_fluidScratch1, solver->getPhi());
1350
1351 DataOps::incr(rhoGas, this->m_fluidScratch1, 1.0 * Z * Units::Qe);
1352 }
1353 }
1354
1355 // Contribution from CDR equations to the gas-side space charge density.
1356 DataOps::incr(rhoGas, m_semiImplicitRhoCDR, 1.0);
1357 CH_STOP(t1);
1358
1359 // Contribution from gas-side particles that diffused into EBs. This might be necessary near dielectric EBs as a comparatively small
1360 // correction in the space charge density. There should be no contribution from the CDR solvers because the diffusion-only divergence
1361 // term was computed using homogeneous Neumann boundary conditions.
1362 if (hasDielectrics) {
1363 CH_START(t2);
1364
1365 EBAMRCellData particleScratch;
1366 EBAMRCellData fluidScratch;
1367
1368 (this->m_amr)->allocate(particleScratch, this->m_particleRealm, phase::solid, 1);
1369 (this->m_amr)->allocate(fluidScratch, this->m_fluidRealm, phase::solid, 1);
1370
1371 for (auto solverIt = (this->m_ito)->iterator(); solverIt.ok(); ++solverIt) {
1372 const RefCountedPtr<ItoSolver>& solver = solverIt();
1373 const RefCountedPtr<ItoSpecies>& species = solver->getSpecies();
1374 const int Z = species->getChargeNumber();
1375
1376 if (Z != 0) {
1377 DataOps::setValue(particleScratch, 0.0);
1378 DataOps::setValue(fluidScratch, 0.0);
1379
1380 (this->m_amr)
1381 ->depositWeight(particleScratch,
1382 this->m_particleRealm,
1383 phase::solid,
1384 *m_rhoDaggerParticles[solverIt.index()],
1385 DepositionType::CIC,
1386 CoarseFineDeposition::Halo,
1387 true);
1388
1389 (this->m_amr)->copyData(fluidScratch, particleScratch);
1390
1391 DataOps::incr(rhoSolid, fluidScratch, 1.0 * Z * Units::Qe);
1392 }
1393 }
1394
1395 CH_STOP(t2);
1396 }
1397
1398 // Sync across levels
1399 this->m_amr->arithmeticAverage(rho, this->m_fluidRealm);
1400 this->m_amr->interpGhostPwl(rho, this->m_fluidRealm);
1401
1402 // User can choose to filter the space charge density. Need to sync after each smoothing.
1403 if (m_rhoFilterNum > 0 && m_rhoFilterMaxStride > 0) {
1404 CH_START(t3);
1405 for (int i = 0; i < m_rhoFilterNum; i++) {
1406 for (int curStride = 1; curStride <= m_rhoFilterMaxStride; curStride++) {
1407
1408 DataOps::filterSmooth(rhoGas, m_rhoFilterAlpha, curStride, true);
1409
1410 this->m_amr->arithmeticAverage(rhoGas, this->m_fluidRealm, this->m_plasmaPhase);
1411 this->m_amr->interpGhost(rhoGas, this->m_fluidRealm, this->m_plasmaPhase);
1412 }
1413 }
1414 CH_STOP(t3);
1415 }
1416
1417 // Put data on centroids.
1418 this->m_amr->interpToCentroids(rhoGas, this->m_fluidRealm, phase::gas);
1419 if (hasDielectrics) {
1420 this->m_amr->interpToCentroids(rhoSolid, this->m_fluidRealm, phase::solid);
1421 }
1422}

1555bool
1556ItoKMCGodunovStepper<I, C, R, F>::solvePoisson() noexcept
1557{
1558 CH_TIME("ItoKMCGodunovStepper::solvePoisson()");
1559 if (this->m_verbosity > 5) {
1560 pout() << this->m_name + "::solvePoisson()" << endl;
1561 }
1562
1563 // Solve the Poisson equation and compute the cell-centered electric field.
1564 MFAMRCellData& phi = this->m_fieldSolver->getPotential();
1565 MFAMRCellData& rho = this->m_fieldSolver->getRho();
1566 EBAMRIVData& sigma = this->m_sigmaSolver->getPhi();
1567
1568 const bool converged = (this->m_fieldSolver)->solve(phi, rho, sigma, false);
1569
1570 (this->m_fieldSolver)->computeElectricField();
1571
1572 // Copy the electric field to appropriate data holders and perform center-to-centroid
1573 // interpolation.
1574 EBAMRCellData E;
1575 (this->m_amr)->allocatePointer(E, this->m_fluidRealm);
1576 (this->m_amr)->alias(E, this->m_plasmaPhase, (this->m_fieldSolver)->getElectricField());
1577
1578 // Fluid realm
1579 (this->m_amr)->copyData(this->m_electricFieldFluid, E);
1580 (this->m_amr)->conservativeAverage(this->m_electricFieldFluid, this->m_fluidRealm, this->m_plasmaPhase);
1581 (this->m_amr)->interpGhostPwl(this->m_electricFieldFluid, this->m_fluidRealm, this->m_plasmaPhase);
1582 (this->m_amr)->interpToCentroids(this->m_electricFieldFluid, this->m_fluidRealm, this->m_plasmaPhase);
1583
1584 // Particle realm
1585 (this->m_amr)->copyData(this->m_electricFieldParticle, E);
1586 (this->m_amr)->conservativeAverage(this->m_electricFieldParticle, this->m_particleRealm, this->m_plasmaPhase);
1587 (this->m_amr)->interpGhostPwl(this->m_electricFieldParticle, this->m_particleRealm, this->m_plasmaPhase);

1593template <typename I, typename C, typename R, typename F>
1594void
1595ItoKMCGodunovStepper<I, C, R, F>::advanceEulerMaruyama(const Real a_dt) noexcept
1596{
1597 CH_TIME("ItoKMCGodunovStepper::advanceEulerMaruyama");
1598 if (this->m_verbosity > 5) {
1599 pout() << this->m_name + "::advanceEulerMaruyama" << endl;
1600 }
1601
1602 // Store X^k positions.
1603 this->setOldPositions();
1604
1605 // Diffuse the particles. This copies onto m_rhoDaggerParticles and stores the hop on the full particles. We need
1606 // to remap the particles species that made a diffusion hop.
1607 this->barrier();
1608 m_timer.startEvent("Diffuse particles");
1609 this->diffuseParticlesEulerMaruyama(m_rhoDaggerParticles, a_dt);
1610 this->remapPointParticles(m_rhoDaggerParticles, SpeciesSubset::ChargedDiffusive);
1611 m_timer.stopEvent("Diffuse particles");
1612
1613 // Perform the diffusive CDR advance.
1614 this->barrier();
1615 m_timer.startEvent("Diffuse CDR");
1616 this->computeDiffusionTermCDR(m_semiImplicitRhoCDR, a_dt);
1617 m_timer.stopEvent("Diffuse CDR");
1618
1619 // Compute the conductivity on the mesh. This deposits q_e * Z * w * mu on the mesh.
1620 this->barrier();
1621 m_timer.startEvent("Compute conductivities");
1622 this->copyConductivityParticles(m_conductivityParticles);
1623 this->computeConductivities(m_conductivityParticles);
1624 m_timer.stopEvent("Compute conductivities");
1625
1626 // Set up the semi-implicit Poisson solver with the computed conductivities.
1627 this->barrier();
1628 m_timer.startEvent("Setup Poisson");
1629 this->setupSemiImplicitPoisson(a_dt);
1630 m_timer.stopEvent("Setup Poisson");
1631
1632 // Compute space charge density arising from the new particle positions X^k + sqrt(2*D*dt)*W. Only need to
1633 // do the diffusive and charged species.
1634 this->barrier();
1635 m_timer.startEvent("Deposit point particles");
1636 this->depositPointParticles(m_rhoDaggerParticles, SpeciesSubset::Charged);
1637 this->computeSemiImplicitRho();
1638 m_timer.stopEvent("Deposit point particles");
1639
1640 // Solve the semi-implicit Poisson equation.
1641 this->barrier();
1642 m_timer.startEvent("Solve Poisson");
1643 const bool converged = this->solvePoisson();
1644 if (!converged) {
1645 const std::string errMsg = "ItoKMCGodunovStepper::advanceEulerMaruyama - Poisson solve did not converge";
1646
1647 pout() << errMsg << endl;
1648
1649 if (this->m_abortOnFailure) {
1650 MayDay::Error(errMsg.c_str());
1651 }
1652 }
1653 m_timer.stopEvent("Solve Poisson");
1654
1655 // Recompute velocities with the new electric field. This interpolates the velocities to the current particle positions, i.e.
1656 // we compute V^(k+1)(X^k) = mu^k * E^(k+1)(X^k)
1657 this->barrier();
1658 m_timer.startEvent("Step-compute v");
1659#if 1 // This is what the algorithm says.
1660 this->setCdrVelocityFunctions();
1661 this->setItoVelocityFunctions();
1662 (this->m_ito)->interpolateVelocities();
1663 this->multiplyCdrVelocitiesByMobilities();
1664#else // Have to use this for LEA - need to debug.
1665 this->computeDriftVelocities();
1666#endif
1667 m_timer.stopEvent("Step-compute v");
1668
1669 // Finalize the Euler-Maruyama update.
1670 this->barrier();
1671 m_timer.startEvent("Euler-Maruyama step");
1672 this->stepEulerMaruyamaParticles(a_dt);
1673 this->remapParticles(SpeciesSubset::AllMobileOrDiffusive);
1674 this->stepEulerMaruyamaCDR(a_dt);
1675 m_timer.stopEvent("Euler-Maruyama step");
1676}

1678template <typename I, typename C, typename R, typename F>
1679void
1680ItoKMCGodunovStepper<I, C, R, F>::diffuseParticlesEulerMaruyama(
1681 Vector<RefCountedPtr<ParticleContainer<NoPayload>>>& a_rhoDaggerParticles,
1682 const Real a_dt) noexcept
1683{
1684 CH_TIME("ItoKMCGodunovStepper::diffuseParticlesEulerMaruyama");
1685 if (this->m_verbosity > 5) {
1686 pout() << this->m_name + "::diffuseParticlesEulerMaruyama" << endl;
1687 }
1688
1689 this->clearPointParticles(a_rhoDaggerParticles, SpeciesSubset::All);
1690
1691 for (auto solverIt = (this->m_ito)->iterator(); solverIt.ok(); ++solverIt) {
1692 RefCountedPtr<ItoSolver>& solver = solverIt();
1693 const RefCountedPtr<ItoSpecies>& species = solver->getSpecies();
1694
1695 const int idx = solverIt.index();
1696
1697 const bool mobile = solver->isMobile();
1698 const bool diffusive = solver->isDiffusive();
1699 const int Z = species->getChargeNumber();
1700
1701 const auto& diffusionFunction = (this->m_physics)->getItoDiffusionFunctions()[idx];
1702
1703 for (int lvl = 0; lvl <= (this->m_amr)->getFinestLevel(); lvl++) {
1704 const DisjointBoxLayout& dbl = (this->m_amr)->getGrids((this->m_particleRealm))[lvl];
1705 const DataIterator& dit = dbl.dataIterator();

1716 ParticleSoA<NoPayload>& pointParticles = (*a_rhoDaggerParticles[idx])[lvl][din];

1719 const Real weight = leaf.weight(i);
1720 const RealVect pos = leaf.position(i);
1721

1723 // columns (diffusion/velocity), so materialize the payload via gather().

1725 const RealVect hop = diffusive ? diffusionFunction(p, a_dt) : RealVect::Zero;
1726
1727 D_DECL(leaf.template get<&ItoParticle::scratch_x>(i) = static_cast<ParticleReal>(hop[0]),
1728 leaf.template get<&ItoParticle::scratch_y>(i) = static_cast<ParticleReal>(hop[1]),
1729 leaf.template get<&ItoParticle::scratch_z>(i) = static_cast<ParticleReal>(hop[2]));
1730
1731 if (Z != 0) {
1732 pointParticles.append(pos + hop, weight);
1733 }
1734 }
1735 }
1736 }

1740template <typename I, typename C, typename R, typename F>
1741void
1742ItoKMCGodunovStepper<I, C, R, F>::computeDiffusionTermCDR(EBAMRCellData& a_semiImplicitRhoCDR, const Real a_dt) noexcept
1743{
1744 CH_TIME("ItoKMCGodunovStepper::diffuseCDREulerMaruyama");
1745 if (this->m_verbosity > 5) {
1746 pout() << this->m_name + "::diffuseCDREulerMaruyama" << endl;
1747 }
1748
1749 DataOps::setValue(a_semiImplicitRhoCDR, 0.0);
1750
1751 for (auto solverIt = this->m_cdr->iterator(); solverIt.ok(); ++solverIt) {
1752 const RefCountedPtr<CdrSolver>& solver = solverIt();
1753 const RefCountedPtr<CdrSpecies>& species = solver->getSpecies();

1755 const int index = solverIt.index();
1756 const int Z = species->getChargeNumber();
1757
1758 const EBAMRCellData& phi = solver->getPhi();

1760 // Compute finite-volume approximation to div(D*grad(phi))
1761 if (solver->isDiffusive()) {

1767 DataOps::incr(a_semiImplicitRhoCDR, phi, 1.0 * Z);
1768 if (solver->isDiffusive()) {
1769 DataOps::incr(a_semiImplicitRhoCDR, m_cdrDivD[index], 1.0 * Z * a_dt);
1770 }
1771 }
1772 }
1773
1774 DataOps::scale(a_semiImplicitRhoCDR, Units::Qe);
1775
1776 this->m_amr->arithmeticAverage(a_semiImplicitRhoCDR, this->m_fluidRealm, this->m_plasmaPhase);
1777 this->m_amr->interpGhostPwl(a_semiImplicitRhoCDR, this->m_fluidRealm, this->m_plasmaPhase);

1780template <typename I, typename C, typename R, typename F>
1781void
1782ItoKMCGodunovStepper<I, C, R, F>::stepEulerMaruyamaParticles(const Real a_dt) noexcept
1783{
1784 CH_TIME("ItoKMCGodunovStepper::stepEulerMaruyamaParticles");
1785 if (this->m_verbosity > 5) {
1786 pout() << this->m_name + "::stepEulerMaruyamaParticles" << endl;
1787 }
1788
1789 for (auto solverIt = (this->m_ito)->iterator(); solverIt.ok(); ++solverIt) {
1790 RefCountedPtr<ItoSolver>& solver = solverIt();
1791
1792 const bool mobile = solver->isMobile();
1793 const bool diffusive = solver->isDiffusive();
1794
1795 const Real f = mobile ? a_dt : 0.0;
1796 const Real g = diffusive ? 1.0 : 0.0;
1797
1798 if (mobile || diffusive) {
1799 for (int lvl = 0; lvl <= (this->m_amr)->getFinestLevel(); lvl++) {
1800 const DisjointBoxLayout& dbl = (this->m_amr)->getGrids((this->m_particleRealm))[lvl];
1801 const DataIterator& dit = dbl.dataIterator();
1802
1803 auto& particles = solver->getParticles(ItoSolver::WhichContainer::Bulk)[lvl];
1804
1805 const int nbox = dit.size();
1806
1807#pragma omp parallel for schedule(runtime)
1808 for (int mybox = 0; mybox < nbox; mybox++) {
1809 const DataIndex& din = dit[mybox];
1810
1811 ParticleSoA<ItoParticle>& leaf = particles[din];
1812

1814 D_DECL(leaf.positionColumn(0), leaf.positionColumn(1), leaf.positionColumn(2))};

1839ItoKMCGodunovStepper<I, C, R, F>::stepEulerMaruyamaCDR(const Real a_dt) noexcept
1840{
1841 CH_TIME("ItoKMCGodunovStepper::stepEulerMaruyamaCDR");
1842 if (this->m_verbosity > 5) {
1843 pout() << this->m_name + "::stepEulerMaruyamaCDR" << endl;
1844 }
1845
1846 for (auto solverIt = (this->m_cdr)->iterator(); solverIt.ok(); ++solverIt) {
1847 RefCountedPtr<CdrSolver>& solver = solverIt();
1848
1849 const int index = solverIt.index();
1850
1851 EBAMRCellData& phi = solver->getPhi();
1852
1853 this->m_amr->conservativeAverage(phi, this->m_fluidRealm, this->m_plasmaPhase);
1854 this->m_amr->interpGhostPwl(phi, this->m_fluidRealm, this->m_plasmaPhase);
1855
1856 // Add in the advective term -- BC comes later.
1857 if (solver->isMobile()) {
1858 DataOps::setValue(solver->getEbFlux(), 0.0);
1859
1860 // Compute the FV approximation to the advective term and do the Euler advance. If the underlying
1861 // CDR solver is a CTU solver, we also add in the transverse terms.
1862 solver->computeDivF(this->m_fluidScratch1, phi, 0.5 * a_dt, false, true, true);
1863
1864 DataOps::incr(phi, this->m_fluidScratch1, -a_dt);
1865 }
1866
1867 // Add in the diffusion term.
1868 if (solver->isDiffusive()) {
1869 DataOps::incr(phi, this->m_cdrDivD[index], a_dt);
1870 }
1871
1872 DataOps::floor(phi, 0.0, this->m_amr->getVofIterator(this->m_fluidRealm, this->m_plasmaPhase));
1873 }
1874
1875 this->coarsenCDRSolvers();
1876}
