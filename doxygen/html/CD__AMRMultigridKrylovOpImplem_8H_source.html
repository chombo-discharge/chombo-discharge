33template <class T>
34void
35AMRMultigridKrylovOp<T>::define(AMRMultiGrid<LevelData<T>>* a_mg,
36 const Vector<RefCountedPtr<LevelData<BaseFab<bool>>>>& a_validCells,
37 const Vector<Real>& a_dx,
38 const int a_lbase,
39 const int a_lmax,
40 const int a_numVCycles)
41{
42 CH_TIME("AMRMultigridKrylovOp::define");
43
44 m_mg = a_mg;
45 m_amrOps = a_mg->getAMROperators();
46 m_lbase = a_lbase;
47 m_lmax = a_lmax;
48 m_numVCycles = std::max(1, a_numVCycles);
49 m_precondInit = false;
50 m_volumeCached = false; // Geometry changed; the cached per-level volumes are stale.
51
52 // AMR inner-product weight dx^SpaceDim per level.
53 m_dxScale.resize(a_lmax + 1, 1.0);
54 for (int lvl = 0; lvl <= a_lmax; lvl++) {
55 m_dxScale[lvl] = D_TERM(a_dx[lvl], *a_dx[lvl], *a_dx[lvl]);
56 }
57
58 // Per-level "valid" mask (true except where covered by a finer level), used so the AMR inner product counts each
59 // cell once. This is exactly AmrMesh's valid-cell mask, so we share it (refcounted) instead of rebuilding it.
60 m_validMask = a_validCells;
61}

86template <class T>
87void
88AMRMultigridKrylovOp<T>::preCond(Vector<LevelData<T>*>& a_cor, const Vector<LevelData<T>*>& a_residual)
89{
90 CH_TIME("AMRMultigridKrylovOp::preCond");
91
92 if (!m_precondInit) {
93 this->create(m_precondTmp, a_residual);
94 this->create(m_precondDcor, a_residual);
95 m_precondInit = true;
96 }
97
98 this->setToZero(a_cor);
99
100 for (int iter = 0; iter < m_numVCycles; iter++) {
101 // The multigrid cycle needs (residual - L(cor)) as its input and mangles that argument, so we feed it a scratch copy.
102 // On the first cycle cor == 0, so the input is just a_residual -- skip the (expensive) residual evaluation. Only
103 // multi-cycle preconditioning (numVCycles > 1) needs the recompute, which keeps the common single-cycle case as
104 // cheap as one cycle (one extra matvec per cycle would otherwise roughly double the work per Krylov iteration).
105 if (iter == 0) {
106 this->assign(m_precondTmp, a_residual);
107 }
108 else {
109 this->residual(m_precondTmp, a_cor, a_residual, true);
110 this->scale(m_precondTmp, -1.0);
111 }
112
113 this->setToZero(m_precondDcor);
114 m_mg->AMRVCycle(m_precondDcor, m_precondTmp, m_lmax, m_lmax, m_lbase);
115
116 this->incr(a_cor, m_precondDcor, 1.0);
117 }
118}

152template <class T>
153Real
154AMRMultigridKrylovOp<T>::dotProduct(const Vector<LevelData<T>*>& a_1, const Vector<LevelData<T>*>& a_2)
155{
156 CH_TIME("AMRMultigridKrylovOp::dotProduct");
157
158 using OpType = typename AMRMultigridKrylovOpTraits<T>::OpType;
159
160 const int nLevels = m_lmax - m_lbase + 1;
161 const bool needVol = !m_volumeCached;
162
163 // Rank-local partials packed into one buffer: per-level numerators, then (first call only) per-level volumes.
164 // A single MPI_Allreduce of this buffer replaces the 2*nLevels separate reductions of the old per-level path.
165 Vector<Real> partials(needVol ? 2 * nLevels : nLevels, 0.0);
166
167 for (int lvl = m_lbase; lvl <= m_lmax; lvl++) {
168 auto* op = static_cast<OpType*>(m_amrOps[lvl]);
169 Real xy = 0.0;
170 Real vol = 0.0;
171 op->dotProductMaskedLocal(xy, vol, *a_1[lvl], *a_2[lvl], *m_validMask[lvl], needVol);
172
173 partials[lvl - m_lbase] = xy;
174 if (needVol) {
175 partials[nLevels + (lvl - m_lbase)] = vol;
176 }
177 }
178
179 ParallelOps::sum(partials);
180
181 if (needVol) {
182 m_volume.resize(m_lmax + 1, 0.0);
183 for (int lvl = m_lbase; lvl <= m_lmax; lvl++) {
184 m_volume[lvl] = partials[nLevels + (lvl - m_lbase)];
185 }
186 m_volumeCached = true;
187 }
188
189 // Combine into a volume-weighted sum of per-level means: sum_lvl mean_lvl(kappa*X*Y) * dx_lvl^d. This is a
190 // self-consistent inner product (all that the outer GMRES/BiCGStab require), but note it is NOT AMRMultiGrid's
191 // internal single-normalization inner product -- so the Krylov solvers' residuals are measured in a different
192 // (though equivalent) norm than the one AMRMultiGrid reports for the inner V-cycles.
193 Real prod = 0.0;
194 for (int lvl = m_lbase; lvl <= m_lmax; lvl++) {
195 if (m_volume[lvl] > 0.0) {
196 prod += (partials[lvl - m_lbase] / m_volume[lvl]) * m_dxScale[lvl];
197 }
198 }
199
200 return prod;
201}

203template <class T>
204void
205AMRMultigridKrylovOp<T>::mDotProduct(const Vector<LevelData<T>*>& a_1,
206 const int a_sz,
207 const Vector<LevelData<T>*> a_2[],
208 Real a_mdots[])
209{
210 CH_TIME("AMRMultigridKrylovOp::mDotProduct");
211
212 using OpType = typename AMRMultigridKrylovOpTraits<T>::OpType;
213
214 const int nLevels = m_lmax - m_lbase + 1;
215 const int nNum = a_sz * nLevels;
216 const bool needVol = !m_volumeCached;
217
218 // All a_sz*nLevels rank-local numerators (plus per-level volumes on the first call) in one buffer -> one reduction.
219 Vector<Real> partials(needVol ? nNum + nLevels : nNum, 0.0);
220
221 for (int lvl = m_lbase; lvl <= m_lmax; lvl++) {
222 auto* op = static_cast<OpType*>(m_amrOps[lvl]);
223 Real vol = 0.0;
224
225 for (int k = 0; k < a_sz; k++) {
226 Real xy = 0.0;
227 Real v = 0.0;
228 const bool needVolHere = needVol && (k == 0); // Volume is geometry-only; compute it once per level.
229
230 op->dotProductMaskedLocal(xy, v, *a_1[lvl], *(a_2[k][lvl]), *m_validMask[lvl], needVolHere);
231
232 partials[k * nLevels + (lvl - m_lbase)] = xy;
233 if (needVolHere) {
234 vol = v;
235 }
236 }
237
238 if (needVol) {
239 partials[nNum + (lvl - m_lbase)] = vol;
240 }
241 }
242
243 ParallelOps::sum(partials);
244
245 if (needVol) {
246 m_volume.resize(m_lmax + 1, 0.0);
247 for (int lvl = m_lbase; lvl <= m_lmax; lvl++) {
248 m_volume[lvl] = partials[nNum + (lvl - m_lbase)];
249 }
250 m_volumeCached = true;
251 }
252
253 for (int k = 0; k < a_sz; k++) {
254 Real prod = 0.0;
255 for (int lvl = m_lbase; lvl <= m_lmax; lvl++) {
256 if (m_volume[lvl] > 0.0) {
257 prod += (partials[k * nLevels + (lvl - m_lbase)] / m_volume[lvl]) * m_dxScale[lvl];
258 }
259 }
260 a_mdots[k] = prod;
261 }
262}

295template <class T>
296Real
297AMRMultigridKrylovOp<T>::norm(const Vector<LevelData<T>*>& a_rhs, int a_ord)
298{
299 if (a_ord == 2) {
300 return std::sqrt(this->dotProduct(a_rhs, a_rhs));
301 }
302
303 Real maxNorm = 0.0;
304 for (int lvl = m_lbase; lvl <= m_lmax; lvl++) {
305 maxNorm = std::max(maxNorm, m_amrOps[lvl]->norm(*a_rhs[lvl], a_ord));
306 }
307
308 return maxNorm;
309}
