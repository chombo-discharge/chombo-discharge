55 inline void
56 Timer::startEvent(const std::string a_event) noexcept
57 {
58  if (m_events.find(a_event) == m_events.end()) { // Create a new event.
59  const TimePoint startTime = Clock::now();
60  const Duration elapsedTime = Duration(0.0);
61 
62  m_events.emplace(a_event, std::make_tuple(false, startTime, elapsedTime));
63  }
64  else { // If the event is not new, we just leave the elapsed time intact.
65  // const TimePoint startTime = Clock::now();
66  std::tuple<bool, TimePoint, Duration>& event = m_events.at(a_event);
67  const bool stoppedEvent = std::get<StoppedEvent>(event);
68 
69  if (stoppedEvent) { // Only start the event if it is not already running.
70  const TimePoint startTime = Clock::now();
71  const Duration& previousElapsedTime = std::get<ElapsedTime>(event);
72 
73  event = std::make_tuple(false, startTime, previousElapsedTime);
74  }
75  }
76 }

78 inline void
79 Timer::stopEvent(const std::string a_event) noexcept
80 {
81  if (m_events.find(a_event) != m_events.end()) {
82  std::tuple<bool, TimePoint, Duration>& event = m_events.at(a_event);
83 
84  const TimePoint startTime = std::get<StartClock>(event);
85  const TimePoint stopTime = Clock::now();
86  const Duration curElapsedTime = stopTime - startTime;
87  const Duration previousElapsedTime = std::get<ElapsedTime>(event);
88  const Duration totalElapsedTime = previousElapsedTime + curElapsedTime;
89 
90  event = std::make_tuple(true, startTime, totalElapsedTime);
91  }
92  else {
93  std::cerr << "Timer::stopEvent -- event '" + a_event + "' has not been started\n";
94  }
95 }

145 inline void
146 Timer::eventReport(std::ostream& a_outputStream, const bool a_localReportOnly) const noexcept
147 {
148  // This routine prints a header, the timing report for the various event, and a tail. All events are included
149  // but unfinished ones are not counted towards the elapsed time.
150 
151  this->printReportHeader(a_outputStream);
152 
153  const std::pair<Real, Real> totalTime = this->computeTotalElapsedTime(a_localReportOnly);
154  const Real totalTimeLocal = totalTime.first;
155 
156  for (const auto& e : m_events) {
157 
158  const std::string& eventName = e.first;
159  const bool finishedEvent = std::get<StoppedEvent>(e.second);
160 
161  std::stringstream ssLocalDuration;
162  std::stringstream ssPercentage;
163  std::stringstream ssMinDuration;
164  std::stringstream ssMaxDuration;
165  std::stringstream ssAvgDuration;
166  std::stringstream ssDevDuration;
167  std::stringstream ssMinRank;
168  std::stringstream ssMaxRank;
169 
170  if (finishedEvent) {
171  const Duration localDuration = std::get<ElapsedTime>(e.second);
172  const Real localDurationInSeconds = localDuration.count();
173 
174  ssLocalDuration << std::fixed << std::setprecision(4) << localDurationInSeconds;
175  ssPercentage << std::fixed << std::setprecision(4) << 100. * localDurationInSeconds / totalTimeLocal;
176 
177 #ifdef CH_MPI
178  if (!a_localReportOnly) {
179 
180  const int srcRank = 0;
181  const int numRank = numProc();
182 
183  Vector<Real> allDurations(numRank);
184  gather(allDurations, localDurationInSeconds, srcRank);
185 
186  // Compute average time and standard deviation
187  Real average;
188  Real sigma;
189 
190  Real maxTime = std::numeric_limits<Real>::min();
191  Real minTime = std::numeric_limits<Real>::max();
192 
193  int minRank;
194  int maxRank;
195 
196  if (procID() == 0) {
197 
198  average = 0.0;
199  for (int i = 0; i < allDurations.size(); i++) {
200  average += allDurations[i];
201 
202  if (allDurations[i] < minTime) {
203  minRank = i;
204  minTime = allDurations[i];
205  }
206  if (allDurations[i] > maxTime) {
207  maxRank = i;
208  maxTime = allDurations[i];
209  }
210  }
211  average *= 1. / numRank;
212 
213  sigma = 0.0;
214  for (int i = 0; i < allDurations.size(); i++) {
215  sigma += sqrt(std::pow(allDurations[i] - average, 2));
216  }
217  sigma = sqrt(sigma / numRank);
218  }
219 
220  broadcast(average, srcRank);
221  broadcast(sigma, srcRank);
222  broadcast(maxTime, srcRank);
223  broadcast(minTime, srcRank);
224  broadcast(minRank, srcRank);
225  broadcast(maxRank, srcRank);
226 
227  ssMinDuration << std::fixed << std::setprecision(4) << minTime;
228  ssMaxDuration << std::fixed << std::setprecision(4) << maxTime;
229  ssAvgDuration << std::fixed << std::setprecision(4) << average;
230  ssDevDuration << std::fixed << std::setprecision(4) << sigma;
231  ssMinRank << std::fixed << std::setprecision(4) << minRank;
232  ssMaxRank << std::fixed << std::setprecision(4) << maxRank;
233  }
234  else {
235  ssMinDuration << " N/A ";
236  ssMaxDuration << " N/A ";
237  ssAvgDuration << " N/A ";
238  ssDevDuration << " N/A ";
239  ssMinRank << " N/A ";
240  ssMaxRank << " N/A ";
241  }
242 #endif
243  }
244  else {
245  ssLocalDuration << " - ";
246  ssPercentage << " - ";
247  ssMinDuration << " - ";
248  ssMaxDuration << " - ";
249  ssAvgDuration << " - ";
250  ssDevDuration << " - ";
251  ssMinRank << " - ";
252  ssMaxRank << " - ";
253  }
254 
255  std::stringstream outputString;
256 
257  // Now print to pout
258  outputString << "| " << std::left << std::setw(25) << eventName << "| " << std::right << std::setw(8)
259  << ssLocalDuration.str() << "| " << std::right << std::setw(8) << ssPercentage.str()
260 #ifdef CH_MPI
261  << "| " << std::right << std::setw(8) << ssMinDuration.str() << "| " << std::right << std::setw(8)
262  << ssMaxDuration.str() << "| " << std::right << std::setw(8) << ssAvgDuration.str() << "| "
263  << std::right << std::setw(8) << ssDevDuration.str() << "| " << std::right << std::setw(8)
264  << ssMinRank.str() << "| " << std::right << std::setw(8) << ssMaxRank.str()
265 #endif
266  << "| "
267  << "\n";
268 
269  a_outputStream << outputString.str();
270  }
271 
272  this->printReportTail(a_outputStream, totalTime);
273 }

275 inline std::pair<Real, Real>
276 Timer::computeTotalElapsedTime(const bool a_localReportOnly) const noexcept
277 {
278 
279  Real elapsedTimeLocal = 0.0;
280  Real elapsedTimeGlobal = 0.0;
281 
282  for (const auto& e : m_events) {
283  const bool finishedEvent = std::get<StoppedEvent>(e.second);
284 
285  if (finishedEvent) {
286  const Duration elapsedTime = std::get<ElapsedTime>(e.second);
287  const Real localDurationInSeconds = elapsedTime.count();
288 
289  elapsedTimeLocal += localDurationInSeconds;
290  }
291  }
292 
293 #ifdef CH_MPI
294  if (!a_localReportOnly) {
295  MPI_Allreduce(&elapsedTimeLocal, &elapsedTimeGlobal, 1, MPI_CH_REAL, MPI_MAX, Chombo_MPI::comm);
296  }
297  else {
298  elapsedTimeGlobal = elapsedTimeLocal;
299  }
300 #else
301  elapsedTimeGlobal = elapsedTimeLocal;
302 #endif
303 
304  return std::make_pair(elapsedTimeLocal, elapsedTimeGlobal);
305 }

307 inline void
308 Timer::writeReportToFile(const std::string a_fileName) const noexcept
309 {
310 
311  // First, go through all events and gather the times on the master rank.
312  std::vector<std::string> headerRow;
313  headerRow.emplace_back("# MPI rank");
314 
315  // These are all the times for the finished events. We organize this is a 2D vector which is M x numRank long. Here,
316  // finishedEvents[0] contains the elapsed times for one of the events. The length of finishedEvents[0] is exactly
317  // numRank long
318  std::vector<std::vector<Real>> finishedEvents;
319 
320 #if CH_MPI
321  constexpr int srcRank = 0;
322  const int numRanks = numProc();
323 #else
324  const int numRanks = 1;
325 #endif
326 
327  for (const auto& e : m_events) {
328  const std::string& eventName = e.first;
329  const std::tuple<bool, TimePoint, Duration>& eventData = e.second;
330 
331  const bool finishedEvent = std::get<StoppedEvent>(eventData);
332 
333  // If we had an event that finished, we gather all the data on the master rank.
334  if (finishedEvent) {
335 
336  const Real localEventDuration = std::get<ElapsedTime>(eventData).count();
337  Vector<Real> allTimes(numRanks, 0.0);
338 #ifdef CH_MPI
339 
340  gather(allTimes, localEventDuration, srcRank);
341 #else
342  allTimes[0] = localEventDuration;
343 #endif
344 
345  finishedEvents.emplace_back(allTimes.stdVector());
346  headerRow.emplace_back(eventName);
347  }
348  }
349 
350  // Open a file and start writing.
351 #if CH_MPI
352  if (procID() == srcRank) {
353 #endif
354  std::ofstream f;
355  f.open(a_fileName, std::ios_base::trunc);
356 
357  const int width = 12;
358  for (int col = 0; col < headerRow.size(); col++) {
359  f << std::left << std::setw(width) << headerRow[col] << "\t";
360  }
361  f << "\n";
362 
363  // Write the data for rank = 0 etc
364  for (int irank = 0; irank < numRanks; irank++) {
365  f << std::left << std::setw(width) << irank << "\t";
366  for (int event = 0; event < finishedEvents.size(); event++) {
367  f << std::left << std::setw(width) << finishedEvents[event][irank] << "\t";
368  }
369  f << "\n";
370  }
371 
372  f.close();
373 #if CH_MPI
374  }
375 #endif
376 }
