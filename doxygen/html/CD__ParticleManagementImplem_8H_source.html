154  template <class P, Real& (P::*weight)(), const RealVect& (P::*position)() const>
155  typename KDNode<P>::Partitioner PartitionEqualWeight = [](KDNode<P>& a_node) -> void {
156  CH_assert(!(a_node.isInteriorNode()));
157 
158  constexpr Real splitThresh = 2.0 - std::numeric_limits<Real>::min();
159 
160  // Particles and node weight.
161  typename KDNode<P>::ParticleList& particles = a_node.getParticles();
162 
163  const Real W = a_node.weight();
164 
165  // A. Figure out which coordinate direction we should partition and sort
166  // the particles.
167  RealVect loCorner = +std::numeric_limits<Real>::max() * RealVect::Unit;
168  RealVect hiCorner = -std::numeric_limits<Real>::max() * RealVect::Unit;
169 
170  for (auto& p : particles) {
171  const RealVect& pos = (p.*position)();
172  for (int dir = 0; dir < SpaceDim; dir++) {
173  loCorner[dir] = std::min(pos[dir], loCorner[dir]);
174  hiCorner[dir] = std::max(pos[dir], hiCorner[dir]);
175  }
176  }
177 
178  const int splitDir = (hiCorner - loCorner).maxDir(true);
179 
180  auto sortCrit = [splitDir](const P& p1, const P& p2) -> bool {
181  return (p1.*position)()[splitDir] < (p2.*position)()[splitDir];
182  };
183 
184  std::sort(particles.begin(), particles.end(), sortCrit);
185 
186  // B. Determine the "median particle" and start computing the weight in the
187  // two halves.
188  size_t id = 0;
189  Real wl = 0.0;
190  Real wr = W - (particles[id].*weight)();
191 
192  for (size_t i = 1; i < particles.size(); i++) {
193  const Real& w = (particles[id].*weight)();
194 
195  if (wl + w < wr) {
196  id = i;
197  wl += w;
198  wr = W - wl - (particles[id].*weight)();
199  }
200  else {
201  break;
202  }
203  }
204 
205  // C. Copy the two particle halves to each subnode.
206  P p = particles[id];
207 
208  typename KDNode<P>::ParticleList pl;
209  typename KDNode<P>::ParticleList pr;
210 
211  std::move(particles.begin(), particles.begin() + id, std::back_inserter(pl));
212  std::move(particles.begin() + id + 1, particles.end(), std::back_inserter(pr));
213 
214  const Real& pw = (p.*weight)();
215  const Real dw = wr - wl;
216 
217  CH_assert(wl + wr + pw == W);
218 
219  // D. Assign the median particle; split the particle if we can.
220  if (pw >= splitThresh && pw >= std::abs(dw)) {
221  Real dwl = dw;
222  Real dwr = 0.0;
223  Real ddw = pw - dw;
224 
225  const long long N = (long long)ddw;
226 
227  if (N > 0LL) {
228 
229  const long long Nr = N / 2;
230  const long long Nl = N - Nr;
231 
232  dwl += (ddw / N) * Nl;
233  dwr += (ddw / N) * Nr;
234  }
235 
236  if (dwl > 0.0) {
237  P il(p);
238 
239  CH_assert(dwl >= 1.0);
240 
241  wl += dwl;
242  (il.*weight)() = dwl;
243  pl.emplace_back(std::move(il));
244  }
245 
246  if (dwr > 0.0) {
247  P ir(p);
248 
249  CH_assert(dwr >= 1.0);
250 
251  wr += dwr;
252  (ir.*weight)() = dwr;
253  pr.emplace_back(std::move(ir));
254  }
255  }
256  else {
257  if (wl <= wr) {
258  wl += pw;
259  pl.emplace_back(std::move(p));
260  }
261  else {
262  wr += pw;
263  pr.emplace_back(std::move(p));
264  }
265  }
266 
267  // E. If this breaks, weight is not conserved or we broke the median particle splitting; the weight difference
268  // between the left/right node should be at most one physical particle.
269  CH_assert(wl + wr == W);
270  CH_assert(std::abs(wl - wr) <= 1.0);
271 
272  // F. Instantiate the child nodes.
273  particles.resize(0);
274 
275  a_node.getLeft() = std::make_shared<KDNode<P>>(pl);
276  a_node.getRight() = std::make_shared<KDNode<P>>(pr);
277 
278  a_node.getLeft()->weight() = wl;
279  a_node.getRight()->weight() = wr;
280 
281  // G. Debug code; make sure particle weights make sense
282 #ifndef NDEBUG
283  Real WL = 0.0;
284  Real WR = 0.0;
285 
286  // Note: Not auto& l/r : pl/pr because the particles were into the child node.
287  for (auto& l : a_node.getLeft()->getParticles()) {
288  CH_assert((l.*weight)() >= 1.0);
289 
290  WL += (l.*weight)();
291  }
292 
293  for (auto& r : a_node.getRight()->getParticles()) {
294  CH_assert((r.*weight)() >= 1.0);
295 
296  WR += (r.*weight)();
297  }
298 
299  CH_assert(WL == wl);
300  CH_assert(WR == wr);
301 #endif
302  };

321  template <typename P, typename T, typename>
322  inline void
323  removePhysicalParticles(List<P>& a_particles, const T a_numPhysPartToRemove) noexcept
324  {
325  CH_TIME("ParticleManagement::removePhysicalParticles");
326 
327  constexpr T zero = (T)0;
328 
329  // Obviously an error that the user should catch.
330  if (a_numPhysPartToRemove < zero) {
331  MayDay::Error("ParticleManagement::removePhysicalParticles - 'a_numPhysPartoToRemove < 0'");
332  }
333 
334  if (a_particles.length() > 0) {
335  ListIterator<P> lit(a_particles);
336 
337  T numRemoved = zero;
338 
339  // 1. Compute the minimum particle weight.
340  T minWeight = std::numeric_limits<T>::max();
341  for (lit.begin(); lit.ok(); ++lit) {
342  minWeight = std::min(minWeight, (T)lit().weight());
343  }
344 
345  // 2. Trim particle weights down to minWeight.
346  for (lit.begin(); lit.ok(); ++lit) {
347  const T diff1 = (T)lit().weight() - minWeight;
348  const T diff2 = a_numPhysPartToRemove - numRemoved;
349 
350  CH_assert(diff1 >= zero);
351  CH_assert(diff2 >= zero);
352 
353  const T r = std::max(0LL, std::min(diff1, diff2));
354 
355  lit().weight() -= 1.0 * r;
356  numRemoved += r;
357  }
358 
359  // 3. "Uniformly" subtract the particle weights.
360  if (a_numPhysPartToRemove - numRemoved > zero) {
361  const T numCompParticles = (T)a_particles.length();
362  const T uniformWeight = (a_numPhysPartToRemove - numRemoved) / numCompParticles;
363  const T uniformRemainder = (a_numPhysPartToRemove - numRemoved) % numCompParticles;
364 
365  // Uniformly remove weight from each particle.
366  if (uniformWeight > zero) {
367  for (lit.begin(); lit.ok(); ++lit) {
368  lit().weight() -= 1.0 * uniformWeight;
369 
370  numRemoved += uniformWeight;
371  }
372  }
373 
374  // May have to remove remainder from multiple particles because their weights might be less
375  // then the actual remainder.
376  if (uniformRemainder > zero) {
377  T W = 0;
378 
379  for (lit.begin(); lit.ok(); ++lit) {
380 
381  // Never remove so that weight is negative.
382  const T w = std::min((T)lit().weight(), uniformRemainder - W);
383 
384  lit().weight() -= 1.0 * w;
385 
386  W += w;
387  numRemoved += w;
388 
389  if (W == uniformRemainder) {
390  break;
391  }
392  }
393  }
394  }
395 
396  // Debug code.
397  CH_assert(numRemoved == a_numPhysPartToRemove);
398 #ifdef NDEBUG
399  for (lit.begin(); lit.ok(); ++lit) {
400  CH_assert(lit().weight() >= 0.0);
401  }
402 #endif
403  }
404  }
