244 template <typename T>
245 void
246 RadiativeTransferStepper<T>::setGaussianSource()
247 {
248  CH_TIME("RadiativeTransferStepper::setGaussianSource");
249 
250  ParmParse pp("RadiativeTransferStepper");
251 
252  Vector<Real> v(SpaceDim);
253  RealVect blobCenter;
254  Real blobAmp;
255  Real blobRad;
256 
257  pp.get("blob_amplitude", blobAmp);
258  pp.get("blob_radius", blobRad);
259  pp.getarr("blob_center", v, 0, SpaceDim);
260  blobCenter = RealVect(D_DECL(v[0], v[1], v[2]));
261 
262  auto sourceFunction = [blobAmp, blobRad, blobCenter](const RealVect a_position) -> Real {
263  const RealVect dist = a_position - blobCenter;
264  const Real dist2 = dist.dotProduct(dist);
265  return blobAmp * exp(-dist2 / (2 * blobRad * blobRad));
266  };
267 
268  m_solver->setSource(sourceFunction);
269 }
