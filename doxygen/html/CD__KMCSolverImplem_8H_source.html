81template <typename R, typename State, typename T>
82inline std::vector<std::vector<T>>
83KMCSolver<R, State, T>::getNu(const State& a_state, const ReactionList& a_reactions) const noexcept
84{
85 CH_TIME("KMCSolver::getNu(State)");
86
87 std::vector<std::vector<T>> ret(a_reactions.size());
88
89 for (int j = 0; j < a_reactions.size(); j++) {
90 std::vector<T>& nu = ret[j];
91
92 // Linearize a trivial state
93 State state = a_state;
94
95 std::vector<T> preState = state.linearOut();
96 for (auto& x : preState) {
97 x = static_cast<T>(0);
98 }
99 state.linearIn(preState);
100
101 // Advance with exactly one reaction.
102 a_reactions[j]->advanceState(state, static_cast<T>(1));
103
104 std::vector<T> postState = state.linearOut();
105
106 // Compute the state change vector.
107 nu.resize(preState.size());
108 for (int i = 0; i < preState.size(); i++) {
109 nu[i] = postState[i] - preState[i];
110 }
111 }
112
113 return ret;
114}

177template <typename R, typename State, typename T>
178inline std::pair<typename KMCSolver<R, State, T>::ReactionList, typename KMCSolver<R, State, T>::ReactionList>
179KMCSolver<R, State, T>::partitionReactions(const State& a_state, const ReactionList& a_reactions) const noexcept
180{
181 CH_TIME("KMCSolver::partitionReactions(State, ReactionList)");
182
183 ReactionList criticalReactions;
184 ReactionList nonCriticalReactions;
185
186 const size_t numReactions = a_reactions.size();
187
188 for (size_t i = 0; i < numReactions; i++) {
189 const T Lj = a_reactions[i]->computeCriticalNumberOfReactions(a_state);
190
191 if (Lj < m_Ncrit) {
192 criticalReactions.emplace_back(a_reactions[i]);
193 }
194 else {
195 nonCriticalReactions.emplace_back(a_reactions[i]);
196 }
197 }
198
199 return std::make_pair(criticalReactions, nonCriticalReactions);
200}

210template <typename R, typename State, typename T>
211inline Real
212KMCSolver<R, State, T>::getCriticalTimeStep(const State& a_state,
213 const ReactionList& a_criticalReactions) const noexcept
214{
215 CH_TIME("KMCSolver::getCriticalTimeStep");
216
217 // TLDR: This computes the time until the firing of the next critical reaction.
218
219 Real dt = std::numeric_limits<Real>::max();
220
221 if (a_criticalReactions.size() > 0) {
222 // Add numeric_limits<Real>::min to A and u to avoid division by zero.
223 const Real A = std::numeric_limits<Real>::min() + this->totalPropensity(a_state, a_criticalReactions);
224
225 dt = this->getCriticalTimeStep(A);
226 }
227
228 return dt;
229}

249KMCSolver<R, State, T>::getCriticalTimeStep(const Real& a_totalPropensity) const noexcept

251 CH_TIME("KineticMonteCarlo::getCriticalTimeStep(Real)");

280template <typename R, typename State, typename T>
281inline Real
282KMCSolver<R, State, T>::getNonCriticalTimeStep(const State& a_state,
283 const ReactionList& a_nonCriticalReactions,
284 const std::vector<Real>& a_nonCriticalPropensities) const noexcept
285{
286 CH_TIME("KMCSolver::getNonCriticalTimeStep(State, ReactionList, std::vector<Real>");
287
288 CH_assert(a_nonCriticalReactions.size() == a_nonCriticalPropensities.size());
289
290 constexpr Real one = 1.0;
291
292 Real dt = std::numeric_limits<Real>::max();
293
294 const size_t numReactions = a_nonCriticalReactions.size();
295
296 if (numReactions > 0) {
297
298 // 1. Gather a list of all reactants involved in the non-critical reactions.
299 auto allReactants = a_nonCriticalReactions[0]->getReactants();
300 for (size_t i = 1; i < numReactions; i++) {
301 const auto& curReactants = a_nonCriticalReactions[i]->getReactants();
302
303 allReactants.insert(allReactants.end(), curReactants.begin(), curReactants.end());
304 }
305
306 // Only do unique reactants.
307 const auto ip = std::unique(allReactants.begin(), allReactants.end());
308 allReactants.resize(std::distance(allReactants.begin(), ip));
309
310 // 2. Iterate through all reactants and compute deviations.
311 for (const auto& reactant : allReactants) {
312
313 // Xi is the population of the current reactant. It might seem weird that we are indexing this
314 // through the reactions rather then through the state. Which it is, but the reason for that is
315 // that it is the REACTION that determines how we index the state population. This is simply a
316 // design choice that permits the user to apply different type of reactions without changing
317 // the underlying state.
318 const T Xi = a_nonCriticalReactions[0]->population(reactant, a_state);
319
320 if (Xi > (T)0) {
321 Real mu = 0.0;
322 Real sigma2 = 0.0;
323
324 // Set gi to 1 for now. A more complex version would parse this through an input parameter
325 // where the user has inspected the highest-order-reaction.
326 constexpr Real gi = 1.0;
327
328 for (size_t i = 0; i < numReactions; i++) {
329 const Real& p = a_nonCriticalPropensities[i];
330 const auto& muIJ = a_nonCriticalReactions[i]->getStateChange(reactant);
331
332 mu += std::abs(muIJ * p);
333 sigma2 += std::abs(muIJ * muIJ * p);
334 }
335
336 Real dt1 = std::numeric_limits<Real>::max();
337 Real dt2 = std::numeric_limits<Real>::max();
338
339 const Real f = std::max(m_eps * Xi / gi, one);
340
341 if (mu > std::numeric_limits<Real>::min()) {
342 dt1 = f / mu;
343 }
344 if (sigma2 > std::numeric_limits<Real>::min()) {
345 dt2 = (f * f) / sigma2;
346 }
347
348 dt = std::min(dt, std::min(dt1, dt2));
349 }
350 }
351 }
352
353 return dt;
354}

380template <typename R, typename State, typename T>
381inline void
382KMCSolver<R, State, T>::stepSSA(State& a_state,
383 const ReactionList& a_reactions,
384 const std::vector<Real>& a_propensities) const noexcept
385{
386 CH_TIME("KMCSolver::stepSSA(State, ReactionList, std::vector<Real>)");
387
388 CH_assert(a_reactions.size() == a_propensities.size());
389
390 const size_t numReactions = a_reactions.size();
391
392 if (numReactions > 0) {
393 constexpr T one = (T)1;
394
395 // Determine the reaction type as per Gillespie algorithm.
396 Real A = 0.0;
397 for (size_t i = 0; i < numReactions; i++) {
398 A += a_propensities[i];
399 }
400
401 const Real u = Random::getUniformReal01();
402
403 size_t r = std::numeric_limits<size_t>::max();
404
405 Real sumProp = 0.0;
406 for (size_t i = 0; i < a_reactions.size(); i++) {
407 sumProp += a_propensities[i];
408
409 if (sumProp >= u * A) {
410 r = i;
411
412 break;
413 }
414 }
415
416 // Advance by one reaction.
417 a_reactions[r]->advanceState(a_state, one);
418 }
419}

430template <typename R, typename State, typename T>
431inline void
432KMCSolver<R, State, T>::advanceSSA(State& a_state, const ReactionList& a_reactions, const Real a_dt) const noexcept
433{
434 CH_TIME("KMCSolver::advanceSSA(State, ReactionList, Real)");
435
436 const size_t numReactions = a_reactions.size();
437
438 if (numReactions > 0) {
439
440 // Simulated time within the SSA.
441 Real curDt = 0.0;
442
443 while (curDt <= a_dt) {
444
445 // Compute the propensities and get the time to the next reaction.
446 const std::vector<Real> propensities = this->propensities(a_state, a_reactions);
447
448 const Real nextDt = this->getCriticalTimeStep(propensities);
449
450 // Fire one reaction if occurs within a_dt.
451 if (curDt + nextDt <= a_dt) {
452 this->stepSSA(a_state, a_reactions, propensities);
453 }
454
455 curDt += nextDt;
456 }
457 }
458}

469template <typename R, typename State, typename T>
470inline void
471KMCSolver<R, State, T>::stepExplicitEuler(State& a_state,
472 const ReactionList& a_reactions,
473 const Real a_dt) const noexcept
474{
475 CH_TIME("KMCSolver::stepExplicitEuler(State, ReactionList, std::vector<Real>, Real)");
476
477 CH_assert(a_dt > 0.0);
478
479 if (a_reactions.size() > 0) {
480 const std::vector<Real> propensities = this->propensities(a_state, a_reactions);
481
482 for (size_t i = 0; i < a_reactions.size(); i++) {
483
484 // Number of reactions is always an integer -- draw from a Poisson distribution in long long. I'm just
485 // using a large integer type to avoid potential overflows.
486 const T numReactions = (T)Random::getPoisson<long long>(propensities[i] * a_dt);
487
488 a_reactions[i]->advanceState(a_state, numReactions);
489 }
490 }

502template <typename R, typename State, typename T>
503inline void
504KMCSolver<R, State, T>::stepMidpoint(State& a_state, const ReactionList& a_reactions, const Real a_dt) const noexcept
505{
506 CH_TIMERS("KMCSolver::stepMidpoint(State&, const ReactionList&, const Real)");
507 CH_TIMER("KMCSolver::stepTauMidPoint::first_loop", t1);
508 CH_TIMER("KMCSolver::stepTauMidPoint::second_loop", t2);
509
510 const int numReactions = a_reactions.size();
511
512 if (numReactions > 0) {
513
514 std::vector<Real> propensities = this->propensities(a_state, a_reactions);
515
516 State Xdagger = a_state;
517
518 CH_START(t1);
519 for (size_t i = 0; i < numReactions; i++) {
520 // TLDR: Predict a midpoint state -- unfortunately this means that as a_dt->0 we end up with plain
521 // tau-leaping. I don't know of a way to fix this without introducing double fluctuations (yet).
522 a_reactions[i]->advanceState(Xdagger, (T)std::round(0.5 * propensities[i] * a_dt));
523 }
524 CH_STOP(t1);
525
526 propensities = this->propensities(Xdagger, a_reactions);
527
528 CH_START(t2);
529 for (size_t i = 0; i < numReactions; i++) {
530 const T curReactions = (T)Random::getPoisson<long long>(propensities[i] * a_dt);
531
532 a_reactions[i]->advanceState(a_state, curReactions);
533 }
534 CH_STOP(t2);
535 }
536}

547template <typename R, typename State, typename T>
548inline void
549KMCSolver<R, State, T>::stepPRC(State& a_state, const ReactionList& a_reactions, const Real a_dt) const noexcept
550{
551 CH_TIMERS("KMCSolver::stepPRC(State&, const ReactionList&, const Real)");
552 CH_TIMER("KMCSolver::stepPRC::first_loop", t1);
553 CH_TIMER("KMCSolver::stepPRC::second_loop", t2);
554
555 const int numReactions = a_reactions.size();
556
557 if (numReactions > 0) {
558
559 std::vector<Real> aj = this->propensities(a_state, a_reactions);
560
561 const std::vector<Real> ak = aj;
562
563 CH_START(t1);
564 for (int j = 0; j < numReactions; j++) {
565 for (int k = 0; k < numReactions; k++) {
566 State x = a_state;
567
568 a_reactions[k]->advanceState(x, (T)1);
569
570 const Real etajk = a_reactions[j]->propensity(x) - ak[j];
571
572 aj[j] += 0.5 * a_dt * ak[k] * etajk;
573 }
574 }
575 CH_STOP(t1);
576
577 CH_START(t2);
578 for (size_t i = 0; i < numReactions; i++) {
579 const T nr = (T)Random::getPoisson<long long>(aj[i] * a_dt);
580
581 a_reactions[i]->advanceState(a_state, nr);
582 }
583 CH_STOP(t2);
584 }
585}

596template <typename R, typename State, typename T>
597inline void
598KMCSolver<R, State, T>::stepImplicitEuler(State& a_state,
599 const ReactionList& a_reactions,
600 const Real a_dt) const noexcept
601{
602 CH_TIME("KMCSolver::stepImplicitEuler(State, ReactionList, a_dt)");
603
604 // TLDR: The implicit Euler tau-leaping scheme is equivalent to the solution of
605 //
606 // F(X) = X - (x + sum_j (nu_j * (P(a(x)*dt) - a(x)*dt)]) - dt*sum_j nu_j a_j(X)
607 // = X - c - dt*sum_j nu_j * a_j(X),
608 // = 0
609 //
610 // where c = (x + sum_j (nu_j * (P(a(x)*dt) - a(x)*dt)]) is a constant term throughout the Newton iterations. This term is presampled
611 // before the Newton iterations begin.
612
613 // Linearize input state onto something understandable to LAPACK
614 const std::vector<T> inputState = a_state.linearOut();
615 const std::vector<std::vector<T>> nu = this->getNu(a_state, a_reactions);
616 const std::vector<Real> ajX = this->propensities(a_state, a_reactions);
617
618 // Number of equations and number of reactions.
619 const int N = inputState.size();
620 const int M = a_reactions.size();
621
622 // Lambda that computes the constant term c and the initial guess.
623 auto compConstantTerm = [&](double* C, double* X, const State& state, const Real a_dt) -> void {
624 // TLDR: To compute the constant term we perform a Poisson sampling and then linearize the output
625 // state.
626
627 State explicitEulerState = state;
628
629 this->advanceTau(explicitEulerState, a_reactions, a_dt, KMCLeapPropagator::ExplicitEuler);
630
631 const std::vector<T> eulerOut = explicitEulerState.linearOut();
632
633 // Make c = (x + sum_j (nu_j * (P(a(x)*dt) - a(x)*dt)])
634 for (int i = 0; i < N; i++) {
635 C[i] = 1.0 * eulerOut[i];
636 X[i] = 1.0 * eulerOut[i];
637 }
638
639 // Subtract the mean.
640 for (int j = 0; j < M; j++) {
641 const std::vector<T>& nuj = nu[j];
642
643 for (int i = 0; i < N; i++) {
644 C[i] -= nuj[i] * ajX[j] * a_dt;
645 }
646 }
647 };
648
649 // Lambda that computes the each equation.
650 auto computeF = [&](double* F, const double* Xit, const double* C) -> void {
651 // Compute the propensities for the Xit state. This requires us to round to the nearest integer.
652 State stateXit = a_state;
653
654 std::vector<T> linState(N);
655 for (int i = 0; i < N; i++) {
656 linState[i] = static_cast<T>(llround(Xit[i]));
657 }
658
659 stateXit.linearIn(linState);
660
661 const std::vector<Real> ajXit = this->propensities(stateXit);
662
663 // Compute Xit - c - dt * sum_j nu_j * aj(round(Xit))
664 for (int i = 0; i < N; i++) {
665 F[i] = Xit[i] - C[i];
666 }
667
668 for (int j = 0; j < M; j++) {
669 const std::vector<T>& nuj = nu[j];
670
671 for (int i = 0; i < N; i++) {
672 F[i] -= a_dt * nuj[i] * ajXit[j];
673 }
674 }
675 };
676
677 // Compute the max-norm
678 auto computeNorm = [&](double* F, const double* X, const double* C) -> Real {
679 computeF(F, X, C);
680
681 Real norm = 0.0;
682
683 for (int i = 0; i < N; i++) {
684 norm = std::max(norm, std::abs(F[i]));
685 }
686
687 return norm;
688 };
689
690 // Allocate memory for the Jacobian (J), the Newton increment (X), and F(X) (F). Also include the constant term from the Poisson sampling (c)
691 double* J = new double[N * N];
692 double* X = new double[N];
693 double* F = new double[N];
694 double* C = new double[N];
695
696 // Temporary storage used for computing the Jacobian matrix.
697 double* X1 = new double[N];
698 double* X2 = new double[N];
699 double* F1 = new double[N];
700 double* F2 = new double[N];
701
702 // Things that are required by LAPACK
703 int INFO = 0;
704 int NRHS = 1;
705 int* IPIV = new int[N];
706
707 // Compute the constant term.
708 compConstantTerm(C, X, a_state, a_dt);
709
710 // Compute the max-norm of F(0) to use as an exit criterion.
711 for (int i = 0; i < N; i++) {
712 X1[i] = 0.0;
713 }
714
715 const Real initNorm = computeNorm(F, X1, C);
716
717 bool converged = true;
718
719 for (int k = 0; k < m_maxIter; k++) {
720
721 // Numerically computed the Jacobian using finite differences. The Jacobian is given by
722 // Jij = dF_i/dx_j
723 for (int j = 0; j < N; j++) {
724
725 for (int s = 0; s < N; s++) {
726 X1[s] = X[s];
727 X2[s] = X[s];
728 }
729
730 X2[j] += std::max(0.01 * X[j], 1.0);
731
732 computeF(F1, X1, C);
733 computeF(F2, X2, C);
734
735 for (int i = 0; i < N; i++) {
736 J[i + j * N] = (F2[i] - F1[i]) / (X2[j] - X1[j]);
737 }
738 }
739
740 // Compute F
741 computeF(F, X, C);
742
743 // Solve J*dX = F, but note that the true system is J*dX = -F, so we invert the
744 // dX vector below.
745 dgesv_((int*)&N, (int*)&NRHS, J, (int*)&N, IPIV, F, (int*)&N, &INFO);
746
747 if (INFO != 0) {
748#if 1 // Could not solve
749 const std::string err = "KMCSolver<R, State, T>::stepImplicitEuler -- could not solve A*x = b";
750
751 pout() << err << endl;
752#endif
753 converged = false;
754
755 break;
756 }
757 else {
758 // Increment and move on to next iteration if necessary. Note that F = -dX as per the comment above.
759 for (int i = 0; i < N; i++) {
760 X[i] = X[i] - F[i];
761 }
762 }
763
764 // Recompute the norm and exit if necessary.
765 const Real norm = computeNorm(F, X, C);
766
767 if (norm / initNorm < m_exitTol) {
768 break;
769 }
770 }
771
772 // Turn X into an integer state.
773 std::vector<T> outputState(N);
774
775 if (converged) {
776 for (int i = 0; i < N; i++) {
777 outputState[i] = static_cast<T>(llround(X[i]));
778 }
779 }
780 else {
781 // Set X to (what is hopefully) an invalid state and rely on step rejection.
782 for (int i = 0; i < N; i++) {
783 outputState[i] = static_cast<T>(-1);
784 }
785 }
786
787 a_state.linearIn(outputState);
788
789 // Cleanup
790 delete[] J;
791 delete[] X;
792 delete[] F;
793 delete[] C;
794
795 delete[] X1;
796 delete[] X2;
797 delete[] F1;
798 delete[] F2;
799
800 delete[] IPIV;
801
802 return;
803}

817inline void
818KMCSolver<R, State, T>::advanceTau(State& a_state,
819 const ReactionList& a_reactions,
820 const Real& a_dt,
821 const KMCLeapPropagator& a_leapPropagator) const noexcept
822{
823 CH_TIME("KMCSolver::advanceTau(State, ReactionList, Real, KMCLeapPropagator)");
824
825 if (a_reactions.size() > 0) {
826 Real curTime = 0.0;
827 Real curDt = a_dt;
828
829 while (curTime < a_dt) {
830
831 // Try big time step first.
832 curDt = a_dt - curTime;
833
834 bool valid = false;
835
836 // Substepping so we end up with a valid state.
837 while (!valid) {
838
839 State state = a_state;
840
841 // Do a tau-leaping step.
842 switch (a_leapPropagator) {
843 case KMCLeapPropagator::ExplicitEuler: {
844 this->stepExplicitEuler(state, a_reactions, curDt);
845
846 break;
847 }
848 case KMCLeapPropagator::Midpoint: {
849 this->stepMidpoint(state, a_reactions, curDt);
850
851 break;
852 }
853 case KMCLeapPropagator::PRC: {
854 this->stepPRC(state, a_reactions, curDt);
855
856 break;
857 }
858 case KMCLeapPropagator::ImplicitEuler: {
859 this->stepImplicitEuler(state, a_reactions, curDt);
860
861 break;
862 }
863 default: {
864 break;
865 }
866 }
867
868 // If this was a valid step, accept it. Else reduce dt.
869 valid = state.isValidState();
870
871 if (valid) {
872 a_state = state;
873
874 curTime += curDt;
875 }
876 else {
877 curDt *= 0.5;

879 }

939template <typename R, typename State, typename T>
940inline void
941KMCSolver<R, State, T>::advanceHybrid(
942 State& a_state,
943 const ReactionList& a_reactions,
944 const Real a_dt,
945 const std::function<void(State&, const ReactionList& a_reactions, const Real a_dt)>& a_propagator) const noexcept
946{
947 CH_TIME("KMCSolver::advanceHybrid(State, ReactionList, Real, std::function)");
948
949 constexpr T one = (T)1;
950
951 // Simulated time within the advancement algorithm.
952 Real curTime = 0.0;
953
954 // Outer loop is for reactive substepping over a_dt.
955 while (curTime < a_dt) {
956
957 // Partition reactions into critical and non-critical reactions and compute the critical and non-critical time steps.
958 const std::pair<ReactionList, ReactionList> partitionedReactions = this->partitionReactions(a_state, a_reactions);
959
960 const ReactionList& criticalReactions = partitionedReactions.first;
961 const ReactionList& nonCriticalReactions = partitionedReactions.second;
962
963 const std::vector<Real> propensitiesCrit = this->propensities(a_state, criticalReactions);
964 const std::vector<Real> propensitiesNonCrit = this->propensities(a_state, nonCriticalReactions);
965
966 Real dtCrit = this->getCriticalTimeStep(propensitiesCrit);
967 Real dtNonCrit = this->getNonCriticalTimeStep(a_state, nonCriticalReactions, propensitiesNonCrit);
968
969 // Try the various advancement algorithms; the loop is for step rejection in case we end up with an invalid state, e.g.
970 // a state with a negative number of particles.
971 bool validStep = false;
972
973 while (!validStep) {
974
975 // Do a backup of the advancement state to operate on. This is necessary because the tau-leaping
976 // algorithms advance the state but we may need to reject those steps if we end up with a thermodynamically
977 // invalid state.
978 State state = a_state;
979
980 // Compute the time step to be used.
981 const Real curDt = std::min(a_dt - curTime, std::min(dtCrit, dtNonCrit));
982
983 // Are we only doing non-critical reactions?
984 const bool nonCriticalOnly = (dtNonCrit < dtCrit) || (criticalReactions.size() == 0) ||
985 (dtCrit > (a_dt - curTime));
986
987 if (nonCriticalOnly) {
988
989 // Compute the total propensity because it might actually be better to run SSA steps.
990 Real A = this->totalPropensity(state, a_reactions);
991
992 if (A * curDt < m_SSAlim) {
993 // TLDR: Tau-leaping is inefficient here so we switch to an SSA-based algorithm for the WHOLE reaction set.
994
995 // Number of SSA steps taken and simulated time within the SSA. We will compute until either dtSSA < curDt or
996 // we've exceeded the maximum number of SSA steps that the user has permitted (m_numSSA).
997 Real dtSSA = 0.0;
998 T numSSA = 0;
999
1000 while (dtSSA < curDt && numSSA < m_numSSA) {
1001
1002 // Recompute propensities for the full reaction set and advance everything using the SSA.
1003 std::vector<Real> propensities = this->propensities(a_state, a_reactions);
1004
1005 A = 0.0;
1006 for (size_t i = 0; i < propensities.size(); i++) {
1007 A += propensities[i];
1008 }
1009
1010 // Compute the time to the next reaction.
1011 const Real dtReact = this->getCriticalTimeStep(A);
1012
1013 if (dtSSA + dtReact < curDt) {
1014 this->stepSSA(a_state, a_reactions, propensities);
1015
1016 dtSSA += dtReact;
1017 numSSA += one;
1018 }
1019 else {
1020
1021 // Next reaction occured outside the substep -- break out of the loop.
1022 dtSSA = curDt;
1023 }
1024 }
1025
1026 validStep = true;
1027 curTime += dtSSA;
1028 }
1029 else {
1030 // Do the tau-leaping.
1031 a_propagator(state, nonCriticalReactions, curDt);
1032
1033 // Check if we need to reject the state and rather try again with a smaller non-critical time step.
1034 validStep = state.isValidState();
1035
1036 if (validStep) {
1037 a_state = state;
1038
1039 curTime += curDt;
1040 }
1041 else {
1042 dtNonCrit *= 0.5;
1043 }
1044 }
1045 }
1046 else {
1047 // TLDR: Here, one critical reaction fired. We advance the critical reaction using the SSA and the rest of the reactions
1048 // using tau-leaping.
1049 this->stepSSA(state, criticalReactions, propensitiesCrit);
1050
1051 // Do the tau-leaping over the non-critical reactions.
1052 a_propagator(state, nonCriticalReactions, curDt);
1053
1054 // Check if we need to reject the state and rather try again with a smaller non-critical time step.
1055 validStep = state.isValidState();
1056
1057 if (validStep) {
1058 a_state = state;
1059
1060 curTime += curDt;
1061 }
1062 else {
1063 dtNonCrit *= 0.5;
1064 }
1065 }
1066 }
1067 }
1068}
