133template <typename R, typename State, typename T>
134inline std::pair<typename KMCSolver<R, State, T>::ReactionList, typename KMCSolver<R, State, T>::ReactionList>
135KMCSolver<R, State, T>::partitionReactions(const State& a_state, const ReactionList& a_reactions) const noexcept
136{
137 CH_TIME("KMCSolver::partitionReactions(State, ReactionList)");
138
139 ReactionList criticalReactions;
140 ReactionList nonCriticalReactions;
141
142 const size_t numReactions = a_reactions.size();
143
144 for (size_t i = 0; i < numReactions; i++) {
145 const T Lj = a_reactions[i]->computeCriticalNumberOfReactions(a_state);
146
147 if (Lj < m_Ncrit) {
148 criticalReactions.emplace_back(a_reactions[i]);
149 }
150 else {
151 nonCriticalReactions.emplace_back(a_reactions[i]);
152 }
153 }
154
155 return std::make_pair(criticalReactions, nonCriticalReactions);
156}

166template <typename R, typename State, typename T>
167inline Real
168KMCSolver<R, State, T>::getCriticalTimeStep(const State& a_state,
169 const ReactionList& a_criticalReactions) const noexcept
170{
171 CH_TIME("KMCSolver::getCriticalTimeStep");
172
173 // TLDR: This computes the time until the firing of the next critical reaction.
174
175 Real dt = std::numeric_limits<Real>::max();
176
177 if (a_criticalReactions.size() > 0) {
178 // Add numeric_limits<Real>::min to A and u to avoid division by zero.
179 const Real A = std::numeric_limits<Real>::min() + this->totalPropensity(a_state, a_criticalReactions);
180
181 dt = this->getCriticalTimeStep(A);
182 }
183
184 return dt;
185}

187template <typename R, typename State, typename T>
188inline Real
189KMCSolver<R, State, T>::getCriticalTimeStep(const std::vector<Real>& a_propensities) const noexcept
190{
191 CH_TIME("KineticMonteCarlo::getCriticalTimeStep(std::vector<Real>)");
192
193 // To avoid division by zero later on.
194 Real A = std::numeric_limits<Real>::min();
195
196 for (const auto& p : a_propensities) {
197 A += p;
198 }
199
200 return this->getCriticalTimeStep(A);
201}

203template <typename R, typename State, typename T>
204inline Real
205KMCSolver<R, State, T>::getCriticalTimeStep(const Real& a_totalPropensity) const noexcept
206{
207 CH_TIME("KineticMonteCarlo::getCriticalTimeStep(Real)");
208
209 const Real u = std::numeric_limits<Real>::min() + Random::getUniformReal01();
210
211 return log(1.0 / u) / a_totalPropensity;

236template <typename R, typename State, typename T>
237inline Real
238KMCSolver<R, State, T>::getNonCriticalTimeStep(const State& a_state,
239 const ReactionList& a_nonCriticalReactions,
240 const std::vector<Real>& a_nonCriticalPropensities) const noexcept
241{
242 CH_TIME("KMCSolver::getNonCriticalTimeStep(State, ReactionList, std::vector<Real>");
243
244 CH_assert(a_nonCriticalReactions.size() == a_nonCriticalPropensities.size());
245
246 constexpr Real one = 1.0;
247
248 Real dt = std::numeric_limits<Real>::max();
249
250 const size_t numReactions = a_nonCriticalReactions.size();
251
252 if (numReactions > 0) {
253
254 // 1. Gather a list of all reactants involved in the non-critical reactions.
255 auto allReactants = a_nonCriticalReactions[0]->getReactants();
256 for (size_t i = 1; i < numReactions; i++) {
257 const auto& curReactants = a_nonCriticalReactions[i]->getReactants();
258
259 allReactants.insert(allReactants.end(), curReactants.begin(), curReactants.end());
260 }
261
262 // Only do unique reactants.
263 const auto ip = std::unique(allReactants.begin(), allReactants.end());
264 allReactants.resize(std::distance(allReactants.begin(), ip));
265
266 // 2. Iterate through all reactants and compute deviations.
267 for (const auto& reactant : allReactants) {
268
269 // Xi is the population of the current reactant. It might seem weird that we are indexing this
270 // through the reactions rather then through the state. Which it is, but the reason for that is
271 // that it is the REACTION that determines how we index the state population. This is simply a
272 // design choice that permits the user to apply different type of reactions without changing
273 // the underlying state.
274 const T Xi = a_nonCriticalReactions[0]->population(reactant, a_state);
275
276 if (Xi > (T)0) {
277 Real mu = 0.0;
278 Real sigma2 = 0.0;
279
280 // Set gi to 4 for now -- this is equivalent to a third-order reaction Si*Si*Si with Xi = 3, I doubt
281 // that the Cao computation of gi is necessary because it becomes an "approximation within an approximation". So
282 // I'm just setting gi to the "worst-case" scenario and live with that.
283 constexpr Real gi = 4.0;
284
285 for (size_t i = 0; i < numReactions; i++) {
286 const Real& p = a_nonCriticalPropensities[i];
287 const auto& muIJ = a_nonCriticalReactions[i]->getStateChange(reactant);
288
289 mu += std::abs(muIJ * p);
290 sigma2 += std::abs(muIJ * muIJ * p);
291 }
292
293 Real dt1 = std::numeric_limits<Real>::max();
294 Real dt2 = std::numeric_limits<Real>::max();
295
296 const Real f = std::max(m_eps * Xi / gi, one);
297
298 if (mu > std::numeric_limits<Real>::min()) {
299 dt1 = f / mu;
300 }
301 if (sigma2 > std::numeric_limits<Real>::min()) {
302 dt2 = (f * f) / sigma2;
303 }
304
305 dt = std::min(dt, std::min(dt1, dt2));
306 }
307 }
308 }
309
310 return dt;
311}

333template <typename R, typename State, typename T>
334inline void
335KMCSolver<R, State, T>::stepTauPlain(State& a_state,
336 const ReactionList& a_reactions,
337 const std::vector<Real>& a_propensities,
338 const Real a_dt) const noexcept
339{
340 CH_TIME("KMCSolver::stepTauPlain(State, ReactionList, std::vector<Real>, Real)");
341
342 CH_assert(a_reactions.size() == a_propensities.size());
343
344 if (a_reactions.size() > 0) {
345 for (size_t i = 0; i < a_reactions.size(); i++) {
346
347 // Number of reactions is always an integer -- draw from a Poisson distribution in long long. I'm just
348 // using a large integer type to avoid potential overflows.
349 const T numReactions = (T)Random::getPoisson<long long>(a_propensities[i] * a_dt);
350
351 a_reactions[i]->advanceState(a_state, numReactions);
352 }
353 }

365template <typename R, typename State, typename T>
366inline void
367KMCSolver<R, State, T>::advanceTauPlain(State& a_state, const ReactionList& a_reactions, const Real a_dt) const noexcept
368{
369 CH_TIME("KMCSolver::advanceTauPlain(State, ReactionList, Real)");
370
371 if (a_reactions.size() > 0) {
372 Real curTime = 0.0;
373 Real curDt = a_dt;
374
375 while (curTime < a_dt) {
376
377 // Try big time step first.
378 curDt = a_dt - curTime;
379
380 bool valid = false;
381
382 // Substepping so we end up with a valid state.
383 while (!valid) {
384
385 State state = a_state;
386
387 // Do a tau-leaping step.
388 this->stepTauPlain(state, a_reactions, curDt);
389
390 // If this was a valid step, accept it. Else reduce dt.
391 valid = state.isValidState();
392
393 if (valid) {
394 a_state = state;
395
396 curTime += curDt;
397 }
398 else {
399 curDt *= 0.5;
400 }
401 }
402 }
403 }
404}

415template <typename R, typename State, typename T>
416inline void
417KMCSolver<R, State, T>::stepTauMidpoint(State& a_state, const ReactionList& a_reactions, const Real a_dt) const noexcept
418{
419 CH_TIME("KMCSolver::stepTauMidpoint(State&, const ReactionList&, const Real)");
420
421 const int numReactions = a_reactions.size();
422
423 if (numReactions > 0) {
424
425 std::vector<Real> propensities = this->propensities(a_state, a_reactions);
426
427 State Xdagger = a_state;
428
429 for (size_t i = 0; i < numReactions; i++) {
430 // TLDR: Predict a midpoint state -- unfortunately this means that as a_dt->0 we end up with plain
431 // tau-leaping. I don't know of a way to fix this without introducing double fluctuations (yet).
432 a_reactions[i]->advanceState(Xdagger, (T)std::ceil(0.5 * propensities[i] * a_dt));
433 }
434
435 propensities = this->propensities(Xdagger, a_reactions);
436 for (size_t i = 0; i < numReactions; i++) {
437 const T curReactions = (T)Random::getPoisson<long long>(propensities[i] * a_dt);
438
439 a_reactions[i]->advanceState(a_state, curReactions);
440 }
441 }
442}

505template <typename R, typename State, typename T>
506inline void
507KMCSolver<R, State, T>::stepTauPRC(State& a_state, const ReactionList& a_reactions, const Real a_dt) const noexcept
508{
509 CH_TIME("KMCSolver::stepTauPRC(State&, const ReactionList&, const Real)");
510
511 const int numReactions = a_reactions.size();
512
513 if (numReactions > 0) {
514
515 std::vector<Real> aj = this->propensities(a_state, a_reactions);
516
517 const std::vector<Real> ak = aj;
518
519 for (int j = 0; j < numReactions; j++) {
520 for (int k = 0; k < numReactions; k++) {
521 State x = a_state;
522
523 a_reactions[k]->advanceState(x, (T)1);
524
525 const Real etajk = a_reactions[j]->propensity(x) - ak[j];
526
527 aj[j] += 0.5 * a_dt * ak[k] * etajk;
528 }
529 }
530
531 for (size_t i = 0; i < numReactions; i++) {
532 const T nr = (T)Random::getPoisson<long long>(aj[i] * a_dt);
533
534 a_reactions[i]->advanceState(a_state, nr);
535 }
536 }
537}

613template <typename R, typename State, typename T>
614inline void
615KMCSolver<R, State, T>::stepSSA(State& a_state,
616 const ReactionList& a_reactions,
617 const std::vector<Real>& a_propensities) const noexcept
618{
619 CH_TIME("KMCSolver::stepSSA(State, ReactionList, std::vector<Real>)");
620
621 CH_assert(a_reactions.size() == a_propensities.size());
622
623 const size_t numReactions = a_reactions.size();
624
625 if (numReactions > 0) {
626 constexpr T one = (T)1;
627
628 // Determine the reaction type as per Gillespie algorithm.
629 Real A = 0.0;
630 for (size_t i = 0; i < numReactions; i++) {
631 A += a_propensities[i];
632 }
633
634 const Real u = Random::getUniformReal01();
635
636 size_t r = std::numeric_limits<size_t>::max();
637
638 Real sumProp = 0.0;
639 for (size_t i = 0; i < a_reactions.size(); i++) {
640 sumProp += a_propensities[i];
641
642 if (sumProp >= u * A) {
643 r = i;
644
645 break;
646 }
647 }
648
649 // Advance by one reaction.
650 a_reactions[r]->advanceState(a_state, one);
651 }
652}

663template <typename R, typename State, typename T>
664inline void
665KMCSolver<R, State, T>::advanceSSA(State& a_state, const ReactionList& a_reactions, const Real a_dt) const noexcept
666{
667 CH_TIME("KMCSolver::advanceSSA(State, ReactionList, Real)");
668
669 const size_t numReactions = a_reactions.size();
670
671 if (numReactions > 0) {
672
673 // Simulated time within the SSA.
674 Real curDt = 0.0;
675
676 while (curDt <= a_dt) {
677
678 // Compute the propensities and get the time to the next reaction.
679 const std::vector<Real> propensities = this->propensities(a_state, a_reactions);
680
681 const Real nextDt = this->getCriticalTimeStep(propensities);
682
683 // Fire one reaction if occurs within a_dt.
684 if (curDt + nextDt <= a_dt) {
685 this->stepSSA(a_state, a_reactions, propensities);
686 }
687
688 curDt += nextDt;
689 }
690 }
691}

711 CH_TIME("KMCSolver::advanceHybrid(State, ReactionList, Real, KMCLeapPropagator)");
712
713 switch (a_leapPropagator) {
714 case KMCLeapPropagator::TauPlain: {
715 this->advanceHybrid(a_state, a_reactions, a_dt, [this](State& s, const ReactionList& r, const Real dt) {
716 this->stepTauPlain(s, r, dt);
717 });
718
719 break;
720 }
721 case KMCLeapPropagator::TauMidpoint: {
722 this->advanceHybrid(a_state, a_reactions, a_dt, [this](State& s, const ReactionList& r, const Real dt) {
723 this->stepTauMidpoint(s, r, dt);
724 });
725
726 break;
727 }
728 case KMCLeapPropagator::TauPRC: {
729 this->advanceHybrid(a_state, a_reactions, a_dt, [this](State& s, const ReactionList& r, const Real dt) {
730 this->stepTauPRC(s, r, dt);
731 });
732
733 break;
734 }
735 default: {
736 MayDay::Error("KMCSolver::advanceHybrid - unknown leap propagator requested");
737 }
738 }
739}
740
741template <typename R, typename State, typename T>
742inline void
743KMCSolver<R, State, T>::advanceHybrid(
744 State& a_state,
745 const ReactionList& a_reactions,
746 const Real a_dt,
747 const std::function<void(State&, const ReactionList& a_reactions, const Real a_dt)>& a_propagator) const noexcept
748{
749 CH_TIME("KMCSolver::advanceHybrid(State, ReactionList, Real, std::function)");
750
751 constexpr T one = (T)1;
752
753 // Simulated time within the advancement algorithm.
754 Real curTime = 0.0;
755
756 // Outer loop is for reactive substepping over a_dt.
757 while (curTime < a_dt) {
758
759 // Partition reactions into critical and non-critical reactions and compute the critical and non-critical time steps.
760 const std::pair<ReactionList, ReactionList> partitionedReactions = this->partitionReactions(a_state, a_reactions);
761
762 const ReactionList& criticalReactions = partitionedReactions.first;
763 const ReactionList& nonCriticalReactions = partitionedReactions.second;
764
765 const std::vector<Real> propensitiesCrit = this->propensities(a_state, criticalReactions);
766 const std::vector<Real> propensitiesNonCrit = this->propensities(a_state, nonCriticalReactions);
767
768 Real dtCrit = this->getCriticalTimeStep(propensitiesCrit);
769 Real dtNonCrit = this->getNonCriticalTimeStep(a_state, nonCriticalReactions, propensitiesNonCrit);
770
771 // Try the various advancement algorithms; the loop is for step rejection in case we end up with an invalid state, e.g.
772 // a state with a negative number of particles.
773 bool validStep = false;
774
775 while (!validStep) {
776
777 // Do a backup of the advancement state to operate on. This is necessary because the tau-leaping
778 // algorithms advance the state but we may need to reject those steps if we end up with a thermodynamically
779 // invalid state.
780 State state = a_state;
781
782 // Compute the time step to be used.
783 const Real curDt = std::min(a_dt - curTime, std::min(dtCrit, dtNonCrit));
784
785 // Are we only doing non-critical reactions?
786 const bool nonCriticalOnly = (dtNonCrit < dtCrit) || (criticalReactions.size() == 0) ||
787 (dtCrit > (a_dt - curTime));
788
789 if (nonCriticalOnly) {
790
791 // Compute the total propensity because it might actually be better to run SSA steps.
792 Real A = this->totalPropensity(state, a_reactions);
793
794 if (A * curDt < m_SSAlim) {
795 // TLDR: Tau-leaping is inefficient here so we switch to an SSA-based algorithm for the WHOLE reaction set.
796
797 // Number of SSA steps taken and simulated time within the SSA. We will compute until either dtSSA < curDt or
798 // we've exceeded the maximum number of SSA steps that the user has permitted (m_numSSA).
799 Real dtSSA = 0.0;
800 T numSSA = 0;
801
802 while (dtSSA < curDt && numSSA < m_numSSA) {
803
804 // Recompute propensities for the full reaction set and advance everything using the SSA.
805 std::vector<Real> propensities = this->propensities(a_state, a_reactions);
806
807 A = 0.0;
808 for (size_t i = 0; i < propensities.size(); i++) {
809 A += propensities[i];
810 }
811
812 // Compute the time to the next reaction.
813 const Real dtReact = this->getCriticalTimeStep(A);
814
815 if (dtSSA + dtReact < curDt) {
816 this->stepSSA(a_state, a_reactions, propensities);
817
818 dtSSA += dtReact;
819 numSSA += one;
820 }
821 else {
822
823 // Next reaction occured outside the substep -- break out of the loop.
824 dtSSA = curDt;
825 }
826 }
827
828 validStep = true;
829 curTime += dtSSA;
830 }
831 else {
832 // Do the tau-leaping.
833 a_propagator(state, nonCriticalReactions, curDt);
834
835 // Check if we need to reject the state and rather try again with a smaller non-critical time step.
836 validStep = state.isValidState();
837
838 if (validStep) {
839 a_state = state;
840
841 curTime += curDt;
842 }
843 else {
844 dtNonCrit *= 0.5;
845 }
846 }
847 }
848 else {
849 // TLDR: Here, one critical reaction fired. We advance the critical reaction using the SSA and the rest of the reactions
850 // using tau-leaping.
851 this->stepSSA(state, criticalReactions, propensitiesCrit);
852
853 // Do the tau-leaping over the non-critical reactions.
854 a_propagator(state, nonCriticalReactions, curDt);
855
856 // Check if we need to reject the state and rather try again with a smaller non-critical time step.
857 validStep = state.isValidState();
858
859 if (validStep) {
860 a_state = state;
861
862 curTime += curDt;
863 }
864 else {
865 dtNonCrit *= 0.5;
866 }
867 }
868 }
869 }
870}
