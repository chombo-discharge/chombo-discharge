203  inline
204  void Tree<T>::buildTree(const int a_firstDir, const int a_numTargetLeaves, const NodePartitioner<T>& a_partitioner){
205  CH_TIME("ItoMerge::Tree<T>::buildTree");
206 
207  CH_assert(m_isDefined );
208  CH_assert(a_numTargetLeaves >= 1 );
209  CH_assert(a_firstDir >= 0 );
210  CH_assert(a_firstDir <= SpaceDim-1);
211 
212  // Should evaluate to an empty loop and be purged by the compiler if DEBUG=FALSE
213  for (const auto& d : m_root->getData()){
214  CH_assert(d.mass() >= 1.0);
215  }
216 
217  // TLDR: This function splits the primitives recursively until we either have a_numTargetLeaves leaf nodes OR none of the leaf nodes can be
218  // partitioned further.
219 
220  std::vector<std::shared_ptr<Node<T> > > newLeaves;
221 
222  // Reset leaves and root. I'm just resizing here so I get the capacity
223  // and avoid vector resizing during the splitting process.
224  m_leaves. resize(a_numTargetLeaves, nullptr);
225  newLeaves.resize(a_numTargetLeaves, nullptr);
226  m_leaves[0] = m_root;
227 
228  // We always begin with one leaf node.
229  int numCurrentLeaves = 1;
230 
231  int splitDir = a_firstDir;
232  bool keepSplitting = (a_numTargetLeaves > 1);
233 
234  while(keepSplitting){
235 
236  // Flag which determines if we can split any of the nodes further. This will remain false if none of the
237  // leaf nodes could be split OR if we didn't need more leaf nodes (i.e. the tree has finished building).
238  bool canSplit = false;
239 
240 #if 0 // I'm not sure this matters so I'm commenting it out. R.M. Nov. 2021.
241  // Sort the leaves by their mass(?). We will split the ones with the largest masses.
242  std::sort(m_leaves.begin(), m_leaves.begin() + numCurrentLeaves,
243  [](const std::shared_ptr<Node<T> >& n1,
244  const std::shared_ptr<Node<T> >& n2){
245  return n1->mass() < n2->mass();
246  });
247 #endif
248 
249  // These are the number of leaves that we need in order to reach a_numTargetLeaves.
250  int leavesNeeded = a_numTargetLeaves - numCurrentLeaves;
251 
252  int N = 0;
253  for (int i = 0; i < numCurrentLeaves; i++){
254 
255  // If we needed a new leaf AND we could split this node, split it and go further. This will give us two leaf nodes
256  // in the "new" list of leaf nodes.
257  if(m_leaves[i]->canSplit() && leavesNeeded > 0){
258  a_partitioner(m_leaves[i], splitDir);
259 
260  newLeaves[N ] = m_leaves[i]->getLeft ();
261  newLeaves[N+1] = m_leaves[i]->getRight();
262 
263  N += 2;
264 
265  leavesNeeded--;
266 
267  canSplit = true;
268  }
269  else{ // We did not split the leaf node so just append the current leaf to the list of "new" leaves.
270  newLeaves[N] = m_leaves[i];
271 
272  N++;
273  }
274  }
275 
276  // Assign new leaf nodes.
277  m_leaves.assign(newLeaves.begin(), newLeaves.begin() + N);
278 
279  // Update the current number of leaf nodes.
280  numCurrentLeaves = N;
281 
282  // Did the tree finish building or are all the nodes unsplittable leaf nodes? If so, exit.
283  keepSplitting = (N < a_numTargetLeaves) && canSplit;
284 
285  // Update the splitting coordinate -- the next leaves are split along a different coordinate.
286  splitDir = (splitDir + 1) % SpaceDim;
287  }
288  }
289 }
