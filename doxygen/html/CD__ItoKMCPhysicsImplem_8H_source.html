122 inline void
123 ItoKMCPhysics::definePhotoPathways() noexcept
124 {
125  CH_TIME("ItoKMCPhysics::definePhotoPathways");
126 
127  // Build a temporary list of pathways. I.e. restructure the list of reactions
128  //
129  // Y1 -> A
130  // Y1 -> B
131  // Y2 -> C
132  // Y2 -> D
133  //
134  // into separate lists for Y1, Y2, ....
135  //
136  std::map<int, std::vector<std::pair<int, Real>>> pathways;
137 
138  for (int i = 0; i < m_photoReactions.size(); i++) {
139  const ItoKMCPhotoReaction& r = *m_photoReactions[i];
140 
141  const size_t& src = r.getSourcePhoton();
142  const Real efficiency = r.getEfficiency();
143 
144  pathways[src].emplace_back(std::make_pair(i, efficiency));
145  }
146 
147  // Go through the temporary pathways list and compute the relative efficiencies of one of the
148  // photons triggering a reaction. The relative efficiencies are given by
149  //
150  // p(i) = R(i)/sum_j R(j).
151  //
152  for (const auto& p : pathways) {
153  const int photoSpecies = p.first;
154  const std::vector<std::pair<int, Real>> reactionsAndEfficiencies = p.second;
155 
156  std::map<int, int> localToGlobalMap;
157  std::list<double> efficiencies;
158  double sumEfficiencies = 0.0;
159 
160  for (int i = 0; i < reactionsAndEfficiencies.size(); i++) {
161  sumEfficiencies += (double)reactionsAndEfficiencies[i].second;
162  }
163 
164  for (int i = 0; i < reactionsAndEfficiencies.size(); i++) {
165  localToGlobalMap.emplace(i, reactionsAndEfficiencies[i].first);
166  efficiencies.emplace_back((double)reactionsAndEfficiencies[i].second / sumEfficiencies);
167  }
168 
169  std::discrete_distribution<int> distribution(efficiencies.begin(), efficiencies.end());
170 
171  m_photoPathways.insert(std::make_pair((int)photoSpecies, std::make_pair(distribution, localToGlobalMap)));
172  }
173 }

310 inline void
311 ItoKMCPhysics::advanceKMC(Vector<FPR>& a_numParticles,
312  Vector<FPR>& a_numNewPhotons,
313  const Vector<Real>& a_phi,
314  const Vector<RealVect>& a_gradPhi,
315  const Real a_dt,
316  const RealVect a_E,
317  const RealVect a_pos,
318  const Real a_dx,
319  const Real a_kappa) const
320 {
321  CH_TIME("ItoKMCPhysics::advanceKMC");
322 
323  CH_assert(m_isDefined);
324 
325  // Populate the KMC solver state.
326  std::vector<FPR>& kmcParticles = m_kmcState.getReactiveState();
327  std::vector<FPR>& kmcPhotons = m_kmcState.getNonReactiveState();
328 
329  for (size_t i = 0; i < a_numParticles.size(); i++) {
330  kmcParticles[i] = a_numParticles[i];
331  }
332 
333  for (auto& p : kmcPhotons) {
334  p = 0LL;
335  }
336 
337  // Update the reaction rates and run the KMC solver.
338  this->updateReactionRates(a_E, a_pos, a_phi, a_gradPhi, a_dx, a_kappa);
339 
340  // // Which algorithm?
341  switch (m_algorithm) {
342  case Algorithm::SSA: {
343  m_kmcSolver.advanceSSA(m_kmcState, a_dt);
344 
345  break;
346  }
347  case Algorithm::TauPlain: {
348  m_kmcSolver.advanceTauPlain(m_kmcState, a_dt);
349 
350  break;
351  }
352  case Algorithm::TauMidpoint: {
353  m_kmcSolver.advanceTauMidpoint(m_kmcState, a_dt);
354 
355  break;
356  }
357  case Algorithm::HybridPlain: {
358  m_kmcSolver.advanceHybrid(m_kmcState, a_dt, KMCLeapPropagator::TauPlain);
359 
360  break;
361  }
362  case Algorithm::HybridMidpoint: {
363  m_kmcSolver.advanceHybrid(m_kmcState, a_dt, KMCLeapPropagator::TauMidpoint);
364 
365  break;
366  }
367  default: {
368  MayDay::Error("ItoKMCPhysics::advanceKMC - logic bust");
369  }
370  }
371 
372  // Put KMC back into ItoKMC
373  for (size_t i = 0; i < a_numParticles.size(); i++) {
374  a_numParticles[i] = (FPR)kmcParticles[i];
375  }
376  for (size_t i = 0; i < a_numNewPhotons.size(); i++) {
377  a_numNewPhotons[i] = (FPR)kmcPhotons[i];
378  }
379 }

381 inline void
382 ItoKMCPhysics::reconcileParticles(Vector<List<ItoParticle>*>& a_particles,
383  const Vector<FPR>& a_newNumParticles,
384  const Vector<FPR>& a_oldNumParticles,
385  const RealVect a_cellPos,
386  const RealVect a_centroidPos,
387  const RealVect a_lo,
388  const RealVect a_hi,
389  const RealVect a_bndryCentroid,
390  const RealVect a_bndryNormal,
391  const Real a_dx,
392  const Real a_kappa) const noexcept
393 {
394  CH_TIME("ItoKMCPhysics::reconcileParticles(Vector<List<ItoParticle>*>, ...)");
395 
396  CH_assert(m_isDefined);
397  CH_assert(a_particles.size() == a_newNumParticles.size());
398  CH_assert(a_oldNumParticles.size() == a_newNumParticles.size());
399 
400  if (m_debug) {
401  for (int i = 0; i < a_particles.size(); i++) {
402  const FPR& numNew = a_newNumParticles[i];
403  const FPR& numOld = a_oldNumParticles[i];
404 
405  if (numNew < (FPR)0) {
406  MayDay::Warning("ItoKMCPhysics::reconcileParticles - new number of particles is < 0 (overflow issue?)");
407  }
408  else if ((long long)numNew < 0LL) {
409  MayDay::Warning("ItoKMCPhysics::reconcileParticles - integer overflow!");
410  }
411 
412  if (numOld < 0) {
413  MayDay::Warning("ItoKMCPhysics::reconcileParticles - old number of particles is < 0");
414  }
415  else if ((long long)numOld < 0LL) {
416  MayDay::Warning("ItoKMCPhysics::reconcileParticles - integer overflow for old particles!");
417  }
418  }
419  }
420 
421  for (int i = 0; i < a_particles.size(); i++) {
422  const long long diff = (long long)(a_newNumParticles[i] - a_oldNumParticles[i]);
423 
424  if (diff > 0LL) {
425  // Adding particles, which is fairly simple. Just choose weights for the particles and go.
426  const std::vector<long long> particleWeights =
427  ParticleManagement::partitionParticleWeights(diff, (long long)m_maxNewParticles);
428 
429  for (const auto& w : particleWeights) {
430  RealVect x = RealVect::Zero;
431 
432  // Figure out where to place the particles.
433  switch (m_particlePlacement) {
434  case ParticlePlacement::Random: {
435  x = Random::randomPosition(a_cellPos, a_lo, a_hi, a_bndryCentroid, a_bndryNormal, a_dx, a_kappa);
436 
437  break;
438  }
439  case ParticlePlacement::Centroid: {
440  x = a_cellPos + a_centroidPos * a_dx;
441 
442  break;
443  }
444  default: {
445  MayDay::Error("ItoKMCPhysics::reconcileParticles - logic bust");
446 
447  break;
448  }
449  }
450 
451  a_particles[i]->add(ItoParticle(1.0 * w, x));
452  }
453  }
454  else if (diff < 0LL) {
455  // Removing particles is a bit more difficult because we need to manipulate weights.
456  this->removeParticles(*a_particles[i], -diff);
457  }
458  }
459 }

461 inline void
462 ItoKMCPhysics::removeParticles(List<ItoParticle>& a_particles, const long long a_numParticlesToRemove) const
463 {
464  CH_TIME("ItoKMCPhysics::removeParticles");
465 
466  constexpr long long zero = 0LL;
467 
468  CH_assert(m_isDefined);
469  CH_assert(a_numParticlesToRemove >= zero);
470 
471  // Quick lambda for getting total particle weight. Used for debugging.
472  auto getTotalWeight = [&]() -> long long {
473  long long W = zero;
474 
475  for (ListIterator<ItoParticle> lit(a_particles); lit.ok(); ++lit) {
476  W += (long long)lit().weight();
477 
478  if (lit().weight() < 1.0) {
479  MayDay::Error("ItoKMCPhysics::removeParticles -- bad particle mass!");
480  }
481  }
482 
483  return W;
484  };
485 
486  if (a_numParticlesToRemove > zero) {
487 
488  // For debugging only.
489  long long totalWeightBefore = 0;
490  long long totalWeightAfter = 0;
491 
492  // Debug hook, compute the total particle weight before we start removing weights.
493  if (m_debug) {
494  totalWeightBefore = getTotalWeight();
495 
496  if (totalWeightBefore < a_numParticlesToRemove) {
497  MayDay::Error("ItoKMCPhysics::removeParticles: logic bust (trying to remove too many particles)");
498  }
499  }
500 
501  // Remove physical particles.
502  ParticleManagement::removePhysicalParticles(a_particles, a_numParticlesToRemove);
503 
504  // Remove particles with too low weight.
505  ParticleManagement::deleteParticles(a_particles, std::numeric_limits<Real>::min());
506 
507  // Debug hook, make sure that particle weights are > 0 AND we've removed the desired
508  // particle weight.
509  if (m_debug) {
510  totalWeightAfter = getTotalWeight();
511 
512  const long long errDiff = std::abs(totalWeightBefore - totalWeightAfter) - a_numParticlesToRemove;
513  if (std::abs(errDiff) != zero) {
514 
515  pout() << "ItoKMCPhysics::removeParticles: Total weight before = " << totalWeightBefore << endl;
516  pout() << "ItoKMCPhysics::removeParticles: Total weight after = " << totalWeightAfter << endl;
517  pout() << "ItoKMCPhysics::removeParticles: Should have removed = " << a_numParticlesToRemove << endl;
518  pout() << "ItoKMCPhysics::removeParticles: Error = " << errDiff << endl;
519 
520  MayDay::Abort("ItoKMCPhysics::removeParticles - incorrect mass removed");
521  }
522  }
